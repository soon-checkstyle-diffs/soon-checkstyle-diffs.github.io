112     private void doElevationTest(final double minElevation,
113                                  final AbsoluteDate start, final AbsoluteDate end,
114                                  final int expectedEvents, final boolean sameSign) throws OrekitException {
115 
116         final ElevationExtremumDetector raw =
117                 new ElevationExtremumDetector(0.001, 1.e-6, new TopocentricFrame(earth, gp, "test")).
118                 withHandler(new ContinueOnEvent<ElevationExtremumDetector>());
119         final EventEnablingPredicateFilter<ElevationExtremumDetector> aboveGroundElevationDetector =
120                 new EventEnablingPredicateFilter<ElevationExtremumDetector>(raw,
121                                 new EnablingPredicate<ElevationExtremumDetector>() {
122                                     public boolean eventIsEnabled(final SpacecraftState state,
123                                                                   final ElevationExtremumDetector eventDetector,
124                                                                   final double g) throws OrekitException {
125                                         return eventDetector.getElevation(state) > minElevation;
126                                     }
127                 }).withMaxCheck(60.0);
128 
129         Assert.assertEquals(0.001, raw.getMaxCheckInterval(), 1.0e-15);
130         Assert.assertEquals(60.0, aboveGroundElevationDetector.getMaxCheckInterval(), 1.0e-15);
131         Assert.assertEquals(1.0e-6, aboveGroundElevationDetector.getThreshold(), 1.0e-15);
132         Assert.assertEquals(AbstractDetector.DEFAULT_MAX_ITER, aboveGroundElevationDetector.getMaxIterationCount());
133 
134 
135         Propagator propagator =
136             new EcksteinHechlerPropagator(orbit,
137                                           Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
138                                           Constants.EIGEN5C_EARTH_MU,
139                                           Constants.EIGEN5C_EARTH_C20,
140                                           Constants.EIGEN5C_EARTH_C30,
141                                           Constants.EIGEN5C_EARTH_C40,
142                                           Constants.EIGEN5C_EARTH_C50,
143                                           Constants.EIGEN5C_EARTH_C60);
144 
145         EventsLogger logger = new EventsLogger();
146         propagator.addEventDetector(logger.monitorDetector(aboveGroundElevationDetector));
147 
148         propagator.propagate(start, end);
149         for (LoggedEvent e : logger.getLoggedEvents()) {
150             final double eMinus = raw.getElevation(e.getState().shiftedBy(-10.0));
151             final double e0     = raw.getElevation(e.getState());
152             final double ePlus  = raw.getElevation(e.getState().shiftedBy(+10.0));
153             Assert.assertTrue(e0 > eMinus);
154             Assert.assertTrue(e0 > ePlus);
155             Assert.assertTrue(e0 > minElevation);
156         }
157         Assert.assertEquals(expectedEvents, logger.getLoggedEvents().size());
158 
159         propagator.clearEventsDetectors();
160         double g1Raw = raw.g(propagator.propagate(orbit.getDate().shiftedBy(18540.0)));
161         double g2Raw = raw.g(propagator.propagate(orbit.getDate().shiftedBy(18624.0)));
162         double g1 = aboveGroundElevationDetector.g(propagator.propagate(orbit.getDate().shiftedBy(18540.0)));
163         double g2 = aboveGroundElevationDetector.g(propagator.propagate(orbit.getDate().shiftedBy(18624.0)));
164         Assert.assertTrue(g1Raw > 0);
165         Assert.assertTrue(g2Raw < 0);
166         if (sameSign) {
167             Assert.assertTrue(g1 > 0);
168             Assert.assertTrue(g2 < 0);
169         } else {
170             Assert.assertTrue(g1 < 0);
171             Assert.assertTrue(g2 > 0);
172         }
173 
174     }

176     @Test
177     public void testResetState() throws OrekitException {
178         final List<AbsoluteDate> reset = new ArrayList<AbsoluteDate>();
179         DateDetector raw = new DateDetector(orbit.getDate().shiftedBy(3600.0)).
180                         withMaxCheck(1000.0).
181                         withHandler(new EventHandler<DateDetector>() {
182                             public SpacecraftState resetState(DateDetector detector,
183                                                               SpacecraftState oldState) {
184                                 reset.add(oldState.getDate());
185                                 return oldState;
186                             }
187                             public Action eventOccurred(SpacecraftState s,
188                                                         DateDetector detector,
189                                                         boolean increasing) {
190                                 return Action.RESET_STATE;
191                             }
192                         });
193         for (int i = 2; i < 10; ++i) {
194             raw.addEventDate(orbit.getDate().shiftedBy(i * 3600.0));
195         }
196         EventEnablingPredicateFilter<DateDetector> filtered =
197                         new EventEnablingPredicateFilter<DateDetector>(raw, new EnablingPredicate<DateDetector>() {
198                             public boolean eventIsEnabled(SpacecraftState state,
199                                                           DateDetector eventDetector,
200                                                           double g) {
201                                 return state.getDate().durationFrom(orbit.getDate()) > 20000.0;
202                             }
203                         });
204         Propagator propagator = new KeplerianPropagator(orbit);
205         EventsLogger logger = new EventsLogger();
206         propagator.addEventDetector(logger.monitorDetector(filtered));
207         propagator.propagate(orbit.getDate().shiftedBy(Constants.JULIAN_DAY));
208         List<LoggedEvent> events = logger.getLoggedEvents();
209         Assert.assertEquals(4, events.size());
210         Assert.assertEquals(6 * 3600, events.get(0).getState().getDate().durationFrom(orbit.getDate()), 1.0e-6);
211         Assert.assertEquals(7 * 3600, events.get(1).getState().getDate().durationFrom(orbit.getDate()), 1.0e-6);
212         Assert.assertEquals(8 * 3600, events.get(2).getState().getDate().durationFrom(orbit.getDate()), 1.0e-6);
213         Assert.assertEquals(9 * 3600, events.get(3).getState().getDate().durationFrom(orbit.getDate()), 1.0e-6);
214         Assert.assertEquals(4, reset.size());
215         Assert.assertEquals(6 * 3600, reset.get(0).durationFrom(orbit.getDate()), 1.0e-6);
216         Assert.assertEquals(7 * 3600, reset.get(1).durationFrom(orbit.getDate()), 1.0e-6);
217         Assert.assertEquals(8 * 3600, reset.get(2).durationFrom(orbit.getDate()), 1.0e-6);
218         Assert.assertEquals(9 * 3600, reset.get(3).durationFrom(orbit.getDate()), 1.0e-6);
219         
220     }

222     @Test
223     public void testExceedHistoryForward() throws OrekitException, IOException {
224         final double period = 900.0;
225 
226         // the raw detector should trigger one event at each 900s period
227         final DateDetector raw = new DateDetector(orbit.getDate().shiftedBy(-0.5 * period)).
228                                  withMaxCheck(period / 3).
229                                  withHandler(new ContinueOnEvent<DateDetector>());
230         for (int i = 0; i < 300; ++i) {
231             raw.addEventDate(orbit.getDate().shiftedBy((i + 0.5) * period));
232         }
233 
234         // in fact, we will filter out half of these events, so we get only one event every 2 periods
235         final EventEnablingPredicateFilter<DateDetector> filtered =
236                         new EventEnablingPredicateFilter<DateDetector>(raw, new EnablingPredicate<DateDetector>() {
237                             public boolean eventIsEnabled(SpacecraftState state,
238                                                           DateDetector eventDetector,
239                                                           double g) {
240                                 double nbPeriod = state.getDate().durationFrom(orbit.getDate()) / period;
241                                 return ((int) FastMath.floor(nbPeriod)) % 2 == 1;
242                             }
243                         });
244         Propagator propagator = new KeplerianPropagator(orbit);
245         EventsLogger logger = new EventsLogger();
246         propagator.addEventDetector(logger.monitorDetector(filtered));
247         propagator.propagate(orbit.getDate().shiftedBy(301 * period));
248         List<LoggedEvent> events = logger.getLoggedEvents();
249 
250         // 300 periods, 150 events as half of them are filtered out
251         Assert.assertEquals(150, events.size());
252 
253         // as we have encountered a lot of enabling status changes, we exceeded the internal history
254         // if we try to display again the filtered g function for dates far in the past,
255         // we will not see the zero crossings anymore, they have been lost
256         propagator.clearEventsDetectors();
257         for (double dt = 5000.0; dt < 10000.0; dt += 3.0) {
258             double filteredG = filtered.g(propagator.propagate(orbit.getDate().shiftedBy(dt)));
259             Assert.assertTrue(filteredG < 0.0);
260         }
261 
262         // on the other hand, if we try to display again the filtered g function for past dates
263         // that are still inside the history, we still see the zero crossings
264         for (double dt = 195400.0; dt < 196200.0; dt += 3.0) {
265             double filteredG = filtered.g(propagator.propagate(orbit.getDate().shiftedBy(dt)));
266             if (dt < 195750) {
267                 Assert.assertTrue(filteredG > 0.0);
268             } else {
269                 Assert.assertTrue(filteredG < 0.0);
270             }
271         }
272 
273     }

275     @Test
276     public void testExceedHistoryBackward() throws OrekitException, IOException {
277         final double period = 900.0;
278 
279         // the raw detector should trigger one event at each 900s period
280         final DateDetector raw = new DateDetector(orbit.getDate().shiftedBy(+0.5 * period)).
281                                  withMaxCheck(period / 3).
282                                  withHandler(new ContinueOnEvent<DateDetector>());
283         for (int i = 0; i < 300; ++i) {
284             raw.addEventDate(orbit.getDate().shiftedBy(-(i + 0.5) * period));
285         }
286 
287         // in fact, we will filter out half of these events, so we get only one event every 2 periods
288         final EventEnablingPredicateFilter<DateDetector> filtered =
289                         new EventEnablingPredicateFilter<DateDetector>(raw, new EnablingPredicate<DateDetector>() {
290                             public boolean eventIsEnabled(SpacecraftState state,
291                                                           DateDetector eventDetector,
292                                                           double g) {
293                                 double nbPeriod = orbit.getDate().durationFrom(state.getDate()) / period;
294                                 return ((int) FastMath.floor(nbPeriod)) % 2 == 1;
295                             }
296                         });
297         Propagator propagator = new KeplerianPropagator(orbit);
298         EventsLogger logger = new EventsLogger();
299         propagator.addEventDetector(logger.monitorDetector(filtered));
300         propagator.propagate(orbit.getDate().shiftedBy(-301 * period));
301         List<LoggedEvent> events = logger.getLoggedEvents();
302 
303         // 300 periods, 150 events as half of them are filtered out
304         Assert.assertEquals(150, events.size());
305 
306         // as we have encountered a lot of enabling status changes, we exceeded the internal history
307         // if we try to display again the filtered g function for dates far in the future,
308         // we will not see the zero crossings anymore, they have been lost
309         propagator.clearEventsDetectors();
310         for (double dt = -5000.0; dt > -10000.0; dt -= 3.0) {
311             double filteredG = filtered.g(propagator.propagate(orbit.getDate().shiftedBy(dt)));
312             Assert.assertTrue(filteredG < 0.0);
313         }
314 
315         // on the other hand, if we try to display again the filtered g function for future dates
316         // that are still inside the history, we still see the zero crossings
317         for (double dt = -195400.0; dt > -196200.0; dt -= 3.0) {
318             double filteredG = filtered.g(propagator.propagate(orbit.getDate().shiftedBy(dt)));
319             if (dt < -195750) {
320                 Assert.assertTrue(filteredG < 0.0);
321             } else {
322                 Assert.assertTrue(filteredG > 0.0);
323             }
324         }

354     @Test
355     public void testSerializable()
356         throws IOException, IllegalArgumentException, IllegalAccessException,
357                ClassNotFoundException, NoSuchFieldException, SecurityException {
358         final EventEnablingPredicateFilter<DateDetector> filter =
359                         new EventEnablingPredicateFilter<DateDetector>(new DateDetector(AbsoluteDate.J2000_EPOCH),
360                                                                        new  DummySerializablePredicate());
361         ByteArrayOutputStream bos = new ByteArrayOutputStream();
362         ObjectOutputStream    oos = new ObjectOutputStream(bos);
363         oos.writeObject(filter);
364 
365         Assert.assertTrue(bos.size() >  900);
366         Assert.assertTrue(bos.size() < 1000);
367 
368         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
369         ObjectInputStream     ois = new ObjectInputStream(bis);
370         @SuppressWarnings("unchecked")
371         EventEnablingPredicateFilter<DateDetector> deserialized  =
372                         (EventEnablingPredicateFilter<DateDetector>) ois.readObject();
373         Field enabler = EventEnablingPredicateFilter.class.getDeclaredField("enabler");
374         enabler.setAccessible(true);
375         Assert.assertEquals(DummySerializablePredicate.class, enabler.get(deserialized).getClass());
376     }
