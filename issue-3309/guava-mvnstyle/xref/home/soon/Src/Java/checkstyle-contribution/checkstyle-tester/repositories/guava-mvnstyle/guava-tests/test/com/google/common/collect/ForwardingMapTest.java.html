56      private final Map<K, V> backingMap;
57  
58      StandardImplForwardingMap(Map<K, V> backingMap) {
59        this.backingMap = backingMap;
60      }
61  
62      @Override protected Map<K, V> delegate() {
63        return backingMap;
64      }
65  
66      @Override public boolean containsKey(Object key) {
67        return standardContainsKey(key);
68      }
69  
70      @Override public boolean containsValue(Object value) {
71        return standardContainsValue(value);
72      }
73  
74      @Override public void putAll(Map<? extends K, ? extends V> map) {
75        standardPutAll(map);
76      }
77  
78      @Override public V remove(Object object) {
79        return standardRemove(object);
80      }
81  
82      @Override public boolean equals(Object object) {
83        return standardEquals(object);
84      }
85  
86      @Override public int hashCode() {
87        return standardHashCode();
88      }
89  
90      @Override public Set<K> keySet() {
91        return new StandardKeySet();
92      }
93  
94      @Override public Collection<V> values() {
95        return new StandardValues();
96      }
97  
98      @Override public String toString() {
99        return standardToString();
100     }
101 
102     @Override public Set<Entry<K, V>> entrySet() {
103       return new StandardEntrySet() {
104         @Override
105         public Iterator<Entry<K, V>> iterator() {
106           return delegate()
107               .entrySet()
108               .iterator();
109         }
110       };
111     }
112 
113     @Override public void clear() {
114       standardClear();
115     }
116 
117     @Override public boolean isEmpty() {
118       return standardIsEmpty();
119     }

122   public static Test suite() {
123     TestSuite suite = new TestSuite();
124 
125     suite.addTestSuite(ForwardingMapTest.class);
126     suite.addTest(MapTestSuiteBuilder.using(new TestStringMapGenerator() {
127 
128       @Override protected Map<String, String> create(
129           Entry<String, String>[] entries) {
130         Map<String, String> map = Maps.newLinkedHashMap();
131         for (Entry<String, String> entry : entries) {
132           map.put(entry.getKey(), entry.getValue());
133         }
134         return new StandardImplForwardingMap<String, String>(map);
135       }
136 
137     }).named("ForwardingMap[LinkedHashMap] with standard implementations")
138         .withFeatures(CollectionSize.ANY, MapFeature.ALLOWS_NULL_VALUES,
139             MapFeature.ALLOWS_NULL_KEYS, MapFeature.ALLOWS_ANY_NULL_QUERIES,
140             MapFeature.GENERAL_PURPOSE,
141             CollectionFeature.SUPPORTS_ITERATOR_REMOVE, CollectionFeature.KNOWN_ORDER)
142         .createTestSuite());
143     suite.addTest(MapTestSuiteBuilder.using(new TestStringMapGenerator() {
144 
145       @Override protected Map<String, String> create(
146           Entry<String, String>[] entries) {
147         ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
148         for (Entry<String, String> entry : entries) {
149           builder.put(entry.getKey(), entry.getValue());
150         }
151         return new StandardImplForwardingMap<String, String>(builder.build());
152       }
153 
154     }).named("ForwardingMap[ImmutableMap] with standard implementations")
155         .withFeatures(
156             CollectionSize.ANY, MapFeature.REJECTS_DUPLICATES_AT_CREATION,
157             MapFeature.ALLOWS_ANY_NULL_QUERIES,
158             CollectionFeature.KNOWN_ORDER)

183   public void testStandardEntrySet() throws InvocationTargetException {
184     @SuppressWarnings("unchecked")
185     final Map<String, Boolean> map = mock(Map.class);
186 
187     Map<String, Boolean> forward =
188         new ForwardingMap<String, Boolean>() {
189           @Override
190           protected Map<String, Boolean> delegate() {
191             return map;
192           }
193 
194           @Override
195           public Set<Entry<String, Boolean>> entrySet() {
196             return new StandardEntrySet() {
197               @Override
198               public Iterator<Entry<String, Boolean>> iterator() {
199                 return Iterators.emptyIterator();
200               }
201             };
202           }
203         };
204     callAllPublicMethods(Set.class, forward.entrySet());
205 
206     // These are the methods specified by StandardEntrySet
207     verify(map, atLeast(0)).clear();
208     verify(map, atLeast(0)).containsKey(anyObject());
209     verify(map, atLeast(0)).get(anyObject());
210     verify(map, atLeast(0)).isEmpty();
211     verify(map, atLeast(0)).remove(anyObject());
212     verify(map, atLeast(0)).size();
213     verifyNoMoreInteractions(map);
214   }

293   private static <K, V> Map<K, V> wrap(final Map<K, V> delegate) {
294     return new ForwardingMap<K, V>() {
295       @Override protected Map<K, V> delegate() {
296         return delegate;
297       }
298     };
299   }

301   private static Object getDefaultValue(Class<?> returnType) {
302     Object defaultValue = ArbitraryInstances.get(returnType);
303     if (defaultValue != null) {
304       return defaultValue;
305     }
306     if ("java.util.function.Predicate".equals(returnType.getCanonicalName())
307         || ("java.util.function.Consumer".equals(returnType.getCanonicalName()))) {
308       // Generally, methods that accept java.util.function.* instances
309       // don't like to get null values.  We generate them dynamically
310       // using Proxy so that we can have Java 7 compliant code.
311       return Reflection.newProxy(returnType, new AbstractInvocationHandler() {
312         @Override public Object handleInvocation(Object proxy, Method method, Object[] args) {
313           // Crude, but acceptable until we can use Java 8.  Other
314           // methods have default implementations, and it is hard to
315           // distinguish.
316           if ("test".equals(method.getName()) || "accept".equals(method.getName())) {
317             return getDefaultValue(method.getReturnType());
318           }
319           throw new IllegalStateException("Unexpected " + method + " invoked on " + proxy);
320         }
321       });
322     } else {
323       return null;
324     }
325   }

327   private static <T> void callAllPublicMethods(Class<T> theClass, T object)
328       throws InvocationTargetException {
329     for (Method method : theClass.getMethods()) {
330       Class<?>[] parameterTypes = method.getParameterTypes();
331       Object[] parameters = new Object[parameterTypes.length];
332       for (int i = 0; i < parameterTypes.length; i++) {
333         parameters[i] = getDefaultValue(parameterTypes[i]);
334       }
335       try {
336         try {
337           method.invoke(object, parameters);
338         } catch (InvocationTargetException ex) {
339           try {
340             throw ex.getCause();
341           } catch (UnsupportedOperationException unsupported) {
342             // this is a legit exception
343           }
344         }
345       } catch (Throwable cause) {
346         throw new InvocationTargetException(cause,
347             method + " with args: " + Arrays.toString(parameters));
348       }
349     }
