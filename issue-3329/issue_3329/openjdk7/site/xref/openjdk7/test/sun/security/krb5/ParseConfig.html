23  /*
24   * @test
25   * @bug 6319046
26   * @run main/othervm ParseConfig
27   * @summary Problem with parsing krb5.conf
28   */
29  
30  import sun.security.krb5.Config;
31  
32  public class ParseConfig {
33      public static void main(String[] args) throws Exception {
34          System.setProperty("java.security.krb5.conf", System.getProperty("test.src", ".") +"/krb5.conf");
35          Config config = Config.getInstance();
36          config.listTable();
37  
38          String sample = "kdc.example.com kdc2.example.com";
39          for ( int i = 0; i < 4; i++ ) {
40              String expected = config.getDefault("kdc", "EXAMPLE_" + i + ".COM");
41              if (!sample.equals(expected)) {
42                  throw new Exception("krb5.conf: unexpected kdc value \"" +
43                          expected + "\"");
44              }
45          }
46      }
47  }
