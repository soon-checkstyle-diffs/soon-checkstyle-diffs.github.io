11  import java.awt.*;
12  import java.awt.event.MouseEvent;
13  import java.awt.event.MouseAdapter;
14  
15  public class AcceptExtraButton extends Frame {
16      static int [] eventID = new int []{MouseEvent.MOUSE_PRESSED, MouseEvent.MOUSE_RELEASED, MouseEvent.MOUSE_CLICKED};
17  
18      public static void main(String []s){
19          AcceptExtraButton f = new AcceptExtraButton();
20          f.setSize(300, 300);
21          f.setVisible(true);
22  
23          for (int buttonId = 0; buttonId<eventID.length; buttonId++) {
24              for (int button = 0; button <= MouseInfo.getNumberOfButtons(); button++){
25                  System.out.println("button == "+button);
26                  MouseEvent me = new MouseEvent(f,
27                                                 eventID[buttonId],
28                                                 System.currentTimeMillis(),
29                                                 0, //MouseEvent.BUTTON1_DOWN_MASK, modifiers
30                                                 100, 100, // x, y
31                                                 150, 150, // x, y on screen
32                                                 1,        //clickCount
33                                                 false,              //popupTrigger
34                                                 button );// MouseEvent.NOBUTTON : button
35  
36                  System.out.println("dispatching >>>"+me);
37                  f.dispatchEvent( ( AWTEvent )me );
38              }
39          }
40          MouseAdapter ma1 = new MouseAdapter() {
41                  public void mousePressed(MouseEvent e) {
42                      System.out.println("PRESSED "+e);
43                  }
44                  public void mouseReleased(MouseEvent e) {
45                      System.out.println("RELEASED "+e);
46                  }
47                  public void mouseClicked(MouseEvent e) {
48                      System.out.println("CLICKED "+e);
49                  }
50              };
51          f.addMouseListener(ma1);
52      }
53  }
