1   package org.infinispan.interceptors.xsite;
2   
3   import org.infinispan.commands.tx.CommitCommand;
4   import org.infinispan.commands.tx.PrepareCommand;
5   import org.infinispan.context.impl.TxInvocationContext;
6   import org.infinispan.remoting.transport.BackupResponse;
7   
8   /**
9    * Handles x-site data backups for pessimistic transactional caches.
10   *
11   * @author Mircea Markus
12   * @since 5.2
13   */
14  public class PessimisticBackupInterceptor extends BaseBackupInterceptor {
15  
16     @Override
17     public Object visitPrepareCommand(TxInvocationContext ctx, PrepareCommand command) throws Throwable {
18        //if this is an "empty" tx no point replicating it to other clusters
19        if (!shouldInvokeRemoteTxCommand(ctx))
20           return super.visitPrepareCommand(ctx, command);
21  
22        boolean isTxFromRemoteSite = isTxFromRemoteSite( command.getGlobalTransaction() );
23        if (isTxFromRemoteSite) {
24           return invokeNextInterceptor(ctx, command);
25        }
26  
27        BackupResponse backupResponse = backupSender.backupPrepare(command);
28        Object result = invokeNextInterceptor(ctx, command);
29        backupSender.processResponses(backupResponse, command, ctx.getTransaction());
30        return result;
31     }
32  
33     @Override
34     public Object visitCommitCommand(TxInvocationContext ctx, CommitCommand command) throws Throwable {
35        //for pessimistic transaction we don't do a 2PC (as we already own the remote lock) but just
36        //a 1PC
37        throw new IllegalStateException("This should never happen!");
38     }
39  }
