24  import java.io.*;
25  import java.net.*;
26  import java.util.jar.*;
27  
28  /*
29   * ResourceBundle from jar not found if jar exists in path
30   * which has symbol !
31   *
32   * 3 votes
33   *
34   * Note: Execute "real" test in separate vm instance so that any locks
35   * held on files will be released when this separate vm exits and the
36   * invoking vm can clean up if necessary.
37   */
38  public class TestBug4361044 extends JarTest
39  {
40      public void run(String[] args) throws Exception {
41          if (args.length == 0 ) {  // execute the test in another vm.
42              System.out.println("Test: " + getClass().getName());
43              Process process = Runtime.getRuntime().exec(javaCmd + " TestBug4361044 -test");
44  
45              BufferedReader isReader = new BufferedReader(new InputStreamReader(process.getInputStream()));
46              BufferedReader esReader = new BufferedReader(new InputStreamReader(process.getErrorStream()));
47              Redirector outRedirector = new Redirector(isReader, System.out);
48              Redirector errRedirector = new Redirector(esReader, System.err);
49  
50              (new Thread(outRedirector)).start();
51              (new Thread(errRedirector)).start();
52  
53              process.waitFor();
54  
55              // Delete any remaining files from the test
56              File testDir = new File(tmpdir + File.separator + getClass().getName());
57              deleteRecursively(testDir);
58  
59              if (outRedirector.getHasReadData() || errRedirector.getHasReadData())
60                  throw new RuntimeException("Failed: No output should have been received from the process");
61  
62          } else {   // run the test.
63              File tmp = createTempDir();
64              try {
65                  File dir = new File(tmp, "dir!name");
66                  dir.mkdir();
67                  File testFile = copyResource(dir, "jar1.jar");
68                  URL[] urls = new URL[1];
69                  urls[0] = new URL("jar:" + testFile.toURL() + "!/");
70                  URLClassLoader loader = new URLClassLoader(urls);
71                  loader.loadClass("jar1.LoadResourceBundle").newInstance();
72              } finally {
73                  deleteRecursively(tmp);
74              }
75          }
76      }
77  
78      public static void main(String[] args) throws Exception {
79          new TestBug4361044().run(args);
80      }
81  }
