25  package sun.awt.windows;
26  
27  import java.awt.*;
28  import java.awt.peer.*;
29  import java.awt.event.ActionEvent;
30  import java.awt.event.KeyEvent;
31  
32  final class WButtonPeer extends WComponentPeer implements ButtonPeer {
33  
34      static {
35          initIDs();
36      }
37  
38      // ComponentPeer overrides
39  
40      public Dimension getMinimumSize() {
41          FontMetrics fm = getFontMetrics(((Button)target).getFont());
42          String label = ((Button)target).getLabel();
43          if ( label == null ) {
44              label = "";
45          }
46          return new Dimension(fm.stringWidth(label) + 14,
47                               fm.getHeight() + 8);
48      }
49      public boolean isFocusable() {
50          return true;
51      }
52  
53      // ButtonPeer implementation
54  
55      public native void setLabel(String label);
56  
57      // Toolkit & peer internals
58  
59      WButtonPeer(Button target) {
60          super(target);
61      }
62  
63      native void create(WComponentPeer peer);
64  
65      // native callbacks
66  
67      // NOTE: This is called on the privileged toolkit thread. Do not
68      //       call directly into user code using this thread!
69      public void handleAction(final long when, final int modifiers) {
70          // Fixed 5064013: the InvocationEvent time should be equals
71          // the time of the ActionEvent
72          WToolkit.executeOnEventHandlerThread(target, new Runnable() {
73              public void run() {
74                  postEvent(new ActionEvent(target, ActionEvent.ACTION_PERFORMED,
75                                            ((Button)target).getActionCommand(),
76                                            when, modifiers));
77              }
78          }, when);
79      }
80  
81  
82      public boolean shouldClearRectBeforePaint() {
83          return false;
84      }
85  
86      /**
87       * Initialize JNI field and method IDs
88       */
89      private static native void initIDs();
90  
91      public boolean handleJavaKeyEvent(KeyEvent e) {
92           switch (e.getID()) {
93              case KeyEvent.KEY_RELEASED:
94                  if (e.getKeyCode() == KeyEvent.VK_SPACE){
95                      handleAction(e.getWhen(), e.getModifiers());
96                  }
97              break;
98           }
99           return false;
100     }
101 }
