63      public void checkForUpdates(Collection<Plugin> plugins, final boolean force) {
64          if (updateChecksGloballyDisabled()) {
65              dfc.pluginUpdateCheckComplete(pluginUpdates, force);
66              return;
67          }
68  
69          URI redirectUri = getRedirectURL(force);
70  
71          final CountDownLatch latch;
72          if (redirectUri != null) {
73              latch = new CountDownLatch(1);
74              startUpdateCheckThread(redirectUri, plugins, latch);
75          } else {
76              MultiMap<URI,Plugin> pluginsByUrl = new MultiMap<URI, Plugin>(HashSet.class);
77              for (Plugin plugin : plugins) {
78                  URI uri = plugin.getUpdateUrl();
79                  if (uri == null) {
80                      logError(Level.FINE, "Not checking for updates for " + plugin.getShortDescription()
81                              + " - no update-url attribute in plugin XML file");
82                      continue;
83                  }
84                  pluginsByUrl.add(uri, plugin);
85              }
86              latch = new CountDownLatch(pluginsByUrl.keySet().size());
87              for (URI uri : pluginsByUrl.keySet()) {
88                  startUpdateCheckThread(uri, pluginsByUrl.get(uri), latch);
89              }
90          }
91  
92          waitForCompletion(latch, force);
93      }

95      public @CheckForNull URI getRedirectURL(final boolean force) {
96          String redirect = dfc.getGlobalOption(KEY_REDIRECT_ALL_UPDATE_CHECKS);
97          String sysprop = System.getProperty("findbugs.redirectUpdateChecks");
98          if (sysprop != null) {
99              redirect = sysprop;
100         }
101         Plugin setter = dfc.getGlobalOptionSetter(KEY_REDIRECT_ALL_UPDATE_CHECKS);
102         URI redirectUri = null;
103         String pluginName = setter == null ? "<unknown plugin>" : setter.getShortDescription();
104         if (redirect != null && !"".equals(redirect.trim())) {
105             try {
106                 redirectUri = new URI(redirect);
107                 logError(Level.INFO, "Redirecting all plugin update checks to " + redirectUri + " (" + pluginName + ")");
108 
109             } catch (URISyntaxException e) {
110                 String error = "Invalid update check redirect URI in " + pluginName + ": " + redirect;
111                 logError(Level.SEVERE, error);
112                 dfc.pluginUpdateCheckComplete(pluginUpdates, force);
113                 throw new IllegalStateException(error);
114             }
115         }
116         return redirectUri;
117     }

119     private long dontWarnAgainUntil() {
120         Preferences prefs = Preferences.userNodeForPackage(UpdateChecker.class);
121 
122         String oldSeen = prefs.get("last-plugin-update-seen", "");
123         if (oldSeen == null || "".equals(oldSeen)) {
124             return 0;
125         }
126         try {
127             return Long.parseLong(oldSeen) + DONT_REMIND_WINDOW;
128         } catch (Exception e) {
129             return 0;
130         }
131     }
132     static final long DONT_REMIND_WINDOW = 3L*24*60*60*1000;
133     public boolean updatesHaveBeenSeenBefore(Collection<UpdateChecker.PluginUpdate> updates) {
134         long now = System.currentTimeMillis();
135         Preferences prefs = Preferences.userNodeForPackage(UpdateChecker.class);
136         String oldHash = prefs.get("last-plugin-update-hash", "");
137 
138         String newHash = Integer.toString(buildPluginUpdateHash(updates));
139         if (oldHash.equals(newHash) && dontWarnAgainUntil() > now) {
140             LOGGER.fine("Skipping update dialog because these updates have been seen before");
141             return true;
142         }
143         prefs.put("last-plugin-update-hash", newHash);
144         prefs.put("last-plugin-update-seen", Long.toString(now));
145         return false;
146     }

148     private int buildPluginUpdateHash(Collection<UpdateChecker.PluginUpdate> updates) {
149         HashSet<String> builder = new HashSet<String>();
150         for (UpdateChecker.PluginUpdate update : updates) {
151             builder.add( update.getPlugin().getPluginId() + update.getVersion());
152         }
153         return builder.hashCode();
154     }

156     private void waitForCompletion(final CountDownLatch latch, final boolean force) {
157         Util.runInDameonThread(new Runnable() {
158             @Override
159             public void run() {
160                 if (DEBUG) {
161                     System.out.println("Checking for version updates");
162                 }
163                 try {
164                     if (! latch.await(15, TimeUnit.SECONDS)) {
165                         logError(Level.INFO, "Update check timed out");
166                     }
167                     dfc.pluginUpdateCheckComplete(pluginUpdates, force);
168                 } catch (Exception ignored) {
169                     assert true;
170                 }
171 
172             }
173         }, "Plugin update checker");
174     }

180     public String getPluginThatDisabledUpdateChecks() {
181         String disable = dfc.getGlobalOption(KEY_DISABLE_ALL_UPDATE_CHECKS);
182         Plugin setter = dfc.getGlobalOptionSetter(KEY_DISABLE_ALL_UPDATE_CHECKS);
183         String pluginName = setter == null ? "<unknown plugin>" : setter.getShortDescription();
184         String disablingPlugin = null;
185         if ("true".equalsIgnoreCase(disable)) {
186             logError(Level.INFO, "Skipping update checks due to " + KEY_DISABLE_ALL_UPDATE_CHECKS + "=true set by "
187                     + pluginName);
188             disablingPlugin = pluginName;
189         } else if (disable != null && !"false".equalsIgnoreCase(disable)) {
190             String error = "Unknown value '" + disable + "' for " + KEY_DISABLE_ALL_UPDATE_CHECKS + " in " + pluginName;
191             logError(Level.SEVERE, error);
192             throw new IllegalStateException(error);
193         }
194         return disablingPlugin;
195     }

197     private void startUpdateCheckThread(final URI url, final Collection<Plugin> plugins, final CountDownLatch latch) {
198         if (url == null) {
199             logError(Level.INFO, "Not checking for plugin updates w/ blank URL: " + getPluginNames(plugins));
200             return;
201         }
202         final String entryPoint = getEntryPoint();
203         if ((entryPoint.contains("edu.umd.cs.findbugs.FindBugsTestCase")
204                 || entryPoint.contains("edu.umd.cs.findbugs.cloud.appEngine.AbstractWebCloudTest"))
205                 && ("http".equals(url.getScheme()) || "https".equals(url.getScheme()))) {
206             LOGGER.fine("Skipping update check because we're running in FindBugsTestCase and using "
207                     + url.getScheme());
208             return;
209         }
210         Runnable r = new Runnable() {
211             @Override
212             public void run() {
213                 try {
214                     actuallyCheckforUpdates(url, plugins, entryPoint);
215                 } catch (Exception e) {
216                     if (e instanceof IllegalStateException && e.getMessage().contains("Shutdown in progress")) {
217                         return;
218                     }
219                     logError(e, "Error doing update check at " + url);
220                 } finally {
221                     latch.countDown();
222                 }
223             }
224         };
225         if (DEBUG) {
226             r.run();
227         } else {
228             Util.runInDameonThread(r, "Check for updates");
229         }
230     }

233     /** protected for testing */
234     protected void actuallyCheckforUpdates(URI url, Collection<Plugin> plugins, String entryPoint) throws IOException {
235         LOGGER.fine("Checking for updates at " + url + " for " + getPluginNames(plugins));
236         if (DEBUG) {
237             System.out.println(url);
238         }
239         HttpURLConnection conn = (HttpURLConnection) url.toURL().openConnection();
240         conn.setDoInput(true);
241         conn.setDoOutput(true);
242         conn.setRequestMethod("POST");
243         conn.connect();
244         OutputStream out = conn.getOutputStream();
245         writeXml(out, plugins, entryPoint, true);
246         // for debugging:
247         if (DEBUG) {
248             System.out.println("Sending");
249             writeXml(System.out, plugins, entryPoint, false);
250         }
251         int responseCode = conn.getResponseCode();
252         if (responseCode != 200) {
253             logError(SystemProperties.ASSERTIONS_ENABLED ? Level.WARNING : Level.FINE,
254                     "Error checking for updates at " + url + ": "
255                             + responseCode + " - " + conn.getResponseMessage());
256         } else {
257             parseUpdateXml(url, plugins, conn.getInputStream());
258         }
259         conn.disconnect();
260     }

263     @SuppressFBWarnings("OBL_UNSATISFIED_OBLIGATION")
264     protected final void writeXml(OutputStream out, Collection<Plugin> plugins, String entryPoint,
265             boolean finish) throws IOException {
266         OutputStreamXMLOutput xmlOutput = new OutputStreamXMLOutput(out);
267         try {
268             xmlOutput.beginDocument();
269 
270             xmlOutput.startTag("findbugs-invocation");
271             xmlOutput.addAttribute("version", Version.RELEASE);
272             String applicationName = Version.getApplicationName();
273             if (applicationName == null || "".equals(applicationName)) {
274                 int lastDot = entryPoint.lastIndexOf('.');
275                 if (lastDot == -1) {
276                     applicationName = entryPoint;
277                 } else {
278                     applicationName = entryPoint.substring(lastDot + 1);
279                 }
280             }
281             xmlOutput.addAttribute("app-name", applicationName);
282             String applicationVersion = Version.getApplicationVersion();
283             if (applicationVersion == null) {
284                 applicationVersion = "";
285             }
286             xmlOutput.addAttribute("app-version", applicationVersion);
287             xmlOutput.addAttribute("entry-point", entryPoint);
288             xmlOutput.addAttribute("os", SystemProperties.getProperty("os.name", ""));
289             xmlOutput.addAttribute("java-version", getMajorJavaVersion());
290             Locale locale = Locale.getDefault();
291             xmlOutput.addAttribute("language", locale.getLanguage());
292             xmlOutput.addAttribute("country", locale.getCountry());
293             xmlOutput.addAttribute("uuid", getUuid());
294             xmlOutput.stopTag(false);
295             for (Plugin plugin : plugins) {
296                 xmlOutput.startTag("plugin");
297                 xmlOutput.addAttribute("id", plugin.getPluginId());
298                 xmlOutput.addAttribute("name", plugin.getShortDescription());
299                 xmlOutput.addAttribute("version", plugin.getVersion());
300                 Date date = plugin.getReleaseDate();
301                 if (date != null) {
302                     xmlOutput.addAttribute("release-date", Long.toString(date.getTime()));
303                 }
304                 xmlOutput.stopTag(true);
305             }
306 
307             xmlOutput.closeTag("findbugs-invocation");
308             xmlOutput.flush();
309         } finally {
310             if (finish) {
311                 xmlOutput.finish();
312             }
313         }
314     }

317     @SuppressWarnings({ "unchecked" })
318     void parseUpdateXml(URI url, Collection<Plugin> plugins, @WillClose
319             InputStream inputStream) {
320         try {
321             Document doc = new SAXReader().read(inputStream);
322             if (DEBUG) {
323                 StringWriter stringWriter = new StringWriter();
324                 XMLWriter xmlWriter = new XMLWriter(stringWriter);
325                 xmlWriter.write(doc);
326                 xmlWriter.close();
327                 System.out.println("UPDATE RESPONSE: " + stringWriter.toString());
328             }
329             List<Element> pluginEls =  XMLUtil.selectNodes(doc, "fb-plugin-updates/plugin");
330             Map<String, Plugin> map = new HashMap<String, Plugin>();
331             for (Plugin p : plugins) {
332                 map.put(p.getPluginId(), p);
333             }
334             for (Element pluginEl : pluginEls) {
335                 String id = pluginEl.attributeValue("id");
336                 Plugin plugin = map.get(id);
337                 if (plugin != null) {
338                     checkPlugin(pluginEl, plugin);
339                 }
340 
341             }
342         } catch (Exception e) {
343             logError(e, "Could not parse plugin version update for " + url);
344         } finally {
345             Util.closeSilently(inputStream);
346         }
347     }

349     @SuppressWarnings({"unchecked"})
350     private void checkPlugin(Element pluginEl, Plugin plugin) {
351         for (Element release : (List<Element>) pluginEl.elements("release")) {
352             checkPluginRelease(plugin, release);
353         }

356     private void checkPluginRelease(Plugin plugin, Element maxEl) {
357         @CheckForNull Date updateDate = parseReleaseDate(maxEl);
358         @CheckForNull Date installedDate = plugin.getReleaseDate();
359         if (updateDate != null && installedDate != null && updateDate.before(installedDate)) {
360             return;
361         }
362         String version = maxEl.attributeValue("version");
363         if (version.equals(plugin.getVersion())) {
364             return;
365         }
366 
367         String url = maxEl.attributeValue("url");
368         String message = maxEl.element("message").getTextTrim();
369 
370         pluginUpdates.add(new PluginUpdate(plugin, version, updateDate, url, message));
371     }

383     private @CheckForNull Date parseReleaseDate(Element releaseEl) {
384         SimpleDateFormat format = new SimpleDateFormat(PLUGIN_RELEASE_DATE_FMT);
385         String dateStr = releaseEl.attributeValue("date");
386         if (dateStr == null) {
387             return null;
388         }
389         try {
390             return format.parse(dateStr);
391         } catch (java.text.ParseException e) {
392             throw new IllegalArgumentException("Error parsing " + dateStr + " using " + PLUGIN_RELEASE_DATE_FMT, e);
393         }
394     }

396     private String getPluginNames(Collection<Plugin> plugins) {
397         String text = "";
398         boolean first = true;
399         for (Plugin plugin : plugins) {
400             text = (first ? "" : ", ") + plugin.getShortDescription();
401             first = false;
402         }
403         return text;
404     }

406     private String getEntryPoint() {
407         String lastFbClass = "<UNKNOWN>";
408         for (StackTraceElement s : Thread.currentThread().getStackTrace()) {
409             String cls = s.getClassName();
410             if (cls.startsWith("edu.umd.cs.findbugs.")) {
411                 lastFbClass = cls;
412             }
413         }
414         return lastFbClass;

420     private static synchronized String getUuid() {
421         try {
422             Preferences prefs = Preferences.userNodeForPackage(UpdateChecker.class);
423             long uuid = prefs.getLong("uuid", 0);
424             if (uuid == 0) {
425                 uuid = random.nextLong();
426                 prefs.putLong("uuid", uuid);
427             }
428             return Long.toString(uuid, 16);
429         } catch (Throwable e) {
430             return Long.toString(42, 16);
431         }
432     }

434     private String getMajorJavaVersion() {
435         String ver = SystemProperties.getProperty("java.version", "");
436         Matcher m = Pattern.compile("^\\d+\\.\\d+").matcher(ver);
437         if (m.find()) {
438             return m.group();
439         }
440         return "";
441     }

443     public static class PluginUpdate {
444         private final Plugin plugin;
445         private final String version;
446         private final @CheckForNull Date date;
447         private final @CheckForNull String url;
448         private final @Nonnull String message;
449 
450         private PluginUpdate(Plugin plugin, String version, @CheckForNull  Date date, @CheckForNull String url, @Nonnull String message) {
451             this.plugin = plugin;
452             this.version = version;
453             this.date = date;
454             this.url = url;
455             this.message = message;
456         }
457 
458         public Plugin getPlugin() {
459             return plugin;
460         }
461 
462         public String getVersion() {
463             return version;
464         }
465 
466         public @CheckForNull Date getDate() {
467             return date;
468         }
469 
470         public @CheckForNull String getUrl() {
471             return url;
472         }
473 
474         public @Nonnull String getMessage() {
475             return message;
476         }
477 
478         @Override
479         public String toString() {
480             SimpleDateFormat format = new SimpleDateFormat(PLUGIN_RELEASE_DATE_FMT);
481             StringBuilder buf = new StringBuilder();
482             String name = getPlugin().isCorePlugin() ? "FindBugs" : "FindBugs plugin " + getPlugin().getShortDescription();
483             buf.append( name + " " + getVersion() );
484             if (date == null) {
485                 buf.append(" has been released");
486             } else {
487                 buf.append(" was released " + format.format(date));
488             }
489             buf.append(
490                     " (you have " + getPlugin().getVersion()
491                     + ")");
492             buf.append("\n");
493 
494             buf.append("   " + message.replaceAll("\n", "\n   "));
495 
496             if (url != null) {
497                 buf.append("\nVisit " + url + " for details.");
498             }
499             return buf.toString();
500         }
501     }
