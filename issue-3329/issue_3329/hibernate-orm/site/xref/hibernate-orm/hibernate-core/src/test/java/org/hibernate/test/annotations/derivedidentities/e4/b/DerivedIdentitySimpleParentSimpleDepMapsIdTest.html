43  	@Test
44  	public void testOneToOneExplicitJoinColumn() throws Exception {
45  		assertTrue( SchemaUtil.isColumnPresent( "MedicalHistory", "FK", configuration() ) );
46  		assertTrue( ! SchemaUtil.isColumnPresent( "MedicalHistory", "id", configuration() ) );
47  		Person e = new Person();
48  		e.ssn = "aaa";
49  		Session s = openSession(  );
50  		s.getTransaction().begin();
51  		s.persist( e );
52  		MedicalHistory d = new MedicalHistory();
53  		d.patient = e;
54  		//d.id = "aaa"; //FIXME not needed when foreign is enabled
55  		s.persist( d );
56  		s.flush();
57  		s.clear();
58  		d = (MedicalHistory) s.get( MedicalHistory.class, d.id );
59  		assertEquals( d.id, d.patient.ssn );
60  		d.lastupdate = new Date();
61  		s.flush();
62  		s.clear();
63  		d = (MedicalHistory) s.get( MedicalHistory.class, d.id );
64  		assertNotNull( d.lastupdate );
65  		s.getTransaction().rollback();
66  		s.close();
67  	}

95  	@Test
96  	public void testExplicitlyAssignedDependentIdAttributeValue() {
97  		// even though the id is by definition generated (using the "foreign" strategy), JPA
98  		// still does allow manually setting the generated id attribute value which providers
99  		// are expected to promptly disregard :?
100 		Session s = openSession();
101 		s.beginTransaction();
102 		Person person = new Person( "123456789" );
103 		MedicalHistory medicalHistory = new MedicalHistory( "987654321", person );
104 		s.persist( person );
105 		s.persist( medicalHistory );
106 		s.getTransaction().commit();
107 		s.close();
108 
109 		// again, even though we specified an id value of "987654321" prior to persist,
110 		// Hibernate should have replaced that with the "123456789" from the associated
111 		// person
112 		assertEquals( person.ssn, medicalHistory.patient.ssn );
113 		assertEquals( person, medicalHistory.patient );
114 		assertEquals( person.ssn, medicalHistory.id );
115 
116 		s = openSession();
117 		s.beginTransaction();
118 		// Should return null...
119 		MedicalHistory separateMedicalHistory = (MedicalHistory) s.get( MedicalHistory.class, "987654321" );
120 		assertNull( separateMedicalHistory );
121 		// Now we should find it...
122 		separateMedicalHistory = (MedicalHistory) s.get( MedicalHistory.class, "123456789" );
123 		assertNotNull( separateMedicalHistory );
124 		s.getTransaction().commit();
125 		s.close();
126 
127 		s = openSession();
128 		s.beginTransaction();
129 		s.delete( medicalHistory );
130 		s.delete( person );
131 		s.getTransaction().commit();
132 		s.close();
133 	}
