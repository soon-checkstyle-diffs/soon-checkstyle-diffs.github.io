24  package org.hibernate.engine.spi;
25  
26  /**
27   * For persistence operations (INSERT, UPDATE, DELETE) what style of determining
28   * results (success/failure) is to be used.
29   *
30   * @author Steve Ebersole
31   */
32  public enum ExecuteUpdateResultCheckStyle {
33  	/**
34  	 * Do not perform checking.  Either user simply does not want checking, or is
35  	 * indicating a {@link java.sql.CallableStatement} execution in which the
36  	 * checks are being performed explicitly and failures are handled through
37  	 * propagation of {@link java.sql.SQLException}s.
38  	 */
39  	NONE( "none" ),
40  
41  	/**
42  	 * Perform row-count checking.  Row counts are the int values returned by both
43  	 * {@link java.sql.PreparedStatement#executeUpdate()} and
44  	 * {@link java.sql.Statement#executeBatch()}.  These values are checked
45  	 * against some expected count.
46  	 */
47  	COUNT( "rowcount" ),
48  
49  	/**
50  	 * Essentially the same as {@link #COUNT} except that the row count actually
51  	 * comes from an output parameter registered as part of a
52  	 * {@link java.sql.CallableStatement}.  This style explicitly prohibits
53  	 * statement batching from being used...
54  	 */
55  	PARAM( "param" );
56  
57  	private final String name;
58  
59  	private ExecuteUpdateResultCheckStyle(String name) {
60  		this.name = name;
61  	}
62  
63  	public static ExecuteUpdateResultCheckStyle fromExternalName(String name) {
64  		if ( name.equals( NONE.name ) ) {
65  			return NONE;
66  		}
67  		else if ( name.equals( COUNT.name ) ) {
68  			return COUNT;
69  		}
70  		else if ( name.equals( PARAM.name ) ) {
71  			return PARAM;
72  		}
73  		else {
74  			return null;
75  		}
76  	}
77  
78  	public static ExecuteUpdateResultCheckStyle determineDefault(String customSql, boolean callable) {
79  		if ( customSql == null ) {
80  			return COUNT;
81  		}
82  		else {
83  			return callable ? PARAM : COUNT;
84  		}
85  	}
86  }
