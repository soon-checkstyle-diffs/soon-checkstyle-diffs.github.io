2   package org.hibernate.test.annotations.indexcoll;
3   import java.util.Date;
4   import java.util.HashMap;
5   import java.util.Map;
6   import javax.persistence.CascadeType;
7   import javax.persistence.Column;
8   import javax.persistence.ElementCollection;
9   import javax.persistence.Entity;
10  import javax.persistence.EnumType;
11  import javax.persistence.GeneratedValue;
12  import javax.persistence.Id;
13  import javax.persistence.JoinColumn;
14  import javax.persistence.JoinTable;
15  import javax.persistence.ManyToMany;
16  import javax.persistence.MapKeyColumn;
17  import javax.persistence.MapKeyEnumerated;
18  import javax.persistence.MapKeyJoinColumn;
19  import javax.persistence.MapKeyJoinColumns;
20  import javax.persistence.MapKeyTemporal;
21  import javax.persistence.TemporalType;
22  
23  /**
24   * @author Emmanuel Bernard
25   */
26  @Entity
27  public class Atmosphere {
28  
29  	public static enum Level {
30  		LOW,
31  		HIGH
32  	}
33  
34  	@Id
35  	@GeneratedValue
36  	public Integer id;
37  
38  	@ManyToMany(cascade = CascadeType.ALL)
39  	@MapKeyColumn(name="gas_name")
40  	public Map<String, Gas> gases = new HashMap<String, Gas>();
41  
42  	@MapKeyTemporal(TemporalType.DATE)
43  	@ElementCollection
44  	@MapKeyColumn(nullable=false)
45  	public Map<Date, String> colorPerDate = new HashMap<Date,String>();
46  
47  	@ElementCollection
48  	@MapKeyEnumerated(EnumType.STRING)
49  	@MapKeyColumn(nullable=false)
50  	public Map<Level, String> colorPerLevel = new HashMap<Level,String>();
51  
52  	@ManyToMany(cascade = CascadeType.ALL)
53  	@MapKeyJoinColumn(name="gas_id" )
54  	@JoinTable(name = "Gas_per_key")
55  	public Map<GasKey, Gas> gasesPerKey = new HashMap<GasKey, Gas>();
56  
57  	@ElementCollection
58  	@Column(name="composition_rate")
59  	@MapKeyJoinColumns( { @MapKeyJoinColumn(name="gas_id" ) } ) //use @MapKeyJoinColumns explicitly for tests
60  	@JoinTable(name = "Composition", joinColumns = @JoinColumn(name = "atmosphere_id"))
61  	public Map<Gas, Double> composition = new HashMap<Gas, Double>();
62  
63  	//use default JPA 2 column name for map key
64  	@ManyToMany(cascade = CascadeType.ALL)
65  	@MapKeyColumn
66  	@JoinTable(name="Atm_Gas_Def")
67  	public Map<String, Gas> gasesDef = new HashMap<String, Gas>();
68  
69  	//use default HAN legacy column name for map key
70  	@ManyToMany(cascade = CascadeType.ALL)
71  	@MapKeyColumn
72  	@JoinTable(name="Atm_Gas_DefLeg")
73  	public Map<String, Gas> gasesDefLeg = new HashMap<String, Gas>();
74  
75  	@ManyToMany(cascade = CascadeType.ALL)
76  	@MapKeyJoinColumn
77  	@JoinTable(name = "Gas_p_key_def")
78  	public Map<GasKey, Gas> gasesPerKeyDef = new HashMap<GasKey, Gas>();
79  
80  }
