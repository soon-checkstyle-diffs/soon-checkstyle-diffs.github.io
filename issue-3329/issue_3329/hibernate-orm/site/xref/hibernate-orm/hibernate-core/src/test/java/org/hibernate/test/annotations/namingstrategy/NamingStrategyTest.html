44      @Test
45  	public void testWithCustomNamingStrategy() throws Exception {
46  		try {
47  			AnnotationConfiguration config = new AnnotationConfiguration();
48  			config.setNamingStrategy(new DummyNamingStrategy());
49  			config.addAnnotatedClass(Address.class);
50  			config.addAnnotatedClass(Person.class);
51  			config.buildSessionFactory( serviceRegistry );
52  		}
53  		catch( Exception e ) {
54  			StringWriter writer = new StringWriter();
55  			e.printStackTrace(new PrintWriter(writer));
56              log.debug(writer.toString());
57  			fail(e.getMessage());
58  		}
59  	}
60      @Test
61  	public void testWithEJB3NamingStrategy() throws Exception {
62  		try {
63  			AnnotationConfiguration config = new AnnotationConfiguration();
64  			config.setNamingStrategy(EJB3NamingStrategy.INSTANCE);
65  			config.addAnnotatedClass(A.class);
66  			config.addAnnotatedClass(AddressEntry.class);
67  			config.buildSessionFactory( serviceRegistry );
68  			Mappings mappings = config.createMappings();
69  			boolean foundIt = false;
70  
71  			for ( Iterator iter = mappings.iterateTables(); iter.hasNext();  ) {
72  				Table table = (Table) iter.next();
73                  log.info("testWithEJB3NamingStrategy table = " + table.getName());
74  				if ( table.getName().equalsIgnoreCase("A_ADDRESS")) {
75  					foundIt = true;
76  				}
77  				// make sure we use A_ADDRESS instead of AEC_address
78  				assertFalse("got table name mapped to: AEC_address (should be A_ADDRESS) which violates JPA-2 spec section 11.1.8 ([OWNING_ENTITY_NAME]_[COLLECTION_ATTRIBUTE_NAME])",table.getName().equalsIgnoreCase("AEC_address"));
79  			}
80  			assertTrue("table not mapped to A_ADDRESS which violates JPA-2 spec section 11.1.8",foundIt);
81  		}
82  		catch( Exception e ) {
83  			StringWriter writer = new StringWriter();
84  			e.printStackTrace(new PrintWriter(writer));
85              log.debug(writer.toString());
86  			fail(e.getMessage());
87  		}
88  	}
