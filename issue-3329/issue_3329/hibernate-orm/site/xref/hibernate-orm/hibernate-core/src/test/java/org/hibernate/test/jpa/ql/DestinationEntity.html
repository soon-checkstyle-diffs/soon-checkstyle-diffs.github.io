21  package org.hibernate.test.jpa.ql;
22  
23  import javax.persistence.Column;
24  import javax.persistence.Entity;
25  import javax.persistence.GeneratedValue;
26  import javax.persistence.Id;
27  import javax.persistence.JoinColumn;
28  import javax.persistence.ManyToOne;
29  import javax.persistence.NamedNativeQueries;
30  import javax.persistence.NamedNativeQuery;
31  import javax.persistence.Table;
32  
33  /**
34   * @author Janario Oliveira
35   */
36  @Entity
37  @Table(name = "destination_entity")
38  @NamedNativeQueries({
39  		@NamedNativeQuery(name = "DestinationEntity.insertSelect", query = "insert into destination_entity(id, from_id, fullNameFrom) "
40  				+ " select fe.id, fe.id, fe.name||fe.lastName from from_entity fe where fe.id in (:ids)"),
41  		@NamedNativeQuery(name = "DestinationEntity.insert", query = "insert into destination_entity(id, from_id, fullNameFrom) "
42  				+ "values (:generatedId, :fromId, :fullName)"),
43  		@NamedNativeQuery(name = "DestinationEntity.update", query = "update destination_entity set from_id=:idFrom, fullNameFrom=:fullName"
44  				+ " where id in (:ids)"),
45  		@NamedNativeQuery(name = "DestinationEntity.delete", query = "delete from destination_entity where id in (:ids)"),
46  		@NamedNativeQuery(name = "DestinationEntity.selectIds", query = "select id, from_id, fullNameFrom from destination_entity where id in (:ids)") })
47  public class DestinationEntity {
48  
49  	@Id
50  	@GeneratedValue
51  	Integer id;
52  	@ManyToOne(optional = false)
53  	@JoinColumn(name = "from_id")
54  	FromEntity from;
55  	@Column(nullable = false)
56  	String fullNameFrom;
57  
58  	public DestinationEntity() {
59  	}
60  
61  	public DestinationEntity(FromEntity from, String fullNameFrom) {
62  		this.from = from;
63  		this.fullNameFrom = fullNameFrom;
64  	}
65  
66  	@Override
67  	public int hashCode() {
68  		final int prime = 31;
69  		int result = 1;
70  		result = prime * result + ( ( from == null ) ? 0 : from.hashCode() );
71  		result = prime * result + ( ( fullNameFrom == null ) ? 0 : fullNameFrom.hashCode() );
72  		return result;
73  	}
74  
75  	@Override
76  	public boolean equals(Object obj) {
77  		if ( this == obj )
78  			return true;
79  		if ( obj == null )
80  			return false;
81  		if ( getClass() != obj.getClass() )
82  			return false;
83  		DestinationEntity other = (DestinationEntity) obj;
84  		if ( from == null ) {
85  			if ( other.from != null )
86  				return false;
87  		}
88  		else if ( !from.equals( other.from ) )
89  			return false;
90  		if ( fullNameFrom == null ) {
91  			if ( other.fullNameFrom != null )
92  				return false;
93  		}
94  		else if ( !fullNameFrom.equals( other.fullNameFrom ) )
95  			return false;
96  		return true;
97  	}
98  
99  }
