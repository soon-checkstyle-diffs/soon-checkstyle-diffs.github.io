24  package org.hibernate.type;
25  
26  import java.io.Serializable;
27  import java.util.HashSet;
28  
29  import org.hibernate.collection.internal.PersistentSet;
30  import org.hibernate.collection.spi.PersistentCollection;
31  import org.hibernate.engine.spi.SessionImplementor;
32  import org.hibernate.persister.collection.CollectionPersister;
33  
34  public class SetType extends CollectionType {
35  
36  	/**
37  	 * @deprecated Use {@link #SetType(org.hibernate.type.TypeFactory.TypeScope, String, String)} instead.
38  	 * See Jira issue: <a href="https://hibernate.onjira.com/browse/HHH-7771">HHH-7771</a>
39  	 */
40  	@Deprecated
41  	public SetType(TypeFactory.TypeScope typeScope, String role, String propertyRef, boolean isEmbeddedInXML) {
42  		super( typeScope, role, propertyRef, isEmbeddedInXML );
43  	}
44  
45  	public SetType(TypeFactory.TypeScope typeScope, String role, String propertyRef) {
46  		super( typeScope, role, propertyRef );
47  	}
48  
49  	public PersistentCollection instantiate(SessionImplementor session, CollectionPersister persister, Serializable key) {
50  		return new PersistentSet(session);
51  	}
52  
53  	public Class getReturnedClass() {
54  		return java.util.Set.class;
55  	}
56  
57  	public PersistentCollection wrap(SessionImplementor session, Object collection) {
58  		return new PersistentSet( session, (java.util.Set) collection );
59  	}
60  
61  	public Object instantiate(int anticipatedSize) {
62  		return anticipatedSize <= 0
63  		       ? new HashSet()
64  		       : new HashSet( anticipatedSize + (int)( anticipatedSize * .75f ), .75f );
65  	}
66  	
67  }
