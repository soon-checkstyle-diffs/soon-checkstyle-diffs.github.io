24  package org.hibernate.service.jdbc.dialect.internal;
25  
26  import java.sql.Connection;
27  import java.sql.DatabaseMetaData;
28  import java.sql.SQLException;
29  import java.util.Map;
30  
31  import org.hibernate.HibernateException;
32  import org.hibernate.annotations.common.util.StringHelper;
33  import org.hibernate.cfg.AvailableSettings;
34  import org.hibernate.dialect.Dialect;
35  import org.hibernate.service.classloading.spi.ClassLoaderService;
36  import org.hibernate.service.classloading.spi.ClassLoadingException;
37  import org.hibernate.service.jdbc.dialect.spi.DialectFactory;
38  import org.hibernate.service.jdbc.dialect.spi.DialectResolver;
39  import org.hibernate.service.spi.InjectService;
40  
41  /**
42   * Standard implementation of the {@link DialectFactory} service.
43   *
44   * @author Steve Ebersole
45   */
46  public class DialectFactoryImpl implements DialectFactory {
47  	private ClassLoaderService classLoaderService;
48  
49  	@InjectService
50  	public void setClassLoaderService(ClassLoaderService classLoaderService) {
51  		this.classLoaderService = classLoaderService;
52  	}
53  
54  	private DialectResolver dialectResolver;
55  
56  	@InjectService
57  	public void setDialectResolver(DialectResolver dialectResolver) {
58  		this.dialectResolver = dialectResolver;
59  	}
60  
61  	@Override
62  	public Dialect buildDialect(Map configValues, Connection connection) throws HibernateException {
63  		final String dialectName = (String) configValues.get( AvailableSettings.DIALECT );
64  		if ( !StringHelper.isEmpty( dialectName ) ) {
65  			return constructDialect( dialectName );
66  		}
67  		else {
68  			return determineDialect( connection );
69  		}
70  	}
71  
72  	private Dialect constructDialect(String dialectName) {
73  		try {
74  			return ( Dialect ) classLoaderService.classForName( dialectName ).newInstance();
75  		}
76  		catch ( ClassLoadingException e ) {
77  			throw new HibernateException( "Dialect class not found: " + dialectName, e );
78  		}
79  		catch ( HibernateException e ) {
80  			throw e;
81  		}
82  		catch ( Exception e ) {
83  			throw new HibernateException( "Could not instantiate dialect class", e );
84  		}
85  	}
86  
87  	/**
88  	 * Determine the appropriate Dialect to use given the connection.
89  	 *
90  	 * @param connection The configured connection.
91  	 * @return The appropriate dialect instance.
92  	 *
93  	 * @throws HibernateException No connection given or no resolver could make
94  	 * the determination from the given connection.
95  	 */
96  	private Dialect determineDialect(Connection connection) {
97  		if ( connection == null ) {
98  			throw new HibernateException( "Connection cannot be null when 'hibernate.dialect' not set" );
99  		}
100 
101 		try {
102 			final DatabaseMetaData databaseMetaData = connection.getMetaData();
103 			final Dialect dialect = dialectResolver.resolveDialect( databaseMetaData );
104 
105 			if ( dialect == null ) {
106 				throw new HibernateException(
107 						"Unable to determine Dialect to use [name=" + databaseMetaData.getDatabaseProductName() +
108 								", majorVersion=" + databaseMetaData.getDatabaseMajorVersion() +
109 								"]; user must register resolver or explicitly set 'hibernate.dialect'"
110 				);
111 			}
112 
113 			return dialect;
114 		}
115 		catch ( SQLException sqlException ) {
116 			throw new HibernateException(
117 					"Unable to access java.sql.DatabaseMetaData to determine appropriate Dialect to use",
118 					sqlException
119 			);
120 		}
121 	}
122 }
