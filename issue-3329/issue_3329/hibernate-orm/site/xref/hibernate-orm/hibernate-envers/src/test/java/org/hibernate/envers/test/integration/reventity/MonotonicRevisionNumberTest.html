22  	@Override
23  	protected Class<?>[] getAnnotatedClasses() {
24  		return new Class<?>[]{ StrIntTestEntity.class }; // Otherwise revision entity is not generated.
25  	}
26  
27  	@Test
28  	public void testOracleSequenceOrder() {
29  		EntityPersister persister = sessionFactory().getEntityPersister( SequenceIdRevisionEntity.class.getName() );
30  		IdentifierGenerator generator = persister.getIdentifierGenerator();
31  		Assert.assertTrue( OrderedSequenceGenerator.class.isInstance( generator ) );
32  		OrderedSequenceGenerator seqGenerator = (OrderedSequenceGenerator) generator;
33  		Assert.assertTrue(
34  				"Oracle sequence needs to be ordered in RAC environment.",
35  				seqGenerator.sqlCreateStrings( getDialect() )[0].endsWith( " order" )
36  		);
37  	}
