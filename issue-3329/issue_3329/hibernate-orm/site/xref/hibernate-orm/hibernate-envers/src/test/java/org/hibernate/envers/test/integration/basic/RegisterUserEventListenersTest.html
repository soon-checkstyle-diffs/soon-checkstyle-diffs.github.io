21  public class RegisterUserEventListenersTest extends BaseEnversFunctionalTestCase {
22  	@Override
23  	protected Class<?>[] getAnnotatedClasses() {
24  		return new Class<?>[] { StrTestEntity.class };
25  	}
26  
27  	@Test
28  	@TestForIssue( jiraKey = "HHH-7478" )
29  	public void testTransactionProcessSynchronization() {
30  		final EventListenerRegistry registry = sessionFactory().getServiceRegistry().getService( EventListenerRegistry.class );
31  		final CountingPostInsertTransactionBoundaryListener listener = new CountingPostInsertTransactionBoundaryListener();
32  
33  		registry.getEventListenerGroup( EventType.POST_INSERT ).appendListener( listener );
34  
35  		Session session = openSession();
36  		session.getTransaction().begin();
37  		StrTestEntity entity = new StrTestEntity( "str1" );
38  		session.save( entity );
39  		session.getTransaction().commit();
40  		session.close();
41  
42  		// Post insert listener invoked three times - before/after insertion of original data,
43  		// revision entity and audit row.
44  		Assert.assertEquals( 3, listener.getBeforeCount() );
45  		Assert.assertEquals( 3, listener.getAfterCount() );
46  	}
47  
48  	private static class CountingPostInsertTransactionBoundaryListener implements PostInsertEventListener {
49  		private final MutableInteger beforeCounter = new MutableInteger();
50  		private final MutableInteger afterCounter = new MutableInteger();
51  
52  		@Override
53  		public void onPostInsert(PostInsertEvent event) {
54  			event.getSession().getActionQueue().registerProcess(new BeforeTransactionCompletionProcess() {
55  				@Override
56  				public void doBeforeTransactionCompletion(SessionImplementor session) {
57  					beforeCounter.increase();
58  				}
59  			});
60  			event.getSession().getActionQueue().registerProcess(new AfterTransactionCompletionProcess() {
61  				@Override
62  				public void doAfterTransactionCompletion(boolean success, SessionImplementor session) {
63  					afterCounter.increase();
64  				}
65  			});
66  		}
67  
68  		public int getBeforeCount() {
69  			return beforeCounter.get();
70  		}
71  
72  		public int getAfterCount() {
73  			return afterCounter.get();
74  		}
75  	}
