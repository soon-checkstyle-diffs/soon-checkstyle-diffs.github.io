41  /**
42   * Base class for all Envers event listeners
43   *
44   * @author Adam Warski (adam at warski dot org)
45   * @author HernпїЅn Chanfreau
46   * @author Steve Ebersole
47   * @author Michal Skowronek (mskowr at o2 dot pl)
48   */
49  public abstract class BaseEnversEventListener implements EnversListener {
50  	private AuditConfiguration enversConfiguration;
51  
52  	protected BaseEnversEventListener(AuditConfiguration enversConfiguration) {
53  		this.enversConfiguration = enversConfiguration;
54  	}
55  
56  	@Override
57  	public AuditConfiguration getAuditConfiguration() {
58  		return enversConfiguration;
59  	}
60  
61  	protected final void generateBidirectionalCollectionChangeWorkUnits(
62  			AuditProcess auditProcess,
63  			EntityPersister entityPersister,
64  			String entityName,
65  			Object[] newState,
66  			Object[] oldState,
67  			SessionImplementor session) {
68  		// Checking if this is enabled in configuration ...
69  		if ( ! enversConfiguration.getGlobalCfg().isGenerateRevisionsForCollections() ) {
70  			return;
71  		}
72  
73  		// Checks every property of the entity, if it is an "owned" to-one relation to another entity.
74  		// If the value of that property changed, and the relation is bi-directional, a new revision
75  		// for the related entity is generated.
76  		String[] propertyNames = entityPersister.getPropertyNames();
77  
78  		for ( int i=0; i<propertyNames.length; i++ ) {
79  			String propertyName = propertyNames[i];
80  			RelationDescription relDesc = enversConfiguration.getEntCfg().getRelationDescription(entityName, propertyName);
81  			if (relDesc != null && relDesc.isBidirectional() && relDesc.getRelationType() == RelationType.TO_ONE &&
82  					relDesc.isInsertable()) {
83  				// Checking for changes
84  				Object oldValue = oldState == null ? null : oldState[i];
85  				Object newValue = newState == null ? null : newState[i];
86  
87  				if (!Tools.entitiesEqual( session, relDesc.getToEntityName(), oldValue, newValue )) {
88  					// We have to generate changes both in the old collection (size decreses) and new collection
89  					// (size increases).
90  					if (newValue != null) {
91  						addCollectionChangeWorkUnit(auditProcess, session, entityName, relDesc, newValue);
92  					}
93  
94  					if (oldValue != null) {
95  						addCollectionChangeWorkUnit(auditProcess, session, entityName, relDesc, oldValue);
96  					}
97  				}
98  			}
99  		}
100 	}
101 
102 	private void addCollectionChangeWorkUnit(AuditProcess auditProcess, SessionImplementor session,
103 											 String fromEntityName, RelationDescription relDesc, Object value) {
104 		// relDesc.getToEntityName() doesn't always return the entity name of the value - in case
105 		// of subclasses, this will be root class, no the actual class. So it can't be used here.
106 		String toEntityName;
107 		Serializable id;
108 
109 		if (value instanceof HibernateProxy) {
110 		    HibernateProxy hibernateProxy = (HibernateProxy) value;
111 		    toEntityName = session.bestGuessEntityName(value);
112 		    id = hibernateProxy.getHibernateLazyInitializer().getIdentifier();
113             // We've got to initialize the object from the proxy to later read its state.
114             value = Tools.getTargetFromProxy(session.getFactory(), hibernateProxy);
115 		} else {
116 	        toEntityName =  session.guessEntityName(value);
117 
118             IdMapper idMapper = enversConfiguration.getEntCfg().get(toEntityName).getIdMapper();
119             id = (Serializable) idMapper.mapToIdFromEntity(value);
120 		}
121 
122 		Set<String> toPropertyNames = enversConfiguration.getEntCfg()
123 				.getToPropertyNames(fromEntityName, relDesc.getFromPropertyName(), toEntityName);
124 		String toPropertyName = toPropertyNames.iterator().next();
125 
126 		auditProcess.addWorkUnit(new CollectionChangeWorkUnit(session, toEntityName,
127 				toPropertyName, enversConfiguration, id, value));
128 	}
129 
130     protected void checkIfTransactionInProgress(SessionImplementor session) {
131         if (!session.isTransactionInProgress()) {
132             // Historical data would not be flushed to audit tables if outside of active transaction
133             // (AuditProcess#doBeforeTransactionCompletion(SessionImplementor) not executed). 
134             throw new AuditException("Unable to create revision because of non-active transaction");
135         }
136     }
