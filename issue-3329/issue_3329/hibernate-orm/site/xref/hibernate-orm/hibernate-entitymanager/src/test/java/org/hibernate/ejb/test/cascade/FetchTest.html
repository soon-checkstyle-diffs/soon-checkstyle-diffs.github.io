104 	@Test
105 	public void testTwoLevelDeepPersist() throws Exception {
106 		EntityTransaction tx;
107 
108 		EntityManager em = getOrCreateEntityManager();
109 		tx = em.getTransaction();
110 		tx.begin();
111 		Conference jbwBarcelona = new Conference();
112 		jbwBarcelona.setDate( new Date() );
113 		ExtractionDocumentInfo info = new ExtractionDocumentInfo();
114 		info.setConference( jbwBarcelona );
115 		jbwBarcelona.setExtractionDocument( info );
116 		info.setLastModified( new Date() );
117 		ExtractionDocument doc = new ExtractionDocument();
118 		doc.setDocumentInfo( info );
119 		info.setDocuments( new ArrayList<ExtractionDocument>() );
120 		info.getDocuments().add( doc );
121 		doc.setBody( new byte[]{'c', 'f'} );
122 		em.persist( jbwBarcelona );
123 		tx.commit();
124 		em.close();
125 
126 		em = getOrCreateEntityManager();
127 		tx = em.getTransaction();
128 		tx.begin();
129 		jbwBarcelona = em.find( Conference.class, jbwBarcelona.getId() );
130 		assertTrue( Hibernate.isInitialized( jbwBarcelona ) );
131 		assertTrue( Hibernate.isInitialized( jbwBarcelona.getExtractionDocument() ) );
132 		assertFalse( Hibernate.isInitialized( jbwBarcelona.getExtractionDocument().getDocuments() ) );
133 		em.flush();
134 		assertTrue( Hibernate.isInitialized( jbwBarcelona ) );
135 		assertTrue( Hibernate.isInitialized( jbwBarcelona.getExtractionDocument() ) );
136 		assertFalse( Hibernate.isInitialized( jbwBarcelona.getExtractionDocument().getDocuments() ) );
137 		em.remove( jbwBarcelona );
138 		tx.commit();
139 		em.close();
140 	}

142 	@Test
143 	public void testTwoLevelDeepPersistOnManyToOne() throws Exception {
144 		EntityTransaction tx;
145 		EntityManager em = getOrCreateEntityManager();
146 		tx = em.getTransaction();
147 		tx.begin();
148 		Grandson gs = new Grandson();
149 		gs.setParent( new Son() );
150 		gs.getParent().setParent( new Parent() );
151 		em.persist( gs );
152 		tx.commit();
153 		em.close();
154 		em = getOrCreateEntityManager();
155 		tx = em.getTransaction();
156 		tx.begin();
157 		gs = em.find( Grandson.class, gs.getId() );
158 		em.flush();
159 		assertTrue( Hibernate.isInitialized( gs.getParent() ) );
160 		assertFalse( Hibernate.isInitialized( gs.getParent().getParent() ) );
161 		em.remove( gs );
162 		tx.commit();
163 		em.close();
164 	}
