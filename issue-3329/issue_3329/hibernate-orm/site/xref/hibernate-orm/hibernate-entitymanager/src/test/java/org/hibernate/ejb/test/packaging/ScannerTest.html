55  @RequiresDialect(H2Dialect.class)
56  public class ScannerTest extends PackagingTestCase {
57  	@Test
58  	public void testNativeScanner() throws Exception {
59  		File defaultPar = buildDefaultPar();
60  		addPackageToClasspath( defaultPar );
61  
62  		Scanner scanner = new NativeScanner();
63  		assertEquals( "defaultpar", scanner.getUnqualifiedJarName( defaultPar.toURL() ) );
64  
65  		Set<Class<? extends Annotation>> annotationsToLookFor = new HashSet<Class<? extends Annotation>>( 3 );
66  		annotationsToLookFor.add( Entity.class );
67  		annotationsToLookFor.add( MappedSuperclass.class );
68  		annotationsToLookFor.add( Embeddable.class );
69  		final Set<Class<?>> classes = scanner.getClassesInJar( defaultPar.toURL(), annotationsToLookFor );
70  
71  		assertEquals( 3, classes.size() );
72  		assertTrue( classes.contains( ApplicationServer.class ) );
73  		assertTrue( classes.contains( org.hibernate.ejb.test.pack.defaultpar.Version.class ) );
74  
75  		Set<String> filePatterns = new HashSet<String>( 2 );
76  		filePatterns.add( "**/*.hbm.xml" );
77  		filePatterns.add( "META-INF/orm.xml" );
78  		final Set<NamedInputStream> files = scanner.getFilesInJar( defaultPar.toURL(), filePatterns );
79  
80  		assertEquals( 2, files.size() );
81  		for ( NamedInputStream file : files ) {
82  			assertNotNull( file.getStream() );
83  			file.getStream().close();
84  		}
85  	}
86  
87  	@Test
88  	public void testCustomScanner() throws Exception {
89  		File defaultPar = buildDefaultPar();
90  		File explicitPar = buildExplicitPar();
91  		addPackageToClasspath( defaultPar, explicitPar );
92  		
93  		EntityManagerFactory emf;
94  		CustomScanner.resetUsed();
95  		final HashMap integration = new HashMap();
96  		emf = Persistence.createEntityManagerFactory( "defaultpar", integration );
97  		assertTrue( ! CustomScanner.isUsed() );
98  		emf.close();
99  
100 		CustomScanner.resetUsed();
101 		emf = Persistence.createEntityManagerFactory( "manager1", integration );
102 		assertTrue( CustomScanner.isUsed() );
103 		emf.close();
104 
105 		CustomScanner.resetUsed();
106 		integration.put( AvailableSettings.SCANNER, new CustomScanner() );
107 		emf = Persistence.createEntityManagerFactory( "defaultpar", integration );
108 		assertTrue( CustomScanner.isUsed() );
109 		emf.close();
110 
111 		CustomScanner.resetUsed();
112 		emf = Persistence.createEntityManagerFactory( "defaultpar", null );
113 		assertTrue( ! CustomScanner.isUsed() );
114 		emf.close();
115 	}
