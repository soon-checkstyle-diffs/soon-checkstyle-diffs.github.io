24  /*
25    @test
26    @bug 4116781
27    @summary Tests that long (more than 64K) MimeType can be serialized
28             and deserialized.
29    @author gas@sparc.spb.su area=datatransfer
30    @modules java.datatransfer
31    @run main MimeTypeSerializationTest
32  */
33  
34  import java.awt.datatransfer.DataFlavor;
35  import java.io.ByteArrayInputStream;
36  import java.io.ByteArrayOutputStream;
37  import java.io.IOException;
38  import java.io.ObjectInputStream;
39  import java.io.ObjectOutputStream;
40  import java.util.Arrays;
41  
42  public class MimeTypeSerializationTest {
43  
44      public static void main(String[] args) throws Exception {
45          boolean failed = false;
46  
47          try {
48              int len = 70000;
49              char[] longValue = new char[len];
50              Arrays.fill(longValue, 'v');
51              DataFlavor longdf = new DataFlavor(DataFlavor.javaJVMLocalObjectMimeType +
52                  "; class=java.lang.String; longParameter=" + new String(longValue));
53  
54              DataFlavor shortdf = new DataFlavor(DataFlavor.javaJVMLocalObjectMimeType +
55                  "; class=java.lang.String");
56  
57              ByteArrayOutputStream baos = new ByteArrayOutputStream(100000);
58              ObjectOutputStream oos = new ObjectOutputStream(baos);
59              oos.writeObject(longdf);
60              oos.writeObject(shortdf);
61              oos.close();
62  
63              ByteArrayInputStream bais = new ByteArrayInputStream(baos.toByteArray());
64              ObjectInputStream ois = new ObjectInputStream(bais);
65              DataFlavor longdf2 = (DataFlavor) ois.readObject();
66              DataFlavor shortdf2 = (DataFlavor) ois.readObject();
67              ois.close();
68  
69              failed = !( longdf.getMimeType().equals(longdf2.getMimeType()) &&
70                  shortdf.getMimeType().equals(shortdf2.getMimeType()) );
71              if (failed) {
72                  System.err.println("deserialized MIME type does not match original one");
73              }
74          } catch (IOException e) {
75              failed = true;
76              e.printStackTrace();
77          } catch (Exception e) {
78              e.printStackTrace();
79          }
80  
81          if (failed) {
82              throw new RuntimeException("test failed: serialization attempt failed");
83          } else {
84              System.err.println("test passed");
85          }
86      }
87  }
