24  package org.hibernate.context.internal;
25  
26  import java.util.HashMap;
27  import java.util.Map;
28  
29  import org.hibernate.HibernateException;
30  import org.hibernate.Session;
31  import org.hibernate.SessionFactory;
32  import org.hibernate.context.spi.AbstractCurrentSessionContext;
33  import org.hibernate.engine.spi.SessionFactoryImplementor;
34  
35  /**
36   * Represents a {@link org.hibernate.context.spi.CurrentSessionContext} the notion of a contextual session
37   * is managed by some external entity (generally some form of interceptor, etc).
38   * This external manager is responsible for scoping these contextual sessions
39   * appropriately binding/unbinding them here for exposure to the application
40   * through {@link SessionFactory#getCurrentSession} calls.
41   * <p/>
42   *  Basically exposes two interfaces.  <ul>
43   * <li>First is the implementation of CurrentSessionContext which is then used
44   * by the {@link SessionFactory#getCurrentSession()} calls.  This
45   * portion is instance-based specific to the session factory owning the given
46   * instance of this impl (there will be one instance of this per each session
47   * factory using this strategy).
48   * <li>Second is the externally facing methods {@link #hasBind}, {@link #bind},
49   * and {@link #unbind} used by the external thing to manage exposure of the
50   * current session it is scoping.  This portion is static to allow easy
51   * reference from that external thing.
52   * </ul>
53   * The underlying storage of the current sessions here is a static
54   * {@link ThreadLocal}-based map where the sessions are keyed by the
55   * the owning session factory.
56   *
57   * @author Steve Ebersole
58   */
59  public class ManagedSessionContext extends AbstractCurrentSessionContext {
60  
61  	private static final ThreadLocal<Map<SessionFactory,Session>> context = new ThreadLocal<Map<SessionFactory,Session>>();
62  
63  	public ManagedSessionContext(SessionFactoryImplementor factory) {
64  		super( factory );
65  	}
66  
67  	@Override
68  	public Session currentSession() {
69  		Session current = existingSession( factory() );
70  		if ( current == null ) {
71  			throw new HibernateException( "No session currently bound to execution context" );
72  		}
73  		else {
74  			validateExistingSession( current );
75  		}
76  		return current;
77  	}
78  
79  	/**
80  	 * Check to see if there is already a session associated with the current
81  	 * thread for the given session factory.
82  	 *
83  	 * @param factory The factory against which to check for a given session
84  	 * within the current thread.
85  	 * @return True if there is currently a session bound.
86  	 */
87  	public static boolean hasBind(SessionFactory factory) {
88  		return existingSession( factory ) != null;
89  	}
90  
91  	/**
92  	 * Binds the given session to the current context for its session factory.
93  	 *
94  	 * @param session The session to be bound.
95  	 * @return Any previously bound session (should be null in most cases).
96  	 */
97  	public static Session bind(Session session) {
98  		return sessionMap( true ).put( session.getSessionFactory(), session );
99  	}
100 
101 	/**
102 	 * Unbinds the session (if one) current associated with the context for the
103 	 * given session.
104 	 *
105 	 * @param factory The factory for which to unbind the current session.
106 	 * @return The bound session if one, else null.
107 	 */
108 	public static Session unbind(SessionFactory factory) {
109 		Session existing = null;
110 		Map<SessionFactory,Session> sessionMap = sessionMap();
111 		if ( sessionMap != null ) {
112 			existing = sessionMap.remove( factory );
113 			doCleanup();
114 		}
115 		return existing;
116 	}
117 
118 	private static Session existingSession(SessionFactory factory) {
119 		Map sessionMap = sessionMap();
120 		if ( sessionMap == null ) {
121 			return null;
122 		}
123 		else {
124 			return ( Session ) sessionMap.get( factory );
125 		}
126 	}
127 
128 	protected static Map<SessionFactory,Session> sessionMap() {
129 		return sessionMap( false );
130 	}
131 
132 	private static synchronized Map<SessionFactory,Session> sessionMap(boolean createMap) {
133 		Map<SessionFactory,Session> sessionMap = context.get();
134 		if ( sessionMap == null && createMap ) {
135 			sessionMap = new HashMap<SessionFactory,Session>();
136 			context.set( sessionMap );
137 		}
138 		return sessionMap;
139 	}
140 
141 	private static synchronized void doCleanup() {
142 		Map<SessionFactory,Session> sessionMap = sessionMap( false );
143 		if ( sessionMap != null ) {
144 			if ( sessionMap.isEmpty() ) {
145 				context.set( null );
146 			}
147 		}
148 	}
149 }
