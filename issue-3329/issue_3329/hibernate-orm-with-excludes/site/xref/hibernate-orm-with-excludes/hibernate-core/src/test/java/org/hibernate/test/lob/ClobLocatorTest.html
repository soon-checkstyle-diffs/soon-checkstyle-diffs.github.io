41  /**
42   * Tests lazy materialization of data mapped by
43   * {@link org.hibernate.type.ClobType} as well as bounded and unbounded
44   * materialization and mutation.
45   *
46   * @author Steve Ebersole
47   */
48  @RequiresDialectFeature(
49  		value = DialectChecks.SupportsExpectedLobUsagePattern.class,
50  		comment = "database/driver does not support expected LOB usage pattern"
51  )
52  public class ClobLocatorTest extends BaseCoreFunctionalTestCase {
53  	private static final int CLOB_SIZE = 10000;
54  
55  	public String[] getMappings() {
56  		return new String[] { "lob/LobMappings.hbm.xml" };
57  	}
58  
59  	@Test
60  	public void testBoundedClobLocatorAccess() throws Throwable {
61  		String original = buildString( CLOB_SIZE, 'x' );
62  		String changed = buildString( CLOB_SIZE, 'y' );
63  		String empty = "";
64  
65  		Session s = openSession();
66  		s.beginTransaction();
67  		LobHolder entity = new LobHolder();
68  		entity.setClobLocator( s.getLobHelper().createClob( original ) );
69  		s.save( entity );
70  		s.getTransaction().commit();
71  		s.close();
72  
73  		s = openSession();
74  		s.beginTransaction();
75  		entity = ( LobHolder ) s.get( LobHolder.class, entity.getId() );
76  		assertEquals( CLOB_SIZE, entity.getClobLocator().length() );
77  		assertEquals( original, extractData( entity.getClobLocator() ) );
78  		s.getTransaction().commit();
79  		s.close();
80  
81  		// test mutation via setting the new clob data...
82  		if ( getDialect().supportsLobValueChangePropogation() ) {
83  			s = openSession();
84  			s.beginTransaction();
85  			entity = ( LobHolder ) s.get( LobHolder.class, entity.getId(), LockMode.UPGRADE );
86  			entity.getClobLocator().truncate( 1 );
87  			entity.getClobLocator().setString( 1, changed );
88  			s.getTransaction().commit();
89  			s.close();
90  
91  			s = openSession();
92  			s.beginTransaction();
93  			entity = ( LobHolder ) s.get( LobHolder.class, entity.getId(), LockMode.UPGRADE );
94  			assertNotNull( entity.getClobLocator() );
95  			assertEquals( CLOB_SIZE, entity.getClobLocator().length() );
96  			assertEquals( changed, extractData( entity.getClobLocator() ) );
97  			entity.getClobLocator().truncate( 1 );
98  			entity.getClobLocator().setString( 1, original );
99  			s.getTransaction().commit();
100 			s.close();
101 		}
102 
103 		// test mutation via supplying a new clob locator instance...
104 		s = openSession();
105 		s.beginTransaction();
106 		entity = ( LobHolder ) s.get( LobHolder.class, entity.getId(), LockMode.UPGRADE );
107 		assertNotNull( entity.getClobLocator() );
108 		assertEquals( CLOB_SIZE, entity.getClobLocator().length() );
109 		assertEquals( original, extractData( entity.getClobLocator() ) );
110 		entity.setClobLocator( s.getLobHelper().createClob( changed ) );
111 		s.getTransaction().commit();
112 		s.close();
113 
114 		// test empty clob
115 		if ( !(getDialect() instanceof SybaseASE157Dialect) ) { // Skip for Sybase. HHH-6425
116 			s = openSession();
117 			s.beginTransaction();
118 			entity = ( LobHolder ) s.get( LobHolder.class, entity.getId() );
119 			assertEquals( CLOB_SIZE, entity.getClobLocator().length() );
120 			assertEquals( changed, extractData( entity.getClobLocator() ) );
121 			entity.setClobLocator( s.getLobHelper().createClob( empty ) );
122 			s.getTransaction().commit();
123 			s.close();
124 
125 			s = openSession();
126 			s.beginTransaction();
127 			entity = ( LobHolder ) s.get( LobHolder.class, entity.getId() );
128 			if ( entity.getClobLocator() != null) {
129 				assertEquals( empty.length(), entity.getClobLocator().length() );
130 				assertEquals( empty, extractData( entity.getClobLocator() ) );
131 			}
132 			s.delete( entity );
133 			s.getTransaction().commit();
134 			s.close();
135 		}
136 
137 	}

139 	@Test
140 	@RequiresDialectFeature(
141 			value = DialectChecks.SupportsUnboundedLobLocatorMaterializationCheck.class,
142 			comment = "database/driver does not support materializing a LOB locator outside the owning transaction"
143 	)
144 	public void testUnboundedClobLocatorAccess() throws Throwable {
145 		// Note: unbounded mutation of the underlying lob data is completely
146 		// unsupported; most databases would not allow such a construct anyway.
147 		// Thus here we are only testing materialization...
148 
149 		String original = buildString( CLOB_SIZE, 'x' );
150 
151 		Session s = openSession();
152 		s.beginTransaction();
153 		LobHolder entity = new LobHolder();
154 		entity.setClobLocator( s.getLobHelper().createClob( original ) );
155 		s.save( entity );
156 		s.getTransaction().commit();
157 		s.close();
158 
159 		// load the entity with the clob locator, and close the session/transaction;
160 		// at that point it is unbounded...
161 		s = openSession();
162 		s.beginTransaction();
163 		entity = ( LobHolder ) s.get( LobHolder.class, entity.getId() );
164 		s.getTransaction().commit();
165 		s.close();
166 
167 		assertEquals( CLOB_SIZE, entity.getClobLocator().length() );
168 		assertEquals( original, extractData( entity.getClobLocator() ) );
169 
170 		s = openSession();
171 		s.beginTransaction();
172 		s.delete( entity );
173 		s.getTransaction().commit();
174 		s.close();
175 	}
176 
177 	public static String extractData(Clob clob) throws Exception {
178 		return DataHelper.extractString( clob.getCharacterStream() );
179 	}
180 
181 	public static String buildString(int size, char baseChar) {
182 		StringBuilder buff = new StringBuilder();
183 		for( int i = 0; i < size; i++ ) {
184 			buff.append( baseChar );
185 		}
186 		return buff.toString();
187 	}
