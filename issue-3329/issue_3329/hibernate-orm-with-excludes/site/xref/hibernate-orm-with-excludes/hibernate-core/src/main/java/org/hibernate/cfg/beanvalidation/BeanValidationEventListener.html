30  import javax.validation.ConstraintViolationException;
31  import javax.validation.TraversableResolver;
32  import javax.validation.Validation;
33  import javax.validation.Validator;
34  import javax.validation.ValidatorFactory;
35  
36  import org.jboss.logging.Logger;
37  
38  import org.hibernate.EntityMode;
39  import org.hibernate.cfg.Configuration;
40  import org.hibernate.engine.spi.SessionFactoryImplementor;
41  import org.hibernate.event.spi.PreDeleteEvent;
42  import org.hibernate.event.spi.PreDeleteEventListener;
43  import org.hibernate.event.spi.PreInsertEvent;
44  import org.hibernate.event.spi.PreInsertEventListener;
45  import org.hibernate.event.spi.PreUpdateEvent;
46  import org.hibernate.event.spi.PreUpdateEventListener;
47  import org.hibernate.internal.CoreMessageLogger;
48  import org.hibernate.persister.entity.EntityPersister;

50  /**
51   * Event listener used to enable Bean Validation for insert/update/delete events.
52   *
53   * @author Emmanuel Bernard
54   * @author Hardy Ferentschik
55   */
56  //FIXME review exception model
57  public class BeanValidationEventListener
58  		implements PreInsertEventListener, PreUpdateEventListener, PreDeleteEventListener {
59  
60      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class,
61                                                                         BeanValidationEventListener.class.getName());
62  
63  	private ValidatorFactory factory;
64  	private ConcurrentHashMap<EntityPersister, Set<String>> associationsPerEntityPersister =
65  			new ConcurrentHashMap<EntityPersister, Set<String>>();
66  	private GroupsPerOperation groupsPerOperation;
67  	boolean initialized;
68  
69  	/**
70  	 * No-arg constructor used when listener is configured via configuration file
71  	 */
72  	public BeanValidationEventListener() {
73  	}
74  
75  	/**
76  	 * Constructor used in an environment where validator factory is injected (JPA2).
77  	 *
78  	 * @param factory The {@code ValidatorFactory} to use to create {@code Validator} instance(s)
79  	 * @param properties Configued properties
80  	 */
81  	public BeanValidationEventListener(ValidatorFactory factory, Properties properties) {
82  		init( factory, properties );
83  	}
84  
85  	public void initialize(Configuration cfg) {
86  		if ( !initialized ) {
87  			ValidatorFactory factory = Validation.buildDefaultValidatorFactory();
88  			Properties props = cfg.getProperties();
89  			init( factory, props );
90  		}
91  	}
92  
93  	public boolean onPreInsert(PreInsertEvent event) {
94  		validate(
95  				event.getEntity(), event.getPersister().getEntityMode(), event.getPersister(),
96  				event.getSession().getFactory(), GroupsPerOperation.Operation.INSERT
97  		);
98  		return false;
99  	}
100 
101 	public boolean onPreUpdate(PreUpdateEvent event) {
102 		validate(
103 				event.getEntity(), event.getPersister().getEntityMode(), event.getPersister(),
104 				event.getSession().getFactory(), GroupsPerOperation.Operation.UPDATE
105 		);
106 		return false;
107 	}
108 
109 	public boolean onPreDelete(PreDeleteEvent event) {
110 		validate(
111 				event.getEntity(), event.getPersister().getEntityMode(), event.getPersister(),
112 				event.getSession().getFactory(), GroupsPerOperation.Operation.DELETE
113 		);
114 		return false;
115 	}
116 
117 	private void init(ValidatorFactory factory, Properties properties) {
118 		this.factory = factory;
119 		groupsPerOperation = new GroupsPerOperation( properties );
120 		initialized = true;
121 	}
122 
123 	private <T> void validate(T object, EntityMode mode, EntityPersister persister,
124 							  SessionFactoryImplementor sessionFactory, GroupsPerOperation.Operation operation) {
125 		if ( object == null || mode != EntityMode.POJO ) {
126 			return;
127 		}
128 		TraversableResolver tr = new HibernateTraversableResolver(
129 				persister, associationsPerEntityPersister, sessionFactory
130 		);
131 		Validator validator = factory.usingContext()
132 				.traversableResolver( tr )
133 				.getValidator();
134 		final Class<?>[] groups = groupsPerOperation.get( operation );
135 		if ( groups.length > 0 ) {
136 			final Set<ConstraintViolation<T>> constraintViolations = validator.validate( object, groups );
137 			if ( constraintViolations.size() > 0 ) {
138 				Set<ConstraintViolation<?>> propagatedViolations =
139 						new HashSet<ConstraintViolation<?>>( constraintViolations.size() );
140 				Set<String> classNames = new HashSet<String>();
141 				for ( ConstraintViolation<?> violation : constraintViolations ) {
142 					LOG.trace( violation );
143 					propagatedViolations.add( violation );
144 					classNames.add( violation.getLeafBean().getClass().getName() );
145 				}
146 				StringBuilder builder = new StringBuilder();
147 				builder.append( "Validation failed for classes " );
148 				builder.append( classNames );
149 				builder.append( " during " );
150 				builder.append( operation.getName() );
151 				builder.append( " time for groups " );
152 				builder.append( toString( groups ) );
153 				builder.append( "\nList of constraint violations:[\n" );
154 				for (ConstraintViolation<?> violation : constraintViolations) {
155 					builder.append( "\t" ).append( violation.toString() ).append("\n");
156 				}
157 				builder.append( "]" );
158 
159 				throw new ConstraintViolationException(
160 						builder.toString(), propagatedViolations
161 				);
162 			}
163 		}
164 	}
165 
166 	private String toString(Class<?>[] groups) {
167 		StringBuilder toString = new StringBuilder( "[" );
168 		for ( Class<?> group : groups ) {
169 			toString.append( group.getName() ).append( ", " );
170 		}
171 		toString.append( "]" );
172 		return toString.toString();
173 	}
174 }
