24  package org.hibernate.event.internal;
25  
26  import java.io.Serializable;
27  
28  import org.hibernate.HibernateException;
29  import org.hibernate.cache.spi.CacheKey;
30  import org.hibernate.cache.spi.entry.CollectionCacheEntry;
31  import org.hibernate.collection.spi.PersistentCollection;
32  import org.hibernate.engine.internal.CacheHelper;
33  import org.hibernate.engine.spi.CollectionEntry;
34  import org.hibernate.engine.spi.PersistenceContext;
35  import org.hibernate.engine.spi.SessionFactoryImplementor;
36  import org.hibernate.engine.spi.SessionImplementor;
37  import org.hibernate.event.spi.InitializeCollectionEvent;
38  import org.hibernate.event.spi.InitializeCollectionEventListener;
39  import org.hibernate.internal.CoreMessageLogger;
40  import org.hibernate.persister.collection.CollectionPersister;
41  import org.hibernate.pretty.MessageHelper;
42  import org.jboss.logging.Logger;
43  
44  /**
45   * @author Gavin King
46   */
47  public class DefaultInitializeCollectionEventListener implements InitializeCollectionEventListener {
48  
49  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, DefaultInitializeCollectionEventListener.class.getName() );
50  
51  	/**
52  	 * called by a collection that wants to initialize itself
53  	 */
54  	public void onInitializeCollection(InitializeCollectionEvent event)
55  	throws HibernateException {
56  
57  		PersistentCollection collection = event.getCollection();
58  		SessionImplementor source = event.getSession();
59  
60  		CollectionEntry ce = source.getPersistenceContext().getCollectionEntry(collection);
61  		if (ce==null) throw new HibernateException("collection was evicted");
62  		if ( !collection.wasInitialized() ) {
63  			final boolean traceEnabled = LOG.isTraceEnabled();
64  			if ( traceEnabled ) {
65  				LOG.tracev( "Initializing collection {0}",
66  						MessageHelper.collectionInfoString( ce.getLoadedPersister(), collection, ce.getLoadedKey(), source ) );
67  				LOG.trace( "Checking second-level cache" );
68  			}
69  
70  			final boolean foundInCache = initializeCollectionFromCache(
71  					ce.getLoadedKey(),
72  					ce.getLoadedPersister(),
73  					collection,
74  					source
75  				);
76  
77  			if ( foundInCache ) {
78  				if ( traceEnabled ) {
79  					LOG.trace( "Collection initialized from cache" );
80  				}
81  			}
82  			else {
83  				if ( traceEnabled ) {
84  					LOG.trace( "Collection not cached" );
85  				}
86  				ce.getLoadedPersister().initialize( ce.getLoadedKey(), source );
87  				if ( traceEnabled ) {
88  					LOG.trace( "Collection initialized" );
89  				}
90  
91  				if ( source.getFactory().getStatistics().isStatisticsEnabled() ) {
92  					source.getFactory().getStatisticsImplementor().fetchCollection(
93  							ce.getLoadedPersister().getRole()
94  						);
95  				}
96  			}
97  		}
98  	}
99  
100 	/**
101 	 * Try to initialize a collection from the cache
102 	 *
103 	 * @param id The id of the collection of initialize
104 	 * @param persister The collection persister
105 	 * @param collection The collection to initialize
106 	 * @param source The originating session
107 	 * @return true if we were able to initialize the collection from the cache;
108 	 * false otherwise.
109 	 */
110 	private boolean initializeCollectionFromCache(
111 			Serializable id,
112 			CollectionPersister persister,
113 			PersistentCollection collection,
114 			SessionImplementor source) {
115 
116 		if ( !source.getLoadQueryInfluencers().getEnabledFilters().isEmpty() && persister.isAffectedByEnabledFilters( source ) ) {
117 			LOG.trace( "Disregarding cached version (if any) of collection due to enabled filters" );
118 			return false;
119 		}
120 
121 		final boolean useCache = persister.hasCache() &&
122 				source.getCacheMode().isGetEnabled();
123 
124         if (!useCache) return false;
125 
126         final SessionFactoryImplementor factory = source.getFactory();
127         final CacheKey ck = source.generateCacheKey( id, persister.getKeyType(), persister.getRole() );
128 		final Object ce = CacheHelper.fromSharedCache( source, ck, persister.getCacheAccessStrategy() );
129 
130 		if ( factory.getStatistics().isStatisticsEnabled() ) {
131 			if (ce == null) {
132 				factory.getStatisticsImplementor()
133 						.secondLevelCacheMiss( persister.getCacheAccessStrategy().getRegion().getName() );
134 			}
135 			else {
136 				factory.getStatisticsImplementor()
137 						.secondLevelCacheHit( persister.getCacheAccessStrategy().getRegion().getName() );
138 			}
139 		}
140 
141         if ( ce == null ) {
142 			return false;
143 		}
144 
145 		CollectionCacheEntry cacheEntry = (CollectionCacheEntry)persister.getCacheEntryStructure().destructure(ce, factory);
146 
147 		final PersistenceContext persistenceContext = source.getPersistenceContext();
148         cacheEntry.assemble(collection, persister, persistenceContext.getCollectionOwner(id, persister));
149         persistenceContext.getCollectionEntry(collection).postInitialize(collection);
150         // addInitializedCollection(collection, persister, id);
151         return true;
152 	}
153 }
