46  /**
47   * Creates synthetic and nodes based on the where fragment part of a JoinSequence.
48   *
49   * @author josh
50   */
51  public class SyntheticAndFactory implements HqlSqlTokenTypes {
52  
53      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, SyntheticAndFactory.class.getName());
54  
55  	private HqlSqlWalker hqlSqlWalker;
56  	private AST thetaJoins;
57  	private AST filters;
58  
59  	public SyntheticAndFactory(HqlSqlWalker hqlSqlWalker) {
60  		this.hqlSqlWalker = hqlSqlWalker;
61  	}
62  
63  	private Node create(int tokenType, String text) {
64  		return ( Node ) ASTUtil.create( hqlSqlWalker.getASTFactory(), tokenType, text );
65  	}
66  
67  	public void addWhereFragment(
68  			JoinFragment joinFragment,
69  			String whereFragment,
70  			QueryNode query,
71  			FromElement fromElement,
72  			HqlSqlWalker hqlSqlWalker) {
73  		if ( whereFragment == null ) {
74  			return;
75  		}
76  
77  		if ( !fromElement.useWhereFragment() && !joinFragment.hasThetaJoins() ) {
78  			return;
79  		}
80  
81  		whereFragment = whereFragment.trim();
82  		if ( StringHelper.isEmpty( whereFragment ) ) {
83  			return;
84  		}
85  
86  		// Forcefully remove leading ands from where fragments; the grammar will
87  		// handle adding them
88  		if ( whereFragment.startsWith( "and" ) ) {
89  			whereFragment = whereFragment.substring( 4 );
90  		}
91  
92  		LOG.debugf( "Using unprocessed WHERE-fragment [%s]", whereFragment );
93  
94  		SqlFragment fragment = ( SqlFragment ) create( SQL_TOKEN, whereFragment );
95  		fragment.setJoinFragment( joinFragment );
96  		fragment.setFromElement( fromElement );
97  
98  		if ( fromElement.getIndexCollectionSelectorParamSpec() != null ) {
99  			fragment.addEmbeddedParameter( fromElement.getIndexCollectionSelectorParamSpec() );
100 			fromElement.setIndexCollectionSelectorParamSpec( null );
101 		}
102 
103 		if ( hqlSqlWalker.isFilter() ) {
104 			if ( whereFragment.indexOf( '?' ) >= 0 ) {
105 				Type collectionFilterKeyType = hqlSqlWalker.getSessionFactoryHelper()
106 						.requireQueryableCollection( hqlSqlWalker.getCollectionFilterRole() )
107 						.getKeyType();
108 				CollectionFilterKeyParameterSpecification paramSpec = new CollectionFilterKeyParameterSpecification(
109 						hqlSqlWalker.getCollectionFilterRole(),
110 						collectionFilterKeyType,
111 						0
112 				);
113 				fragment.addEmbeddedParameter( paramSpec );
114 			}
115 		}
116 
117 		JoinProcessor.processDynamicFilterParameters(
118 				whereFragment,
119 				fragment,
120 				hqlSqlWalker
121 		);
122 
123 		if ( LOG.isDebugEnabled() ) {
124 			LOG.debugf( "Using processed WHERE-fragment [%s]", fragment.getText() );
125 		}
126 
127 		// Filter conditions need to be inserted before the HQL where condition and the
128 		// theta join node.  This is because org.hibernate.loader.Loader binds the filter parameters first,
129 		// then it binds all the HQL query parameters, see org.hibernate.loader.Loader.processFilterParameters().
130 		if ( fragment.getFromElement().isFilter() || fragment.hasFilterCondition() ) {
131 			if ( filters == null ) {
132 				// Find or create the WHERE clause
133 				AST where = query.getWhereClause();
134 				// Create a new FILTERS node as a parent of all filters
135 				filters = create( FILTERS, "{filter conditions}" );
136 				// Put the FILTERS node before the HQL condition and theta joins
137 				ASTUtil.insertChild( where, filters );
138 			}
139 
140 			// add the current fragment to the FILTERS node
141 			filters.addChild( fragment );
142 		}
143 		else {
144 			if ( thetaJoins == null ) {
145 				// Find or create the WHERE clause
146 				AST where = query.getWhereClause();
147 				// Create a new THETA_JOINS node as a parent of all filters
148 				thetaJoins = create( THETA_JOINS, "{theta joins}" );
149 				// Put the THETA_JOINS node before the HQL condition, after the filters.
150 				if (filters==null) {
151 					ASTUtil.insertChild( where, thetaJoins );
152 				}
153 				else {
154 					ASTUtil.insertSibling( thetaJoins, filters );
155 				}
156 			}
157 
158 			// add the current fragment to the THETA_JOINS node
159 			thetaJoins.addChild(fragment);
160 		}
161 
162 	}
163 
164 	public void addDiscriminatorWhereFragment(
165 			RestrictableStatement statement,
166 			Queryable persister,
167 			Map enabledFilters,
168 			String alias) {
169 		String whereFragment = persister.filterFragment( alias, enabledFilters ).trim();
170 		if ( "".equals( whereFragment ) ) {
171 			return;
172 		}
173 		if ( whereFragment.startsWith( "and" ) ) {
174 			whereFragment = whereFragment.substring( 4 );
175 		}
176 
177 		// Need to parse off the column qualifiers; this is assuming (which is true as of now)
178 		// that this is only used from update and delete HQL statement parsing
179 		whereFragment = StringHelper.replace( whereFragment, persister.generateFilterConditionAlias( alias ) + ".", "" );
180 
181 		// Note: this simply constructs a "raw" SQL_TOKEN representing the
182 		// where fragment and injects this into the tree.  This "works";
183 		// however it is probably not the best long-term solution.
184 		//
185 		// At some point we probably want to apply an additional grammar to
186 		// properly tokenize this where fragment into constituent parts
187 		// focused on the operators embedded within the fragment.
188 		SqlFragment discrimNode = ( SqlFragment ) create( SQL_TOKEN, whereFragment );
189 
190 		JoinProcessor.processDynamicFilterParameters(
191 				whereFragment,
192 				discrimNode,
193 				hqlSqlWalker
194 		);
195 
196 		if ( statement.getWhereClause().getNumberOfChildren() == 0 ) {
197 			statement.getWhereClause().setFirstChild( discrimNode );
198 		}
199 		else {
200 			AST and = create( AND, "{and}" );
201 			AST currentFirstChild = statement.getWhereClause().getFirstChild();
202 			and.setFirstChild( discrimNode );
203 			and.addChild( currentFirstChild );
204 			statement.getWhereClause().setFirstChild( and );
205 		}
206 	}
