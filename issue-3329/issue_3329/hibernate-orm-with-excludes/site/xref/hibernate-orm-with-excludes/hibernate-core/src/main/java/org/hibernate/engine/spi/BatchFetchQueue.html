42  /**
43   * Tracks entity and collection keys that are available for batch
44   * fetching, and the queries which were used to load entities, which
45   * can be re-used as a subquery for loading owned collections.
46   *
47   * @author Gavin King
48   * @author Steve Ebersole
49   * @author Guenther Demetz
50   */
51  public class BatchFetchQueue {
52  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, BatchFetchQueue.class.getName() );
53  
54  	private final PersistenceContext context;
55  
56  	/**
57  	 * A map of {@link SubselectFetch subselect-fetch descriptors} keyed by the
58  	 * {@link EntityKey) against which the descriptor is registered.
59  	 */
60  	private final Map<EntityKey, SubselectFetch> subselectsByEntityKey = new HashMap<EntityKey, SubselectFetch>(8);
61  
62  	/**
63  	 * Used to hold information about the entities that are currently eligible for batch-fetching.  Ultimately
64  	 * used by {@link #getEntityBatch} to build entity load batches.
65  	 * <p/>
66  	 * A Map structure is used to segment the keys by entity type since loading can only be done for a particular entity
67  	 * type at a time.
68  	 */
69  	private final Map <String,LinkedHashSet<EntityKey>> batchLoadableEntityKeys = new HashMap <String,LinkedHashSet<EntityKey>>(8);
70  	
71  	/**
72  	 * Used to hold information about the collections that are currently eligible for batch-fetching.  Ultimately
73  	 * used by {@link #getCollectionBatch} to build collection load batches.
74  	 */
75  	private final Map<String, LinkedHashMap<CollectionEntry, PersistentCollection>> batchLoadableCollections =
76  			new HashMap<String, LinkedHashMap <CollectionEntry, PersistentCollection>>(8);
77  
78  	/**
79  	 * Constructs a queue for the given context.
80  	 *
81  	 * @param context The owning context.
82  	 */
83  	public BatchFetchQueue(PersistenceContext context) {
84  		this.context = context;
85  	}
86  
87  	/**
88  	 * Clears all entries from this fetch queue.
89  	 * <p/>
90  	 * Called after flushing or clearing the session.
91  	 */
92  	public void clear() {
93  		batchLoadableEntityKeys.clear();
94  		batchLoadableCollections.clear();
95  		subselectsByEntityKey.clear();
96  	}

99  	// sub-select support ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
100 
101 	/**
102 	 * Retrieve the fetch descriptor associated with the given entity key.
103 	 *
104 	 * @param key The entity key for which to locate any defined subselect fetch.
105 	 * @return The fetch descriptor; may return null if no subselect fetch queued for
106 	 * this entity key.
107 	 */
108 	public SubselectFetch getSubselect(EntityKey key) {
109 		return subselectsByEntityKey.get( key );
110 	}
111 
112 	/**
113 	 * Adds a subselect fetch decriptor for the given entity key.
114 	 *
115 	 * @param key The entity for which to register the subselect fetch.
116 	 * @param subquery The fetch descriptor.
117 	 */
118 	public void addSubselect(EntityKey key, SubselectFetch subquery) {
119 		subselectsByEntityKey.put( key, subquery );
120 	}
121 
122 	/**
123 	 * After evicting or deleting an entity, we don't need to
124 	 * know the query that was used to load it anymore (don't
125 	 * call this after loading the entity, since we might still
126 	 * need to load its collections)
127 	 */
128 	public void removeSubselect(EntityKey key) {
129 		subselectsByEntityKey.remove( key );
130 	}
131 
132 	// entity batch support ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
133 
134 	/**
135 	 * If an EntityKey represents a batch loadable entity, add
136 	 * it to the queue.
137 	 * <p/>
138 	 * Note that the contract here is such that any key passed in should
139 	 * previously have been been checked for existence within the
140 	 * {@link PersistenceContext}; failure to do so may cause the
141 	 * referenced entity to be included in a batch even though it is
142 	 * already associated with the {@link PersistenceContext}.
143 	 */
144 	public void addBatchLoadableEntityKey(EntityKey key) {
145 		if ( key.isBatchLoadable() ) {
146 			LinkedHashSet<EntityKey> set =  batchLoadableEntityKeys.get( key.getEntityName());
147 			if (set == null) {
148 				set = new LinkedHashSet<EntityKey>(8);
149 				batchLoadableEntityKeys.put( key.getEntityName(), set);
150 			}
151 			set.add(key);
152 		}
153 	}
154 
155 	/**
156 	 * After evicting or deleting or loading an entity, we don't
157 	 * need to batch fetch it anymore, remove it from the queue
158 	 * if necessary
159 	 */
160 	public void removeBatchLoadableEntityKey(EntityKey key) {
161 		if ( key.isBatchLoadable() ) {
162 			LinkedHashSet<EntityKey> set =  batchLoadableEntityKeys.get( key.getEntityName());
163 			if (set != null) {
164 				set.remove(key);
165 			}
166 		}
167 	}
168 
169 	/**
170 	 * Get a batch of unloaded identifiers for this class, using a slightly
171 	 * complex algorithm that tries to grab keys registered immediately after
172 	 * the given key.
173 	 *
174 	 * @param persister The persister for the entities being loaded.
175 	 * @param id The identifier of the entity currently demanding load.
176 	 * @param batchSize The maximum number of keys to return
177 	 * @return an array of identifiers, of length batchSize (possibly padded with nulls)
178 	 */
179 	public Serializable[] getEntityBatch(
180 			final EntityPersister persister,
181 			final Serializable id,
182 			final int batchSize,
183 			final EntityMode entityMode) {
184 		Serializable[] ids = new Serializable[batchSize];
185 		ids[0] = id; //first element of array is reserved for the actual instance we are loading!
186 		int i = 1;
187 		int end = -1;
188 		boolean checkForEnd = false;
189 
190 		// TODO: this needn't exclude subclasses...
191 
192 		LinkedHashSet<EntityKey> set =  batchLoadableEntityKeys.get( persister.getEntityName() );
193 		if ( set != null ) {
194 			for ( EntityKey key : set ) {
195 				if ( checkForEnd && i == end ) {
196 					//the first id found after the given id
197 					return ids;
198 				}
199 				if ( persister.getIdentifierType().isEqual( id, key.getIdentifier() ) ) {
200 					end = i;
201 				}
202 				else {
203 					if ( !isCached( key, persister ) ) {
204 						ids[i++] = key.getIdentifier();
205 					}
206 				}
207 				if ( i == batchSize ) {
208 					i = 1; // end of array, start filling again from start
209 					if ( end != -1 ) {
210 						checkForEnd = true;
211 					}
212 				}
213 			}
214 		}
215 		return ids; //we ran out of ids to try
216 	}
217 
218 	private boolean isCached(EntityKey entityKey, EntityPersister persister) {
219 		if ( context.getSession().getCacheMode().isGetEnabled() && persister.hasCache() ) {
220 			final CacheKey key = context.getSession().generateCacheKey(
221 					entityKey.getIdentifier(),
222 					persister.getIdentifierType(),
223 					entityKey.getEntityName()
224 			);
225 			return CacheHelper.fromSharedCache( context.getSession(), key, persister.getCacheAccessStrategy() ) != null;
226 		}
227 		return false;
228 	}

231 	// collection batch support ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
232 
233 	/**
234 	 * If an CollectionEntry represents a batch loadable collection, add
235 	 * it to the queue.
236 	 */
237 	public void addBatchLoadableCollection(PersistentCollection collection, CollectionEntry ce) {
238 		final CollectionPersister persister = ce.getLoadedPersister();
239 
240 		LinkedHashMap<CollectionEntry, PersistentCollection> map =  batchLoadableCollections.get( persister.getRole() );
241 		if ( map == null ) {
242 			map = new LinkedHashMap<CollectionEntry, PersistentCollection>( 16 );
243 			batchLoadableCollections.put( persister.getRole(), map );
244 		}
245 		map.put( ce, collection );
246 	}
247 	
248 	/**
249 	 * After a collection was initialized or evicted, we don't
250 	 * need to batch fetch it anymore, remove it from the queue
251 	 * if necessary
252 	 */
253 	public void removeBatchLoadableCollection(CollectionEntry ce) {
254 		LinkedHashMap<CollectionEntry, PersistentCollection> map =  batchLoadableCollections.get( ce.getLoadedPersister().getRole() );
255 		if ( map != null ) {
256 			map.remove( ce );
257 		}
258 	}

260 	/**
261 	 * Get a batch of uninitialized collection keys for a given role
262 	 *
263 	 * @param collectionPersister The persister for the collection role.
264 	 * @param id A key that must be included in the batch fetch
265 	 * @param batchSize the maximum number of keys to return
266 	 * @return an array of collection keys, of length batchSize (padded with nulls)
267 	 */
268 	public Serializable[] getCollectionBatch(
269 			final CollectionPersister collectionPersister,
270 			final Serializable id,
271 			final int batchSize) {
272 
273 		Serializable[] keys = new Serializable[batchSize];
274 		keys[0] = id;
275 
276 		int i = 1;
277 		int end = -1;
278 		boolean checkForEnd = false;
279 
280 		final LinkedHashMap<CollectionEntry, PersistentCollection> map =  batchLoadableCollections.get( collectionPersister.getRole() );
281 		if ( map != null ) {
282 			for ( Map.Entry<CollectionEntry, PersistentCollection> me : map.entrySet() ) {
283 				final CollectionEntry ce = me.getKey();
284 				final PersistentCollection collection = me.getValue();
285 				
286 				if ( ce.getLoadedKey() == null ) {
287 					// the loadedKey of the collectionEntry might be null as it might have been reset to null
288 					// (see for example Collections.processDereferencedCollection()
289 					// and CollectionEntry.afterAction())
290 					// though we clear the queue on flush, it seems like a good idea to guard
291 					// against potentially null loadedKeys (which leads to various NPEs as demonstrated in HHH-7821).
292 					continue;
293 				}
294 
295 				if ( collection.wasInitialized() ) {
296 					// should never happen
297 					LOG.warn( "Encountered initialized collection in BatchFetchQueue, this should not happen." );
298 					continue;
299 				}
300 
301 				if ( checkForEnd && i == end ) {
302 					return keys; //the first key found after the given key
303 				}
304 
305 				final boolean isEqual = collectionPersister.getKeyType().isEqual(
306 						id,
307 						ce.getLoadedKey(),
308 						collectionPersister.getFactory()
309 				);
310 
311 				if ( isEqual ) {
312 					end = i;
313 					//checkForEnd = false;
314 				}
315 				else if ( !isCached( ce.getLoadedKey(), collectionPersister ) ) {
316 					keys[i++] = ce.getLoadedKey();
317 					//count++;
318 				}
319 
320 				if ( i == batchSize ) {
321 					i = 1; //end of array, start filling again from start
322 					if ( end != -1 ) {
323 						checkForEnd = true;
324 					}
325 				}
326 			}
327 		}
328 		return keys; //we ran out of keys to try
329 	}
330 
331 	private boolean isCached(Serializable collectionKey, CollectionPersister persister) {
332 		if ( context.getSession().getCacheMode().isGetEnabled() && persister.hasCache() ) {
333 			CacheKey cacheKey = context.getSession().generateCacheKey(
334 					collectionKey,
335 			        persister.getKeyType(),
336 			        persister.getRole()
337 			);
338 			return CacheHelper.fromSharedCache( context.getSession(), cacheKey, persister.getCacheAccessStrategy() ) != null;
339 		}
340 		return false;
341 	}
