24  package org.hibernate.test.hql;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.cfg.Configuration;
30  import org.hibernate.cfg.Environment;
31  import org.hibernate.hql.internal.classic.ClassicQueryTranslatorFactory;
32  
33  /**
34   * Some simple test queries using the classic translator explicitly
35   * to ensure that code is not broken in changes for the new translator.
36   * <p/>
37   * Only really checking translation and syntax, not results.
38   *
39   * @author Steve Ebersole
40   */
41  public class ClassicTranslatorTest extends QueryTranslatorTestCase {
42  	@Override
43  	public void configure(Configuration cfg) {
44  		super.configure( cfg );
45  		cfg.setProperty( Environment.QUERY_TRANSLATOR, ClassicQueryTranslatorFactory.class.getName() );
46  	}
47  
48  	@Override
49  	public boolean createSchema() {
50  		return true;
51  	}
52  
53  	@Override
54  	public boolean rebuildSessionFactoryOnError() {
55  		return true;
56  	}
57  
58  	@Test
59  	public void testQueries() {
60  		Session session = openSession();
61  		session.beginTransaction();
62  
63  		session.createQuery( "from Animal" ).list();
64  
65  		session.createQuery( "select a from Animal as a" ).list();
66  		session.createQuery( "select a.mother from Animal as a" ).list();
67  		session.createQuery( "select m from Animal as a inner join a.mother as m" ).list();
68  		session.createQuery( "select a from Animal as a inner join fetch a.mother" ).list();
69  
70  		session.createQuery( "from Animal as a where a.description = ?" ).setString( 0, "jj" ).list();
71  		session.createQuery( "from Animal as a where a.description = :desc" ).setString( "desc", "jr" ).list();
72  		session.createQuery( "from Animal as a where a.description = ? or a.description = :desc" )
73  				.setString( 0, "jj" )
74  				.setString( "desc", "jr" )
75  				.list();
76  
77  		session.getTransaction().commit();
78  		session.close();
79  	}
80  }
