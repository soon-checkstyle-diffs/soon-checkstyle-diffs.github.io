25  package org.hibernate.tool.hbm2ddl;
26  
27  import java.sql.Connection;
28  import java.sql.SQLException;
29  
30  import org.hibernate.engine.jdbc.spi.SqlExceptionHelper;
31  
32  /**
33   * A {@link ConnectionHelper} implementation based on an explicitly supplied
34   * connection.
35   *
36   * @author Steve Ebersole
37   */
38  class SuppliedConnectionHelper implements ConnectionHelper {
39  	private Connection connection;
40  	private boolean toggleAutoCommit;
41  
42  	public SuppliedConnectionHelper(Connection connection) {
43  		this.connection = connection;
44  	}
45  
46  	public void prepare(boolean needsAutoCommit) throws SQLException {
47  		toggleAutoCommit = needsAutoCommit && !connection.getAutoCommit();
48  		if ( toggleAutoCommit ) {
49  			try {
50  				connection.commit();
51  			}
52  			catch( Throwable ignore ) {
53  				// might happen with a managed connection
54  			}
55  			connection.setAutoCommit( true );
56  		}
57  	}
58  
59  	public Connection getConnection() {
60  		return connection;
61  	}
62  
63  	public void release() throws SQLException {
64  		new SqlExceptionHelper().logAndClearWarnings( connection );
65  		if ( toggleAutoCommit ) {
66  			connection.setAutoCommit( false );
67  		}
68  		connection = null;
69  	}
70  }
