24  package org.hibernate.metamodel.relational;
25  
26  import org.jboss.logging.Logger;
27  
28  import org.hibernate.internal.CoreMessageLogger;
29  import org.hibernate.metamodel.ValidationException;
30  
31  /**
32   * Basic support for {@link SimpleValue} implementations.
33   *
34   * @author Steve Ebersole
35   */
36  public abstract class AbstractSimpleValue implements SimpleValue {
37  
38      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, AbstractSimpleValue.class.getName());
39  
40  	private final TableSpecification table;
41  	private final int position;
42  	private Datatype datatype;
43  
44  	protected AbstractSimpleValue(TableSpecification table, int position) {
45  		this.table = table;
46  		this.position = position;
47  	}
48  
49  	@Override
50  	public TableSpecification getTable() {
51  		return table;
52  	}
53  
54  	public int getPosition() {
55  		return position;
56  	}
57  
58  	@Override
59  	public Datatype getDatatype() {
60  		return datatype;
61  	}
62  
63  	@Override
64  	public void setDatatype(Datatype datatype) {
65  		LOG.debugf( "setting datatype for column %s : %s", toLoggableString(), datatype );
66  		if ( this.datatype != null && ! this.datatype.equals( datatype ) ) {
67  			LOG.debugf( "overriding previous datatype : %s", this.datatype );
68  		}
69  		this.datatype = datatype;
70  	}
71  
72  	@Override
73  	public void validateJdbcTypes(JdbcCodes typeCodes) {
74  		// todo : better compatibility testing...
75  		if ( datatype.getTypeCode() != typeCodes.nextJdbcCde() ) {
76  			throw new ValidationException( "Mismatched types" );
77  		}
78  	}
79  }
