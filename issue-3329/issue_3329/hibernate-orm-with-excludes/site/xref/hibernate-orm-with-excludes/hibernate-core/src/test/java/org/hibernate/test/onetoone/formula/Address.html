2   package org.hibernate.test.onetoone.formula;
3   import java.io.Serializable;
4   
5   /**
6    * @author Gavin King
7    */
8   public class Address implements Serializable {
9   	private Person person;
10  	private String type;
11  	private String street;
12  	private String state;
13  	private String zip;
14  	
15  	public Person getPerson() {
16  		return person;
17  	}
18  	public void setPerson(Person person) {
19  		this.person = person;
20  	}
21  	public String getState() {
22  		return state;
23  	}
24  	public void setState(String state) {
25  		this.state = state;
26  	}
27  	public String getStreet() {
28  		return street;
29  	}
30  	public void setStreet(String street) {
31  		this.street = street;
32  	}
33  	public String getType() {
34  		return type;
35  	}
36  	public void setType(String type) {
37  		this.type = type;
38  	}
39  	public String getZip() {
40  		return zip;
41  	}
42  	public void setZip(String zip) {
43  		this.zip = zip;
44  	}
45  	
46  	public boolean equals(Object that) {
47  		if ( !(that instanceof Address) ) return false;
48  		Address address = (Address) that;
49  		return address.getType().equals(type) && 
50  			address.getPerson().getName().equals( person.getName() );
51  	}
52  	
53  	public int hashCode() {
54  		return person.getName().hashCode() + type.hashCode();
55  	}
56  }
