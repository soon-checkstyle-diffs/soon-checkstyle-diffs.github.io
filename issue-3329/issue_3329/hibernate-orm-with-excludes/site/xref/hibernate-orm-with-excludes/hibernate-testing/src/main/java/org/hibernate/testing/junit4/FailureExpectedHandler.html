24  package org.hibernate.testing.junit4;
25  
26  import org.jboss.logging.Logger;
27  import org.junit.runners.model.FrameworkMethod;
28  import org.junit.runners.model.Statement;
29  
30  import org.hibernate.testing.FailureExpected;
31  
32  /**
33  * @author Steve Ebersole
34  */
35  class FailureExpectedHandler extends Statement {
36  	private static final Logger log = Logger.getLogger( FailureExpectedHandler.class );
37  
38  	private final TestClassMetadata testClassMetadata;
39  	private final ExtendedFrameworkMethod extendedFrameworkMethod;
40  	private final Statement realInvoker;
41  	private final Object testInstance;
42  
43  	public FailureExpectedHandler(
44  			Statement realInvoker,
45  			TestClassMetadata testClassMetadata,
46  			ExtendedFrameworkMethod extendedFrameworkMethod,
47  			Object testInstance) {
48  		this.realInvoker = realInvoker;
49  		this.testClassMetadata = testClassMetadata;
50  		this.extendedFrameworkMethod = extendedFrameworkMethod;
51  		this.testInstance = testInstance;
52  	}
53  
54  	@Override
55  	public void evaluate() throws Throwable {
56  		final FailureExpected failureExpected = extendedFrameworkMethod.getFailureExpectedAnnotation();
57  		try {
58  			realInvoker.evaluate();
59  			// reaching here is expected, unless the test is marked as an expected failure
60  			if ( failureExpected != null ) {
61  				throw new FailureExpectedTestPassedException( extendedFrameworkMethod );
62  			}
63  		}
64  		catch (FailureExpectedTestPassedException e) {
65  			// just pass this along
66  			throw e;
67  		}
68  		catch (Throwable e) {
69  			// on error handling is very different based on whether the test was marked as an expected failure
70  			if ( failureExpected != null ) {
71  
72  				// handle the expected failure case
73  				log.infof(
74  						"Ignoring expected failure [{}] : {}",
75  						Helper.extractTestName( extendedFrameworkMethod ),
76  						Helper.extractMessage( failureExpected )
77  				);
78  				testClassMetadata.performOnExpectedFailureCallback( testInstance );
79  				// most importantly, do not propagate exception...
80  			}
81  			else {
82  				// handle the non-expected failure case
83  				testClassMetadata.performOnFailureCallback( testInstance );
84  				throw e;
85  			}
86  		}
87  	}
88  
89  	public static class FailureExpectedTestPassedException extends Exception {
90  		public FailureExpectedTestPassedException(FrameworkMethod frameworkMethod) {
91  			super( "Test marked as FailureExpected, but did not fail : " + Helper.extractTestName( frameworkMethod ) );
92  		}
93  	}
94  }
