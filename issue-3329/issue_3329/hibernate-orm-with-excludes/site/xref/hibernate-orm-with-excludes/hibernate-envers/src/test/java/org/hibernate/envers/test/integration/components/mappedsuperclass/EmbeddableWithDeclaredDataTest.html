26  import java.util.List;
27  import javax.persistence.EntityManager;
28  import javax.persistence.EntityTransaction;
29  
30  import org.junit.Assert;
31  import org.junit.Test;
32  
33  import org.hibernate.envers.AuditReader;
34  import org.hibernate.envers.AuditReaderFactory;
35  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
36  import org.hibernate.envers.test.Priority;
37  import org.hibernate.testing.FailureExpected;
38  import org.hibernate.testing.TestForIssue;
39  
40  /**
41   * @author Jakob Braeuchi.
42   * @author Gail Badner
43   */
44  @TestForIssue(jiraKey = "HHH-9193")
45  public class EmbeddableWithDeclaredDataTest extends BaseEnversJPAFunctionalTestCase {
46  	private long id;
47  
48  	@Override
49  	protected Class<?>[] getAnnotatedClasses() {
50  		return new Class[] { EntityWithEmbeddableWithDeclaredData.class, AbstractEmbeddable.class, EmbeddableWithDeclaredData.class };
51  	}
52  
53  	@Test
54  	@Priority(10)
55  	public void initData() {
56  		EntityManager em = getEntityManager();
57  
58  		EntityWithEmbeddableWithDeclaredData entity = new EntityWithEmbeddableWithDeclaredData();
59  		entity.setName( "Entity 1" );
60  		entity.setValue( new EmbeddableWithDeclaredData( 42, "TestCodeart" ) );
61  
62  		EntityTransaction tx = em.getTransaction();
63  		tx.begin();
64  		em.persist( entity );
65  		tx.commit();
66  		em.close();
67  
68  		id = entity.getId();
69  	}
70  
71  	@Test
72  	@FailureExpected( jiraKey = "HHH-9193" )
73  	public void testEmbeddableThatExtendsMappedSuperclass() {
74  
75  		// Reload and Compare Revision
76  		EntityManager em = getEntityManager();
77  		EntityWithEmbeddableWithDeclaredData entityLoaded = em.find( EntityWithEmbeddableWithDeclaredData.class, id );
78  
79  		AuditReader reader = AuditReaderFactory.get( em );
80  
81  		List<Number> revs = reader.getRevisions( EntityWithEmbeddableWithDeclaredData.class, id );
82  		Assert.assertEquals( 1, revs.size() );
83  
84  		EntityWithEmbeddableWithDeclaredData entityRev1 = reader.find( EntityWithEmbeddableWithDeclaredData.class, id, revs.get( 0 ) );
85  
86  		Assert.assertEquals( entityLoaded.getName(), entityRev1.getName() );
87  
88  		// only value.codeArt should be audited because it is the only audited field in EmbeddableWithDeclaredData;
89  		// fields in AbstractEmbeddable should not be audited.
90  		Assert.assertEquals( entityLoaded.getValue().getCodeart(), entityRev1.getValue().getCodeart() );
91  		Assert.assertNull( entityRev1.getValue().getCode() );
92  	}
93  }
