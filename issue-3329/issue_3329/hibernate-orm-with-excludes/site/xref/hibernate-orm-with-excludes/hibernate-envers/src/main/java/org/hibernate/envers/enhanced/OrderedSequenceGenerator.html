1   package org.hibernate.envers.enhanced;

3   import org.hibernate.HibernateException;
4   import org.hibernate.dialect.Dialect;
5   import org.hibernate.dialect.Oracle8iDialect;
6   import org.hibernate.id.enhanced.SequenceStyleGenerator;
7   import org.hibernate.internal.util.StringHelper;
8   
9   /**
10   * Revision number generator has to produce values in ascending order (gaps may occur).
11   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
12   */
13  public class OrderedSequenceGenerator extends SequenceStyleGenerator {
14  	@Override
15  	public String[] sqlCreateStrings(Dialect dialect) throws HibernateException {
16  		String[] create = super.sqlCreateStrings( dialect );
17  		if ( dialect instanceof Oracle8iDialect ) {
18  			// Make sure that sequence produces increasing values in Oracle RAC environment.
19  			create = StringHelper.suffix( create, " order" );
20  		}
21  		return create;
22  	}
23  }
