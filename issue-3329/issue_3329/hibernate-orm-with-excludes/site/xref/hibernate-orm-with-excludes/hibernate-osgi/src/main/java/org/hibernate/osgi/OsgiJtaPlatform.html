21  package org.hibernate.osgi;
22  
23  import javax.transaction.Synchronization;
24  import javax.transaction.SystemException;
25  import javax.transaction.Transaction;
26  import javax.transaction.TransactionManager;
27  import javax.transaction.UserTransaction;
28  
29  import org.hibernate.TransactionException;
30  import org.hibernate.engine.transaction.internal.jta.JtaStatusHelper;
31  import org.hibernate.service.jta.platform.spi.JtaPlatform;
32  import org.osgi.framework.BundleContext;
33  import org.osgi.framework.ServiceReference;
34  
35  /**
36   * Offers the JTA Platform provided by the OSGi container. The Enterprise
37   * OSGi spec requires all containers to register UserTransaction
38   * and TransactionManager OSGi services.
39   * 
40   * @author Brett Meyer
41   */
42  public class OsgiJtaPlatform implements JtaPlatform {
43  
44  	private static final long serialVersionUID = 1L;
45  	
46  	private BundleContext bundleContext;
47  
48  	public OsgiJtaPlatform(BundleContext bundleContext) {
49  		this.bundleContext = bundleContext;
50  	}
51  
52  	@Override
53  	public TransactionManager retrieveTransactionManager() {
54  		final ServiceReference sr = bundleContext.getServiceReference( TransactionManager.class.getName() );
55  		return sr == null ? null : (TransactionManager) bundleContext.getService( sr );
56  	}
57  
58  	@Override
59  	public UserTransaction retrieveUserTransaction() {
60  		final ServiceReference sr = bundleContext.getServiceReference( UserTransaction.class.getName() );
61  		return sr == null ? null : (UserTransaction) bundleContext.getService( sr );
62  	}
63  
64  	@Override
65  	public Object getTransactionIdentifier(Transaction transaction) {
66  		// AbstractJtaPlatform just uses the transaction itself.
67  		return transaction;
68  	}
69  
70  	@Override
71  	public boolean canRegisterSynchronization() {
72  		return JtaStatusHelper.isActive( retrieveTransactionManager() );
73  	}
74  
75  	@Override
76  	public void registerSynchronization(Synchronization synchronization) {
77  		try {
78  			retrieveTransactionManager().getTransaction().registerSynchronization( synchronization );
79  		}
80  		catch (Exception e) {
81  			throw new TransactionException( "Could not obtain transaction from OSGi services!" );
82  		}
83  	}
84  
85  	@Override
86  	public int getCurrentStatus() throws SystemException {
87  		return retrieveTransactionManager().getStatus();
88  	}
89  
90  }
