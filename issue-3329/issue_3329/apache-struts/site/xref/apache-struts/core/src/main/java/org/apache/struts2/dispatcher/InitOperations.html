28  import java.util.*;
29  import java.util.regex.Pattern;

31  /**
32   * Contains initialization operations
33   */
34  public class InitOperations {
35  
36      public InitOperations() {
37      }
38  
39      /**
40       * Initializes the internal Struts logging
41       *
42       * @param filterConfig host configuration
43       * @deprecated since 2.5
44       */
45      @Deprecated
46      public void initLogging( HostConfig filterConfig ) {
47          String factoryName = filterConfig.getInitParameter("loggerFactory");
48          if (factoryName != null) {
49              try {
50                  Class cls = ClassLoaderUtil.loadClass(factoryName, this.getClass());
51                  LoggerFactory fac = (LoggerFactory) cls.newInstance();
52                  LoggerFactory.setLoggerFactory(fac);
53              } catch ( InstantiationException e ) {
54                  System.err.println("Unable to instantiate logger factory: " + factoryName + ", using default");
55                  e.printStackTrace();
56              } catch ( IllegalAccessException e ) {
57                  System.err.println("Unable to access logger factory: " + factoryName + ", using default");
58                  e.printStackTrace();
59              } catch ( ClassNotFoundException e ) {
60                  System.err.println("Unable to locate logger factory class: " + factoryName + ", using default");
61                  e.printStackTrace();
62              }
63          }
64      }
65  
66      /**
67       * Creates and initializes the dispatcher
68       *
69       * @param filterConfig host configuration
70       *
71       * @return the dispatcher
72       */
73      public Dispatcher initDispatcher( HostConfig filterConfig ) {
74          Dispatcher dispatcher = createDispatcher(filterConfig);
75          dispatcher.init();
76          return dispatcher;
77      }
78  
79      /**
80       * Initializes the static content loader with the filter configuration
81       *
82       * @param filterConfig host configuration
83       * @param dispatcher the dispatcher
84       * @return the static content loader
85       */
86      public StaticContentLoader initStaticContentLoader( HostConfig filterConfig, Dispatcher dispatcher ) {
87          StaticContentLoader loader = dispatcher.getContainer().getInstance(StaticContentLoader.class);
88          loader.setHostConfig(filterConfig);
89          return loader;
90      }
91  
92      /**
93       * @return The dispatcher on the thread.
94       *
95       * @throws IllegalStateException If there is no dispatcher available
96       */
97      public Dispatcher findDispatcherOnThread() {
98          Dispatcher dispatcher = Dispatcher.getInstance();
99          if (dispatcher == null) {
100             throw new IllegalStateException("Must have the StrutsPrepareFilter execute before this one");
101         }
102         return dispatcher;
103     }
104 
105     /**
106      * Create a {@link Dispatcher}
107      *
108      * @param filterConfig host configuration
109      *
110      * @return The dispatcher on the thread.
111      */
112     protected Dispatcher createDispatcher(HostConfig filterConfig) {
113         Map<String, String> params = new HashMap<>();
114         for ( Iterator e = filterConfig.getInitParameterNames(); e.hasNext(); ) {
115             String name = (String) e.next();
116             String value = filterConfig.getInitParameter(name);
117             params.put(name, value);
118         }
119         return new Dispatcher(filterConfig.getServletContext(), params);
120     }
121 
122     public void cleanup() {
123         ActionContext.setContext(null);
124     }
125 
126     /**
127      * Extract a list of patterns to exclude from request filtering
128      *
129      * @param dispatcher The dispatcher to check for exclude pattern configuration
130      *
131      * @return a List of Patterns for request to exclude if apply, or <tt>null</tt>
132      *
133      * @see org.apache.struts2.StrutsConstants#STRUTS_ACTION_EXCLUDE_PATTERN
134      */
135     public List<Pattern> buildExcludedPatternsList( Dispatcher dispatcher ) {
136         return buildExcludedPatternsList(dispatcher.getContainer().getInstance(String.class, StrutsConstants.STRUTS_ACTION_EXCLUDE_PATTERN));
137     }
138             
139     private List<Pattern> buildExcludedPatternsList( String patterns ) {
140         if (null != patterns && patterns.trim().length() != 0) {
141             List<Pattern> list = new ArrayList<>();
142             String[] tokens = patterns.split(",");
143             for ( String token : tokens ) {
144                 list.add(Pattern.compile(token.trim()));
145             }
146             return Collections.unmodifiableList(list);
147         } else {
148             return null;
149         }
150     }
