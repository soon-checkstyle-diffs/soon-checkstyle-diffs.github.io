17  package org.orekit.utils;
18  
19  import java.io.Serializable;
20  
21  import org.hipparchus.RealFieldElement;
22  import org.hipparchus.geometry.euclidean.threed.FieldRotation;
23  import org.hipparchus.geometry.euclidean.threed.FieldVector3D;
24  import org.hipparchus.geometry.euclidean.threed.RotationConvention;
25  import org.orekit.time.TimeShiftable;

27  /** Simple container for FieldRotation<T>/FieldRotation<T> rate pairs, using {@link RealFieldElement}.
28   * <p>
29   * The state can be slightly shifted to close dates. This shift is based on
30   * a simple quadratic model. It is <em>not</em> intended as a replacement for
31   * proper attitude propagation but should be sufficient for either small
32   * time shifts or coarse accuracy.
33   * </p>
34   * <p>
35   * This class is the angular counterpart to {@link FieldPVCoordinates}.
36   * </p>
37   * <p>Instances of this class are guaranteed to be immutable.</p>
38   * @param <T> the type of the field elements
39   * @author Luc Maisonobe
40   * @since 6.0
41   * @see AngularCoordinates
42   */
43  public class FieldAngularCoordinates<T extends RealFieldElement<T>>
44       implements TimeShiftable<FieldAngularCoordinates<T>>, Serializable {
45  
46      /** Serializable UID. */
47      private static final long serialVersionUID = 20140414L;
48  
49      /** FieldRotation<T>. */
50      private final FieldRotation<T> rotation;
51  
52      /** FieldRotation<T> rate. */
53      private final FieldVector3D<T> rotationRate;
54  
55      /** FieldRotation<T> acceleration. */
56      private final FieldVector3D<T> rotationAcceleration;
57  
58      /** Builds a rotation/rotation rate pair.
59       * @param rotation rotation
60       * @param rotationRate rotation rate Ω (rad/s)
61       */
62      public FieldAngularCoordinates(final FieldRotation<T> rotation, final FieldVector3D<T> rotationRate) {
63          this(rotation, rotationRate,
64               new FieldVector3D<T>(rotation.getQ0().getField().getZero(),
65                                    rotation.getQ0().getField().getZero(),
66                                    rotation.getQ0().getField().getZero()));
67      }
68  
69      /** Builds a FieldRotation<T>/FieldRotation<T> rate/FieldRotation<T> acceleration triplet.
70       * @param rotation FieldRotation<T>
71       * @param rotationRate FieldRotation<T> rate Ω (rad/s)
72       * @param rotationAcceleration FieldRotation<T> acceleration dΩ/dt (rad²/s²)
73       */
74      public FieldAngularCoordinates(final FieldRotation<T> rotation, final FieldVector3D<T> rotationRate,
75                                     final FieldVector3D<T> rotationAcceleration) {
76          this.rotation             = rotation;
77          this.rotationRate         = rotationRate;
78          this.rotationAcceleration = rotationAcceleration;
79      }
80  
81      /** Estimate FieldRotation<T> rate between two orientations.
82       * <p>Estimation is based on a simple fixed rate FieldRotation<T>
83       * during the time interval between the two orientations.</p>
84       * @param start start orientation
85       * @param end end orientation
86       * @param dt time elapsed between the dates of the two orientations
87       * @param <T> the type of the field elements
88       * @return FieldRotation<T> rate allowing to go from start to end orientations
89       */
90      public static <T extends RealFieldElement<T>>
91          FieldVector3D<T> estimateRate(final FieldRotation<T> start,
92                                        final FieldRotation<T> end,
93                                        final double dt) {
94          final FieldRotation<T> evolution = start.compose(end.revert(), RotationConvention.VECTOR_OPERATOR);
95          return new FieldVector3D<T>(evolution.getAngle().divide(dt),
96                                      evolution.getAxis(RotationConvention.VECTOR_OPERATOR));
97      }
98  
99      /** Revert a FieldRotation<T>/FieldRotation<T>/FieldRotation<T> acceleration triplet.
100      * Build a triplet which reverse the effect of another triplet.
101      * @return a new triplet whose effect is the reverse of the effect
102      * of the instance
103      */
104     public FieldAngularCoordinates<T> revert() {
105         return new FieldAngularCoordinates<T>(rotation.revert(),
106                                               rotation.applyInverseTo(rotationRate.negate()),
107                                               rotation.applyInverseTo(rotationAcceleration.negate()));
108     }
109 
110     /** Get a time-shifted state.
111      * <p>
112      * The state can be slightly shifted to close dates. This shift is based on
113      * a simple quadratic model. It is <em>not</em> intended as a replacement for
114      * proper attitude propagation but should be sufficient for either small
115      * time shifts or coarse accuracy.
116      * </p>
117      * @param dt time shift in seconds
118      * @return a new state, shifted with respect to the instance (which is immutable)
119      */
120     public FieldAngularCoordinates<T> shiftedBy(final double dt) {
121 
122         // the shiftedBy method is based on a local approximation.
123         // It considers separately the contribution of the constant
124         // rotation, the linear contribution or the rate and the
125         // quadratic contribution of the acceleration. The rate
126         // and acceleration contributions are small rotations as long
127         // as the time shift is small, which is the crux of the algorithm.
128         // Small rotations are almost commutative, so we append these small
129         // contributions one after the other, as if they really occurred
130         // successively, despite this is not what really happens.
131 
132         // compute the linear contribution first, ignoring acceleration
133         // BEWARE: there is really a minus sign here, because if
134         // the target frame rotates in one direction, the vectors in the origin
135         // frame seem to rotate in the opposite direction
136         final T rate = rotationRate.getNorm();
137         final T zero = rate.getField().getZero();
138         final T one  = rate.getField().getOne();
139         final FieldRotation<T> rateContribution = (rate.getReal() == 0.0) ?
140                                                   new FieldRotation<T>(one, zero, zero, zero, false) :
141                                                   new FieldRotation<T>(rotationRate,
142                                                                        rate.multiply(dt),
143                                                                        RotationConvention.FRAME_TRANSFORM);
144 
145         // append rotation and rate contribution
146         final FieldAngularCoordinates<T> linearPart =
147                 new FieldAngularCoordinates<T>(rateContribution.compose(rotation, RotationConvention.VECTOR_OPERATOR),
148                                                rotationRate);
149 
150         final T acc  = rotationAcceleration.getNorm();
151         if (acc.getReal() == 0.0) {
152             // no acceleration, the linear part is sufficient
153             return linearPart;
154         }
155 
156         // compute the quadratic contribution, ignoring initial rotation and rotation rate
157         // BEWARE: there is really a minus sign here, because if
158         // the target frame rotates in one direction, the vectors in the origin
159         // frame seem to rotate in the opposite direction
160         final FieldAngularCoordinates<T> quadraticContribution =
161                 new FieldAngularCoordinates<T>(new FieldRotation<T>(rotationAcceleration,
162                                                                     acc.multiply(0.5 * dt * dt),
163                                                                     RotationConvention.FRAME_TRANSFORM),
164                                                new FieldVector3D<T>(dt, rotationAcceleration),
165                                                rotationAcceleration);
166 
167         // the quadratic contribution is a small rotation:
168         // its initial angle and angular rate are both zero.
169         // small rotations are almost commutative, so we append the small
170         // quadratic part after the linear part as a simple offset
171         return quadraticContribution.addOffset(linearPart);
172 
173     }

175     /** Get the FieldRotation<T>.
176      * @return the FieldRotation<T>.
177      */
178     public FieldRotation<T> getRotation() {
179         return rotation;
180     }
181 
182     /** Get the FieldRotation<T> rate.
183      * @return the FieldRotation<T> rate vector (rad/s).
184      */
185     public FieldVector3D<T> getRotationRate() {
186         return rotationRate;
187     }
188 
189     /** Get the rotation acceleration.
190      * @return the rotation acceleration vector dΩ/dt (rad²/s²).
191      */
192     public FieldVector3D<T> getRotationAcceleration() {
193         return rotationAcceleration;
194     }
195 
196     /** Add an offset from the instance.
197      * <p>
198      * We consider here that the offset FieldRotation<T> is applied first and the
199      * instance is applied afterward. Note that angular coordinates do <em>not</em>
200      * commute under this operation, i.e. {@code a.addOffset(b)} and {@code
201      * b.addOffset(a)} lead to <em>different</em> results in most cases.
202      * </p>
203      * <p>
204      * The two methods {@link #addOffset(FieldAngularCoordinates) addOffset} and
205      * {@link #subtractOffset(FieldAngularCoordinates) subtractOffset} are designed
206      * so that round trip applications are possible. This means that both {@code
207      * ac1.subtractOffset(ac2).addOffset(ac2)} and {@code
208      * ac1.addOffset(ac2).subtractOffset(ac2)} return angular coordinates equal to ac1.
209      * </p>
210      * @param offset offset to subtract
211      * @return new instance, with offset subtracted
212      * @see #subtractOffset(FieldAngularCoordinates)
213      */
214     public FieldAngularCoordinates<T> addOffset(final FieldAngularCoordinates<T> offset) {
215         final FieldVector3D<T> rOmega    = rotation.applyTo(offset.rotationRate);
216         final FieldVector3D<T> rOmegaDot = rotation.applyTo(offset.rotationAcceleration);
217         return new FieldAngularCoordinates<T>(rotation.compose(offset.rotation, RotationConvention.VECTOR_OPERATOR),
218                                               rotationRate.add(rOmega),
219                                               new FieldVector3D<T>( 1.0, rotationAcceleration,
220                                                                     1.0, rOmegaDot,
221                                                                    -1.0, FieldVector3D.crossProduct(rotationRate, rOmega)));
222     }
223 
224     /** Subtract an offset from the instance.
225      * <p>
226      * We consider here that the offset Rotation is applied first and the
227      * instance is applied afterward. Note that angular coordinates do <em>not</em>
228      * commute under this operation, i.e. {@code a.subtractOffset(b)} and {@code
229      * b.subtractOffset(a)} lead to <em>different</em> results in most cases.
230      * </p>
231      * <p>
232      * The two methods {@link #addOffset(FieldAngularCoordinates) addOffset} and
233      * {@link #subtractOffset(FieldAngularCoordinates) subtractOffset} are designed
234      * so that round trip applications are possible. This means that both {@code
235      * ac1.subtractOffset(ac2).addOffset(ac2)} and {@code
236      * ac1.addOffset(ac2).subtractOffset(ac2)} return angular coordinates equal to ac1.
237      * </p>
238      * @param offset offset to subtract
239      * @return new instance, with offset subtracted
240      * @see #addOffset(FieldAngularCoordinates)
241      */
242     public FieldAngularCoordinates<T> subtractOffset(final FieldAngularCoordinates<T> offset) {
243         return addOffset(offset.revert());
244     }
245 
246     /** Convert to a regular angular coordinates.
247      * @return a regular angular coordinates
248      */
249     public AngularCoordinates toAngularCoordinates() {
250         return new AngularCoordinates(rotation.toRotation(), rotationRate.toVector3D(),
251                                       rotationAcceleration.toVector3D());
252     }
