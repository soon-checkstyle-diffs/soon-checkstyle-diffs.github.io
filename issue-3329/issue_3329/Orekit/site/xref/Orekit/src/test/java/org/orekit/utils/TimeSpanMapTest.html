28      @Test
29      public void testSingleEntry() {
30          String single = "single";
31          TimeSpanMap<String> map = new TimeSpanMap<String>(single);
32          Assert.assertSame(single, map.get(AbsoluteDate.CCSDS_EPOCH));
33          Assert.assertSame(single, map.get(AbsoluteDate.FIFTIES_EPOCH));
34          Assert.assertSame(single, map.get(AbsoluteDate.FUTURE_INFINITY));
35          Assert.assertSame(single, map.get(AbsoluteDate.GALILEO_EPOCH));
36          Assert.assertSame(single, map.get(AbsoluteDate.GPS_EPOCH));
37          Assert.assertSame(single, map.get(AbsoluteDate.J2000_EPOCH));
38          Assert.assertSame(single, map.get(AbsoluteDate.JAVA_EPOCH));
39          Assert.assertSame(single, map.get(AbsoluteDate.JULIAN_EPOCH));
40          Assert.assertSame(single, map.get(AbsoluteDate.MODIFIED_JULIAN_EPOCH));
41          Assert.assertSame(single, map.get(AbsoluteDate.PAST_INFINITY));
42      }

44      @Test
45      public void testForwardAdd() {
46          final AbsoluteDate ref = AbsoluteDate.J2000_EPOCH;
47          TimeSpanMap<Integer> map = new TimeSpanMap<Integer>(Integer.valueOf(0));
48          for (int i = 1; i < 100; ++i) {
49              map.addValidAfter(Integer.valueOf(i), ref.shiftedBy(i));
50          }
51          Assert.assertEquals(0, map.get(ref.shiftedBy(-1000.0)).intValue());
52          Assert.assertEquals(0, map.get(ref.shiftedBy( -100.0)).intValue());
53          for (int i = 0; i < 100; ++i) {
54              Assert.assertEquals(i, map.get(ref.shiftedBy(i + 0.1)).intValue());
55              Assert.assertEquals(i, map.get(ref.shiftedBy(i + 0.9)).intValue());
56          }
57          Assert.assertEquals(99, map.get(ref.shiftedBy(  100.0)).intValue());
58          Assert.assertEquals(99, map.get(ref.shiftedBy( 1000.0)).intValue());
59      }
60  
61      @Test
62      public void testBackwardAdd() {
63          final AbsoluteDate ref = AbsoluteDate.J2000_EPOCH;
64          TimeSpanMap<Integer> map = new TimeSpanMap<Integer>(Integer.valueOf(0));
65          for (int i = -1; i > -100; --i) {
66              map.addValidBefore(Integer.valueOf(i), ref.shiftedBy(i));
67          }
68          Assert.assertEquals(0, map.get(ref.shiftedBy( 1000.0)).intValue());
69          Assert.assertEquals(0, map.get(ref.shiftedBy(  100.0)).intValue());
70          for (int i = 0; i > -100; --i) {
71              Assert.assertEquals(i, map.get(ref.shiftedBy(i - 0.1)).intValue());
72              Assert.assertEquals(i, map.get(ref.shiftedBy(i - 0.9)).intValue());
73          }
74          Assert.assertEquals(-99, map.get(ref.shiftedBy( -100.0)).intValue());
75          Assert.assertEquals(-99, map.get(ref.shiftedBy(-1000.0)).intValue());
76      }

78      @Test
79      public void testRandomAdd() {
80          final AbsoluteDate ref = AbsoluteDate.J2000_EPOCH;
81          TimeSpanMap<Integer> map = new TimeSpanMap<Integer>(Integer.valueOf(0));
82          map.addValidAfter(Integer.valueOf(10), ref.shiftedBy(10.0));
83          map.addValidAfter(Integer.valueOf( 3), ref.shiftedBy( 2.0));
84          map.addValidAfter(Integer.valueOf( 9), ref.shiftedBy( 5.0));
85          map.addValidBefore(Integer.valueOf( 2), ref.shiftedBy( 3.0));
86          map.addValidBefore(Integer.valueOf( 5), ref.shiftedBy( 9.0));
87          Assert.assertEquals( 0, map.get(ref.shiftedBy( -1.0)).intValue());
88          Assert.assertEquals( 0, map.get(ref.shiftedBy(  1.9)).intValue());
89          Assert.assertEquals( 2, map.get(ref.shiftedBy(  2.1)).intValue());
90          Assert.assertEquals( 2, map.get(ref.shiftedBy(  2.9)).intValue());
91          Assert.assertEquals( 3, map.get(ref.shiftedBy(  3.1)).intValue());
92          Assert.assertEquals( 3, map.get(ref.shiftedBy(  4.9)).intValue());
93          Assert.assertEquals( 5, map.get(ref.shiftedBy(  5.1)).intValue());
94          Assert.assertEquals( 5, map.get(ref.shiftedBy(  8.9)).intValue());
95          Assert.assertEquals( 9, map.get(ref.shiftedBy(  9.1)).intValue());
96          Assert.assertEquals( 9, map.get(ref.shiftedBy(  9.9)).intValue());
97          Assert.assertEquals(10, map.get(ref.shiftedBy( 10.1)).intValue());
98          Assert.assertEquals(10, map.get(ref.shiftedBy(100.0)).intValue());
99      }
