55      public void testCalendarWeek() {
56          Assert.assertEquals(52, new DateComponents(1995,  1,  1).getCalendarWeek());
57          Assert.assertEquals(52, new DateComponents(1996, 12, 29).getCalendarWeek());
58          Assert.assertEquals( 1, new DateComponents(1996, 12, 30).getCalendarWeek());
59          Assert.assertEquals( 1, new DateComponents(1997,  1,  5).getCalendarWeek());
60          Assert.assertEquals(52, new DateComponents(1997, 12, 28).getCalendarWeek());
61          Assert.assertEquals( 1, new DateComponents(1997, 12, 29).getCalendarWeek());
62          Assert.assertEquals( 1, new DateComponents(1998,  1,  4).getCalendarWeek());
63          Assert.assertEquals( 2, new DateComponents(1998,  1,  5).getCalendarWeek());
64          Assert.assertEquals(52, new DateComponents(1998, 12, 27).getCalendarWeek());
65          Assert.assertEquals(53, new DateComponents(1998, 12, 28).getCalendarWeek());
66          Assert.assertEquals(53, new DateComponents(1999,  1,  3).getCalendarWeek());
67          Assert.assertEquals( 1, new DateComponents(1999,  1,  4).getCalendarWeek());
68          DateComponents[] firstWeekMonday = new DateComponents[502];
69          for (int i = 0; i < firstWeekMonday.length; ++i) {
70              firstWeekMonday[i] = firstWeekMonday(1599 + i);
71          }
72          int startDay = firstWeekMonday[0].getJ2000Day();
73          int endDay   = firstWeekMonday[firstWeekMonday.length - 1].getJ2000Day();
74          int index    = 0;
75          for (int day = startDay; day < endDay; ++day) {
76              DateComponents d = new DateComponents(day);
77              if (firstWeekMonday[index + 1].compareTo(d) <= 0) {
78                  ++index;
79              }
80              int delta = d.getJ2000Day() - firstWeekMonday[index].getJ2000Day();
81              Assert.assertEquals(1 + delta / 7, d.getCalendarWeek());
82          }
83      }

118     // poor man's (slow) implementation of first calendar week computation, using ISO rules
119     private DateComponents firstWeekMonday(final int year) {
120         int i = 0;
121         while (true) {
122             DateComponents d = new DateComponents(year, 1, ++i);
123             if (d.getDayOfWeek() == 4) {
124                 // this is the first Thursday of the year, Monday is 3 days before
125                 return new DateComponents(d, -3);
126             }
127         }
128     }

229     @Test
230     public void testComparisons() {
231         DateComponents[][] dates = {
232                 { new DateComponents(2003,  1,  1), new DateComponents(2003,   1) },
233                 { new DateComponents(2003,  2, 28), new DateComponents(2003,  59) },
234                 { new DateComponents(2003,  3,  1), new DateComponents(2003,  60) },
235                 { new DateComponents(2003,  9, 26), new DateComponents(2003, 269) },
236                 { new DateComponents(2003, 12, 31), new DateComponents(2003, 365) },
237                 { new DateComponents(2004,  2, 28), new DateComponents(2004,  59) },
238                 { new DateComponents(2004,  2, 29), new DateComponents(2004,  60) },
239                 { new DateComponents(2004,  3,  1), new DateComponents(2004,  61) },
240                 { new DateComponents(2004, 12, 31), new DateComponents(2004, 366) }
241         };
242         for (int i = 0; i < dates.length; ++i) {
243             for (int j = 0; j < dates.length; ++j) {
244                 if (dates[i][0].compareTo(dates[j][1]) < 0) {
245                     Assert.assertTrue(dates[j][1].compareTo(dates[i][0]) > 0);
246                     Assert.assertFalse(dates[i][0].equals(dates[j][1]));
247                     Assert.assertFalse(dates[j][1].equals(dates[i][0]));
248                     Assert.assertTrue(dates[i][0].hashCode() != dates[j][1].hashCode());
249                     Assert.assertTrue(i < j);
250                 } else if (dates[i][0].compareTo(dates[j][1]) > 0) {
251                     Assert.assertTrue(dates[j][1].compareTo(dates[i][0]) < 0);
252                     Assert.assertFalse(dates[i][0].equals(dates[j][1]));
253                     Assert.assertFalse(dates[j][1].equals(dates[i][0]));
254                     Assert.assertTrue(dates[i][0].hashCode() != dates[j][1].hashCode());
255                     Assert.assertTrue(i > j);
256                 } else {
257                     Assert.assertTrue(dates[j][1].compareTo(dates[i][0]) == 0);
258                     Assert.assertTrue(dates[i][0].equals(dates[j][1]));
259                     Assert.assertTrue(dates[j][1].equals(dates[i][0]));
260                     Assert.assertTrue(dates[i][0].hashCode() == dates[j][1].hashCode());
261                     Assert.assertTrue(i == j);
262                 }
263             }
264         }
265         Assert.assertFalse(dates[0][0].equals(this));
266     }

338     public void testWellFormed() {
339         checkWellFormed(-4800, -4700, -2483687, -2446797);
340         checkWellFormed(   -5,     5,  -732313,  -728296);
341         checkWellFormed( 1580,  1605,  -153392,  -143906);
342         checkWellFormed( 1695,  1705,  -111398,  -107382);
343         checkWellFormed( 1795,  1805,   -74874,   -70858);
344         checkWellFormed( 1895,  1905,   -38350,   -34334);
345         checkWellFormed( 1995,  2005,    -1826,     2191);
346     }

348     public void checkWellFormed(int startYear, int endYear, int startJ2000, int endJ2000) {
349 
350         int[] commonLength = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
351         int[] leapLength   = { 0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
352 
353         int k = startJ2000;
354         for (int year = startYear; year <= endYear; ++year) {
355             for (int month = 0; month < 14; ++month) {
356                 for (int day = 0; day < 33; ++day) {
357 
358                     // ill-formed dates have predictable components
359                     boolean expectedIllFormed = false;
360                     if ((month < 1) || (month > 12)) {
361                         expectedIllFormed = true;
362                     } else if ((year == 1582) && (month == 10) && (day > 4) && (day < 15)) {
363                         expectedIllFormed = true;
364                     } else if ((year < 1582) && (year % 4 == 0)) {
365                         if ((day < 1) || (day > leapLength[month])) {
366                             expectedIllFormed = true;
367                         }
368                     } else if ((year >= 1582) && (year % 4 == 0) &&
369                             ((year % 100 != 0) || (year % 400 == 0))) {
370                         if ((day < 1) || (day > leapLength[month])) {
371                             expectedIllFormed = true;
372                         }
373                     } else {
374                         if ((day < 1) || (day > commonLength[month])) {
375                             expectedIllFormed = true;
376                         }
377                     }
378 
379                     try {
380                         // well-formed dates should have sequential J2000 days
381                         DateComponents date = new DateComponents(year, month, day);
382                         Assert.assertEquals(k++, date.getJ2000Day());
383                         Assert.assertTrue(!expectedIllFormed);
384                     } catch (IllegalArgumentException iae) {
385                         Assert.assertTrue(expectedIllFormed);
386                     }
387                 }
388             }
389         }
390 
391         Assert.assertEquals(endJ2000, --k);
392 
393     }
