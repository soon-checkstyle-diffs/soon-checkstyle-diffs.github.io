83      @Test
84      public void testDuringLeap() {
85          AbsoluteDate d = new AbsoluteDate(new DateComponents(1983, 06, 30),
86                                            new TimeComponents(23, 59, 59),
87                                            utc);
88          Assert.assertEquals("1983-06-30T23:58:59.000", d.shiftedBy(-60).toString(utc));
89          Assert.assertEquals(60, utc.minuteDuration(d.shiftedBy(-60)));
90          Assert.assertFalse(utc.insideLeap(d.shiftedBy(-60)));
91          Assert.assertEquals("1983-06-30T23:59:59.000", d.toString(utc));
92          Assert.assertEquals(61, utc.minuteDuration(d));
93          Assert.assertFalse(utc.insideLeap(d));
94          d = d.shiftedBy(0.251);
95          Assert.assertEquals("1983-06-30T23:59:59.251", d.toString(utc));
96          Assert.assertEquals(61, utc.minuteDuration(d));
97          Assert.assertFalse(utc.insideLeap(d));
98          d = d.shiftedBy(0.251);
99          Assert.assertEquals("1983-06-30T23:59:59.502", d.toString(utc));
100         Assert.assertEquals(61, utc.minuteDuration(d));
101         Assert.assertFalse(utc.insideLeap(d));
102         d = d.shiftedBy(0.251);
103         Assert.assertEquals("1983-06-30T23:59:59.753", d.toString(utc));
104         Assert.assertEquals(61, utc.minuteDuration(d));
105         Assert.assertFalse(utc.insideLeap(d));
106         d = d.shiftedBy( 0.251);
107         Assert.assertEquals("1983-06-30T23:59:60.004", d.toString(utc));
108         Assert.assertEquals(61, utc.minuteDuration(d));
109         Assert.assertTrue(utc.insideLeap(d));
110         d = d.shiftedBy(0.251);
111         Assert.assertEquals("1983-06-30T23:59:60.255", d.toString(utc));
112         Assert.assertEquals(61, utc.minuteDuration(d));
113         Assert.assertTrue(utc.insideLeap(d));
114         d = d.shiftedBy(0.251);
115         Assert.assertEquals("1983-06-30T23:59:60.506", d.toString(utc));
116         Assert.assertEquals(61, utc.minuteDuration(d));
117         d = d.shiftedBy(0.251);
118         Assert.assertEquals("1983-06-30T23:59:60.757", d.toString(utc));
119         Assert.assertEquals(61, utc.minuteDuration(d));
120         Assert.assertTrue(utc.insideLeap(d));
121         d = d.shiftedBy(0.251);
122         Assert.assertEquals("1983-07-01T00:00:00.008", d.toString(utc));
123         Assert.assertEquals(60, utc.minuteDuration(d));
124         Assert.assertFalse(utc.insideLeap(d));
125     }

133     @Test
134     public void testMinuteDuration() {
135         final AbsoluteDate t0 = new AbsoluteDate("1983-06-30T23:58:59.000", utc);
136         for (double dt = 0; dt < 63; dt += 0.3) {
137             if (dt < 1.0) {
138                 // before the minute of the leap
139                 Assert.assertEquals(60, utc.minuteDuration(t0.shiftedBy(dt)));
140             } else if (dt < 62.0) {
141                 // during the minute of the leap
142                 Assert.assertEquals(61, utc.minuteDuration(t0.shiftedBy(dt)));
143             } else {
144                 // after the minute of the leap
145                 Assert.assertEquals(60, utc.minuteDuration(t0.shiftedBy(dt)));                
146             }
147         }
148     }

163     public void testOffsets() {
164 
165         // we arbitrary put UTC == TAI before 1961-01-01
166         checkOffset(1950,  1,  1,   0);
167 
168         // excerpt from UTC-TAI.history file:
169         //  1961  Jan.  1 - 1961  Aug.  1     1.422 818 0s + (MJD - 37 300) x 0.001 296s
170         //        Aug.  1 - 1962  Jan.  1     1.372 818 0s +        ""
171         //  1962  Jan.  1 - 1963  Nov.  1     1.845 858 0s + (MJD - 37 665) x 0.001 123 2s
172         //  1963  Nov.  1 - 1964  Jan.  1     1.945 858 0s +        ""
173         //  1964  Jan.  1 -       April 1     3.240 130 0s + (MJD - 38 761) x 0.001 296s
174         //        April 1 -       Sept. 1     3.340 130 0s +        ""
175         //        Sept. 1 - 1965  Jan.  1     3.440 130 0s +        ""
176         //  1965  Jan.  1 -       March 1     3.540 130 0s +        ""
177         //        March 1 -       Jul.  1     3.640 130 0s +        ""
178         //        Jul.  1 -       Sept. 1     3.740 130 0s +        ""
179         //        Sept. 1 - 1966  Jan.  1     3.840 130 0s +        ""
180         //  1966  Jan.  1 - 1968  Feb.  1     4.313 170 0s + (MJD - 39 126) x 0.002 592s
181         //  1968  Feb.  1 - 1972  Jan.  1     4.213 170 0s +        ""
182         checkOffset(1961,  1,  2,  -(1.422818 +   1 * 0.001296));  // MJD 37300 +   1
183         checkOffset(1961,  8,  2,  -(1.372818 + 213 * 0.001296));  // MJD 37300 + 213
184         checkOffset(1962,  1,  2,  -(1.845858 +   1 * 0.0011232)); // MJD 37665 +   1
185         checkOffset(1963, 11,  2,  -(1.945858 + 670 * 0.0011232)); // MJD 37665 + 670
186         checkOffset(1964,  1,  2,  -(3.240130 - 365 * 0.001296));  // MJD 38761 - 365
187         checkOffset(1964,  4,  2,  -(3.340130 - 274 * 0.001296));  // MJD 38761 - 274
188         checkOffset(1964,  9,  2,  -(3.440130 - 121 * 0.001296));  // MJD 38761 - 121
189         checkOffset(1965,  1,  2,  -(3.540130 +   1 * 0.001296));  // MJD 38761 +   1
190         checkOffset(1965,  3,  2,  -(3.640130 +  60 * 0.001296));  // MJD 38761 +  60
191         checkOffset(1965,  7,  2,  -(3.740130 + 182 * 0.001296));  // MJD 38761 + 182
192         checkOffset(1965,  9,  2,  -(3.840130 + 244 * 0.001296));  // MJD 38761 + 244
193         checkOffset(1966,  1,  2,  -(4.313170 +   1 * 0.002592));  // MJD 39126 +   1
194         checkOffset(1968,  2,  2,  -(4.213170 + 762 * 0.002592));  // MJD 39126 + 762
195 
196         // since 1972-01-01, offsets are only whole seconds
197         checkOffset(1972,  3,  5, -10);
198         checkOffset(1972,  7, 14, -11);
199         checkOffset(1979, 12, 31, -18);
200         checkOffset(1980,  1, 22, -19);
201         checkOffset(2006,  7,  7, -33);

210     @Test
211     public void testCreatingInLeapDateUTC() {
212         AbsoluteDate previous = null;
213         final double step = 0.0625;
214         for (double seconds = 59.0; seconds < 61.0; seconds += step) {
215             final AbsoluteDate date = new AbsoluteDate(2008, 12, 31, 23, 59, seconds, utc);
216             if (previous != null) {
217                 Assert.assertEquals(step, date.durationFrom(previous), 1.0e-12);
218             }
219             previous = date;
220         }
221         AbsoluteDate ad0 = new AbsoluteDate("2008-12-31T23:59:60", utc);
222         Assert.assertTrue(ad0.toString(utc).startsWith("2008-12-31T23:59:"));
223         AbsoluteDate ad1 = new AbsoluteDate("2008-12-31T23:59:59", utc).shiftedBy(1);
224         Assert.assertEquals(0, ad1.durationFrom(ad0), 1.0e-15);
225         Assert.assertEquals(1, new AbsoluteDate("2009-01-01T00:00:00", utc).durationFrom(ad0), 1.0e-15);
226         Assert.assertEquals(2, new AbsoluteDate("2009-01-01T00:00:01", utc).durationFrom(ad0), 1.0e-15);
227     }

287     @Test
288     public void testMultithreading() {
289 
290         // generate reference offsets using a single thread
291         RandomGenerator random = new Well1024a(6392073424l);
292         List<AbsoluteDate> datesList = new ArrayList<AbsoluteDate>();
293         List<Double> offsetsList = new ArrayList<Double>();
294         AbsoluteDate reference = utc.getFirstKnownLeapSecond().shiftedBy(-Constants.JULIAN_YEAR);
295         double testRange = utc.getLastKnownLeapSecond().durationFrom(reference) + Constants.JULIAN_YEAR;
296         for (int i = 0; i < 10000; ++i) {
297             AbsoluteDate randomDate = reference.shiftedBy(random.nextDouble() * testRange);
298             datesList.add(randomDate);
299             offsetsList.add(utc.offsetFromTAI(randomDate));
300         }
301         
302         // check the offsets in multi-threaded mode
303         ExecutorService executorService = Executors.newFixedThreadPool(100);
304 
305         for (int i = 0; i < datesList.size(); ++i) {
306             final AbsoluteDate date = datesList.get(i);
307             final double offset = offsetsList.get(i);
308             executorService.execute(new Runnable() {
309                 public void run() {
310                     Assert.assertEquals(offset, utc.offsetFromTAI(date), 1.0e-12);
311                 }
312             });
313         }
314 
315         try {
316             executorService.shutdown();
317             executorService.awaitTermination(3, TimeUnit.SECONDS);
318         } catch (InterruptedException ie) {
319             Assert.fail(ie.getLocalizedMessage());
320         }
321 
322     }
