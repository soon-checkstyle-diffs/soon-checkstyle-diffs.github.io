34      public void testZeroRate() throws OrekitException {
35          FieldAngularCoordinates<DerivativeStructure> angularCoordinates =
36                  new FieldAngularCoordinates<DerivativeStructure>(createRotation(0.48, 0.64, 0.36, 0.48, false),

38          Assert.assertEquals(createVector(0, 0, 0, 4), angularCoordinates.getRotationRate());
39          double dt = 10.0;
40          FieldAngularCoordinates<DerivativeStructure> shifted = angularCoordinates.shiftedBy(dt);
41          Assert.assertEquals(0.0, shifted.getRotationAcceleration().getNorm().getReal(), 1.0e-15);
42          Assert.assertEquals(0.0, shifted.getRotationRate().getNorm().getReal(), 1.0e-15);
43          Assert.assertEquals(0.0, FieldRotation.distance(angularCoordinates.getRotation(), shifted.getRotation()).getReal(), 1.0e-15);
44      }

46      @Test
47      public void testShift() throws OrekitException {
48          double rate = 2 * FastMath.PI / (12 * 60);
49          FieldAngularCoordinates<DerivativeStructure> angularCoordinates =
50                  new FieldAngularCoordinates<DerivativeStructure>(createRotation(1, 0, 0, 0, false),
51                                         new FieldVector3D<DerivativeStructure>(rate, createVector(0, 0, 1, 4)), createVector(0, 0, 0, 4));
52          Assert.assertEquals(rate, angularCoordinates.getRotationRate().getNorm().getReal(), 1.0e-10);
53          double dt = 10.0;
54          double alpha = rate * dt;
55          FieldAngularCoordinates<DerivativeStructure> shifted = angularCoordinates.shiftedBy(dt);
56          Assert.assertEquals(rate, shifted.getRotationRate().getNorm().getReal(), 1.0e-10);
57          Assert.assertEquals(alpha, FieldRotation.distance(angularCoordinates.getRotation(), shifted.getRotation()).getReal(), 1.0e-10);
58  
59          FieldVector3D<DerivativeStructure> xSat = shifted.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
60          Assert.assertEquals(0.0, xSat.subtract(createVector(FastMath.cos(alpha), FastMath.sin(alpha), 0, 4)).getNorm().getReal(), 1.0e-10);
61          FieldVector3D<DerivativeStructure> ySat = shifted.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
62          Assert.assertEquals(0.0, ySat.subtract(createVector(-FastMath.sin(alpha), FastMath.cos(alpha), 0, 4)).getNorm().getReal(), 1.0e-10);
63          FieldVector3D<DerivativeStructure> zSat = shifted.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
64          Assert.assertEquals(0.0, zSat.subtract(createVector(0, 0, 1, 4)).getNorm().getReal(), 1.0e-10);
65  
66      }

82      @Test
83      public void testSpin() throws OrekitException {
84          double rate = 2 * FastMath.PI / (12 * 60);
85          FieldAngularCoordinates<DerivativeStructure> angularCoordinates =
86                  new FieldAngularCoordinates<DerivativeStructure>(createRotation(0.48, 0.64, 0.36, 0.48, false),
87                                         new FieldVector3D<DerivativeStructure>(rate, createVector(0, 0, 1, 4)),
88                                         createVector(0, 0, 0, 4));
89          Assert.assertEquals(rate, angularCoordinates.getRotationRate().getNorm().getReal(), 1.0e-10);
90          double dt = 10.0;
91          FieldAngularCoordinates<DerivativeStructure> shifted = angularCoordinates.shiftedBy(dt);
92          Assert.assertEquals(rate, shifted.getRotationRate().getNorm().getReal(), 1.0e-10);
93          Assert.assertEquals(rate * dt, FieldRotation.distance(angularCoordinates.getRotation(), shifted.getRotation()).getReal(), 1.0e-10);
94  
95          FieldVector3D<DerivativeStructure> shiftedX  = shifted.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
96          FieldVector3D<DerivativeStructure> shiftedY  = shifted.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
97          FieldVector3D<DerivativeStructure> shiftedZ  = shifted.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
98          FieldVector3D<DerivativeStructure> originalX = angularCoordinates.getRotation().applyInverseTo(createVector(1, 0, 0, 4));
99          FieldVector3D<DerivativeStructure> originalY = angularCoordinates.getRotation().applyInverseTo(createVector(0, 1, 0, 4));
100         FieldVector3D<DerivativeStructure> originalZ = angularCoordinates.getRotation().applyInverseTo(createVector(0, 0, 1, 4));
101         Assert.assertEquals( FastMath.cos(rate * dt), FieldVector3D.dotProduct(shiftedX, originalX).getReal(), 1.0e-10);
102         Assert.assertEquals( FastMath.sin(rate * dt), FieldVector3D.dotProduct(shiftedX, originalY).getReal(), 1.0e-10);
103         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedX, originalZ).getReal(), 1.0e-10);
104         Assert.assertEquals(-FastMath.sin(rate * dt), FieldVector3D.dotProduct(shiftedY, originalX).getReal(), 1.0e-10);
105         Assert.assertEquals( FastMath.cos(rate * dt), FieldVector3D.dotProduct(shiftedY, originalY).getReal(), 1.0e-10);
106         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedY, originalZ).getReal(), 1.0e-10);
107         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedZ, originalX).getReal(), 1.0e-10);
108         Assert.assertEquals( 0.0,                 FieldVector3D.dotProduct(shiftedZ, originalY).getReal(), 1.0e-10);
109         Assert.assertEquals( 1.0,                 FieldVector3D.dotProduct(shiftedZ, originalZ).getReal(), 1.0e-10);
110 
111         FieldVector3D<DerivativeStructure> forward = FieldAngularCoordinates.estimateRate(angularCoordinates.getRotation(), shifted.getRotation(), dt);
112         Assert.assertEquals(0.0, forward.subtract(angularCoordinates.getRotationRate()).getNorm().getReal(), 1.0e-10);
113 
114         FieldVector3D<DerivativeStructure> reversed = FieldAngularCoordinates.estimateRate(shifted.getRotation(), angularCoordinates.getRotation(), dt);
115         Assert.assertEquals(0.0, reversed.add(angularCoordinates.getRotationRate()).getNorm().getReal(), 1.0e-10);
116 
117     }

121         Random random = new Random(0x4ecca9d57a8f1611l);
122         for (int i = 0; i < 100; ++i) {
123             FieldRotation<DerivativeStructure> r = randomRotation(random);

125             FieldVector3D<DerivativeStructure> a = randomVector(random, 1.0e-3);
126             FieldAngularCoordinates<DerivativeStructure> ac = new FieldAngularCoordinates<DerivativeStructure>(r, o, a);

129             Assert.assertEquals(0.0, sum.getRotationRate().getNorm().getReal(), 1.0e-15);

150     @Test
151     public void testRoundTripNoOp() {
152         Random random = new Random(0x1e610cfe89306669l);
153         for (int i = 0; i < 100; ++i) {
154 
155             FieldRotation<DerivativeStructure> r1 = randomRotation(random);
156             FieldVector3D<DerivativeStructure> o1 = randomVector(random, 1.0e-2);
157             FieldVector3D<DerivativeStructure> a1 = randomVector(random, 1.0e-2);
158             FieldAngularCoordinates<DerivativeStructure> ac1 = new FieldAngularCoordinates<DerivativeStructure>(r1, o1, a1);
159 
160             FieldRotation<DerivativeStructure> r2 = randomRotation(random);
161             FieldVector3D<DerivativeStructure> o2 = randomVector(random, 1.0e-2);
162             FieldVector3D<DerivativeStructure> a2 = randomVector(random, 1.0e-2);
163             FieldAngularCoordinates<DerivativeStructure> ac2 = new FieldAngularCoordinates<DerivativeStructure>(r2, o2, a2);
164 
165             FieldAngularCoordinates<DerivativeStructure> roundTripSA = ac1.subtractOffset(ac2).addOffset(ac2);
166             Assert.assertEquals(0.0, FieldRotation.distance(ac1.getRotation(), roundTripSA.getRotation()).getReal(), 1.0e-15);
167             Assert.assertEquals(0.0, FieldVector3D.distance(ac1.getRotationRate(), roundTripSA.getRotationRate()).getReal(), 2.0e-17);
168             Assert.assertEquals(0.0, FieldVector3D.distance(ac1.getRotationAcceleration(), roundTripSA.getRotationAcceleration()).getReal(), 2.0e-17);
169 
170             FieldAngularCoordinates<DerivativeStructure> roundTripAS = ac1.addOffset(ac2).subtractOffset(ac2);
171             Assert.assertEquals(0.0, FieldRotation.distance(ac1.getRotation(), roundTripAS.getRotation()).getReal(), 1.0e-15);
172             Assert.assertEquals(0.0, FieldVector3D.distance(ac1.getRotationRate(), roundTripAS.getRotationRate()).getReal(), 2.0e-17);
173             Assert.assertEquals(0.0, FieldVector3D.distance(ac1.getRotationAcceleration(), roundTripAS.getRotationAcceleration()).getReal(), 2.0e-17);
174 
175         }
176     }

178     private FieldVector3D<DerivativeStructure> randomVector(Random random, double norm) {
179         double n = random.nextDouble() * norm;
180         double x = random.nextDouble();
181         double y = random.nextDouble();
182         double z = random.nextDouble();
183         return new FieldVector3D<DerivativeStructure>(n, createVector(x, y, z, 4).normalize());
184     }
185 
186     private FieldRotation<DerivativeStructure> randomRotation(Random random) {
187         double q0 = random.nextDouble() * 2 - 1;
188         double q1 = random.nextDouble() * 2 - 1;
189         double q2 = random.nextDouble() * 2 - 1;
190         double q3 = random.nextDouble() * 2 - 1;
191         double q  = FastMath.sqrt(q0 * q0 + q1 * q1 + q2 * q2 + q3 * q3);
192         return createRotation(q0 / q, q1 / q, q2 / q, q3 / q, false);
193     }
194 
195     private FieldRotation<DerivativeStructure> createRotation(double q0, double q1, double q2, double q3,
196                                       boolean needsNormalization) {
197         return new FieldRotation<DerivativeStructure>(new DerivativeStructure(4, 1, 0, q0),
198                               new DerivativeStructure(4, 1, 1, q1),
199                               new DerivativeStructure(4, 1, 2, q2),
200                               new DerivativeStructure(4, 1, 3, q3),
201                               needsNormalization);
202     }
203 
204     private FieldVector3D<DerivativeStructure> createVector(double x, double y, double z, int params) {
205         return new FieldVector3D<DerivativeStructure>(new DerivativeStructure(params, 1, 0, x),
206                               new DerivativeStructure(params, 1, 1, y),
207                               new DerivativeStructure(params, 1, 2, z));
208     }
