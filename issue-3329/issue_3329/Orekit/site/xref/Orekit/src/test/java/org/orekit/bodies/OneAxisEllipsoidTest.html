240     private double[] derivativesErrors(PVCoordinatesProvider provider, AbsoluteDate date, Frame frame,
241                                        OneAxisEllipsoid model)
242         throws OrekitException {
243         List<TimeStampedPVCoordinates> pvList       = new ArrayList<TimeStampedPVCoordinates>();
244         List<TimeStampedPVCoordinates> groundPVList = new ArrayList<TimeStampedPVCoordinates>();
245         for (double dt = -0.25; dt <= 0.25; dt += 0.125) {
246             TimeStampedPVCoordinates shiftedPV = provider.getPVCoordinates(date.shiftedBy(dt), frame);
247             Vector3D p = model.projectToGround(shiftedPV.getPosition(), shiftedPV.getDate(), frame);
248             pvList.add(shiftedPV);
249             groundPVList.add(new TimeStampedPVCoordinates(shiftedPV.getDate(),
250                                                           p, Vector3D.ZERO, Vector3D.ZERO));
251         }
252         TimeStampedPVCoordinates computed =
253                 model.projectToGround(TimeStampedPVCoordinates.interpolate(date,
254                                                                            CartesianDerivativesFilter.USE_P,
255                                                                            pvList),
256                                                                            frame);
257         TimeStampedPVCoordinates reference =
258                 TimeStampedPVCoordinates.interpolate(date,
259                                                      CartesianDerivativesFilter.USE_P,
260                                                      groundPVList);
261 
262         TimeStampedPVCoordinates pv0 = provider.getPVCoordinates(date, frame);
263         Vector3D p0 = pv0.getPosition();
264         Vector3D v0 = pv0.getVelocity();
265         Vector3D a0 = pv0.getAcceleration();
266 
267         return new double[] {
268             Vector3D.distance(computed.getPosition(),     reference.getPosition())     / p0.getNorm(),
269             Vector3D.distance(computed.getVelocity(),     reference.getVelocity())     / v0.getNorm(),
270             Vector3D.distance(computed.getAcceleration(), reference.getAcceleration()) / a0.getNorm(),
271         };
272 
273     }

275     @Test
276     public void testGroundProjectionTaylor()
277             throws OrekitException {
278         Frame itrf = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
279         Frame eme2000 = FramesFactory.getEME2000();
280         OneAxisEllipsoid model =
281             new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
282                                  Constants.WGS84_EARTH_FLATTENING,
283                                  itrf);
284 
285         TimeStampedPVCoordinates initPV =
286                 new TimeStampedPVCoordinates(AbsoluteDate.J2000_EPOCH.shiftedBy(584.),
287                                              new Vector3D(3220103., 69623., 6449822.),
288                                              new Vector3D(6414.7, -2006., -3180.),
289                                              Vector3D.ZERO);
290         Orbit orbit = new EquinoctialOrbit(initPV, eme2000, Constants.EIGEN5C_EARTH_MU);
291 
292         TimeStampedPVCoordinates pv0 = orbit.getPVCoordinates(orbit.getDate(), model.getBodyFrame());
293         PVCoordinatesProvider groundTaylor =
294                 model.projectToGround(pv0, model.getBodyFrame()).toTaylorProvider(model.getBodyFrame());
295 
296         TimeStampedPVCoordinates g0 = groundTaylor.getPVCoordinates(orbit.getDate(), model.getBodyFrame());
297         Vector3D zenith       = pv0.getPosition().subtract(g0.getPosition()).normalize();
298         Vector3D acrossTrack  = Vector3D.crossProduct(zenith, g0.getVelocity()).normalize();
299         Vector3D alongTrack   = Vector3D.crossProduct(acrossTrack, zenith).normalize();
300         for (double dt = -1; dt < 1; dt += 0.01) {
301             AbsoluteDate date = orbit.getDate().shiftedBy(dt);
302             Vector3D taylorP = groundTaylor.getPVCoordinates(date, model.getBodyFrame()).getPosition();
303             Vector3D refP    = model.projectToGround(orbit.getPVCoordinates(date, model.getBodyFrame()).getPosition(),
304                                                      date, model.getBodyFrame());
305             Vector3D delta = taylorP.subtract(refP);
306             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, alongTrack),  0.0015);
307             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, acrossTrack), 0.0007);
308             Assert.assertEquals(0.0, Vector3D.dotProduct(delta, zenith),      0.00002);
309         }
310

606     @Test
607     public void testMovingGeodeticPoint() throws OrekitException {
608 
609         final OneAxisEllipsoid earth = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
610                                                             Constants.WGS84_EARTH_FLATTENING,
611                                                             FramesFactory.getITRF(IERSConventions.IERS_2010, true));
612         double lat0 = FastMath.toRadians(60.0);
613         double lon0 = FastMath.toRadians(25.0);
614         double alt0 = 100.0;
615         double lat1 =   1.0e-3;
616         double lon1 =  -2.0e-3;
617         double alt1 =   1.2;
618         double lat2 =  -1.0e-5;
619         double lon2 =  -3.0e-5;
620         double alt2 =  -0.01;
621         final DerivativeStructure latDS = new DerivativeStructure(1, 2, lat0, lat1, lat2);
622         final DerivativeStructure lonDS = new DerivativeStructure(1, 2, lon0, lon1, lon2);
623         final DerivativeStructure altDS = new DerivativeStructure(1, 2, alt0, alt1, alt2);
624 
625         // direct computation of position, velocity and acceleration
626         PVCoordinates pv = earth.transform(new FieldGeodeticPoint<DerivativeStructure>(latDS, lonDS, altDS));
627 
628         // finite differences computation
629         FiniteDifferencesDifferentiator differentiator =
630                 new FiniteDifferencesDifferentiator(5, 0.1);
631         UnivariateDifferentiableFunction fx =
632                 differentiator.differentiate(new UnivariateFunction() {
633                     public double value(double dt) {
634                         GeodeticPoint gp =
635                                 new GeodeticPoint(latDS.taylor(dt), lonDS.taylor(dt), altDS.taylor(dt));
636                         return earth.transform(gp).getX();
637                     }
638                 });
639         UnivariateDifferentiableFunction fy =
640                 differentiator.differentiate(new UnivariateFunction() {
641                     public double value(double dt) {
642                         GeodeticPoint gp =
643                                 new GeodeticPoint(latDS.taylor(dt), lonDS.taylor(dt), altDS.taylor(dt));
644                         return earth.transform(gp).getY();
645                     }
646                 });
647         UnivariateDifferentiableFunction fz =
648                 differentiator.differentiate(new UnivariateFunction() {
649                     public double value(double dt) {
650                         GeodeticPoint gp =
651                                 new GeodeticPoint(latDS.taylor(dt), lonDS.taylor(dt), altDS.taylor(dt));
652                         return earth.transform(gp).getZ();
653                     }
654                 });
655         DerivativeStructure dtZero = new DerivativeStructure(1, 2, 0, 0.0);
656         DerivativeStructure xDS    = fx.value(dtZero);
657         DerivativeStructure yDS    = fy.value(dtZero);
658         DerivativeStructure zDS    = fz.value(dtZero);
659         Assert.assertEquals(xDS.getValue(),              pv.getPosition().getX(),
660                             2.0e-20 * FastMath.abs(xDS.getValue()));
661         Assert.assertEquals(xDS.getPartialDerivative(1), pv.getVelocity().getX(),
662                             2.0e-12 * FastMath.abs(xDS.getPartialDerivative(1)));
663         Assert.assertEquals(xDS.getPartialDerivative(2), pv.getAcceleration().getX(),
664                             2.0e-9  * FastMath.abs(xDS.getPartialDerivative(2)));
665         Assert.assertEquals(yDS.getValue(),              pv.getPosition().getY(),
666                             2.0e-20 * FastMath.abs(yDS.getValue()));
667         Assert.assertEquals(yDS.getPartialDerivative(1), pv.getVelocity().getY(),
668                             2.0e-12 * FastMath.abs(yDS.getPartialDerivative(1)));
669         Assert.assertEquals(yDS.getPartialDerivative(2), pv.getAcceleration().getY(),
670                             2.0e-9  * FastMath.abs(yDS.getPartialDerivative(2)));
671         Assert.assertEquals(zDS.getValue(),              pv.getPosition().getZ(),
672                             2.0e-20 * FastMath.abs(zDS.getValue()));
673         Assert.assertEquals(zDS.getPartialDerivative(1), pv.getVelocity().getZ(),
674                             2.0e-12 * FastMath.abs(zDS.getPartialDerivative(1)));
675         Assert.assertEquals(zDS.getPartialDerivative(2), pv.getAcceleration().getZ(),
676                             2.0e-9  * FastMath.abs(zDS.getPartialDerivative(2)));
677     }

680                                              double x, double y, double z,
681                                              double longitude, double latitude,
682                                              double altitude)
683         throws OrekitException {
684 
685         AbsoluteDate date = AbsoluteDate.J2000_EPOCH;

693         Assert.assertEquals(0, rebuiltNadir.subtract(gp.getNadir()).getNorm(), 1.0e-15);
