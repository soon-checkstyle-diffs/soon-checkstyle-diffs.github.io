144 
145         TileAiming aiming = new ConstantAzimuthAiming(ellipsoid, FastMath.toRadians(193.7));
146         EllipsoidTessellator tessellator =
147                 new EllipsoidTessellator(ellipsoid, aiming, 16);

149         SphericalPolygonsSet small = buildSimpleZone(1.0e-10, new double[][] {

168                                              ellipsoid.transform(t.getVertices()[2])),
169                             0.4);
170         Assert.assertEquals(150000.0,
171                             Vector3D.distance(ellipsoid.transform(t.getVertices()[2]),
172                                              ellipsoid.transform(t.getVertices()[3])),

183 
184         TileAiming aiming = new ConstantAzimuthAiming(ellipsoid, FastMath.toRadians(193.7));
185         EllipsoidTessellator tessellator =
186                 new EllipsoidTessellator(ellipsoid, aiming, 16);
187 
188         SphericalPolygonsSet small = buildSimpleZone(1.0e-10, new double[][] {
189             { 43.6543, 1.4268 }, { 43.6120, 1.4179 }, { 43.6016, 1.3994 }, { 43.5682, 1.4159 },
190             { 43.5707, 1.4358 }, { 43.5573, 1.4941 }, { 43.6041, 1.4866 }
191         });
192 
193         final List<List<Tile>> tiles = tessellator.tessellate(small, 50000.0, 150000.0, 0, 0,

195         Assert.assertEquals(1, tiles.size());

205                             Vector3D.distance(ellipsoid.transform(t.getVertices()[1]),
206                                              ellipsoid.transform(t.getVertices()[2])),
207                             0.01);
208         Assert.assertEquals(3.0 / 16.0 * 150000.0,
209                            Vector3D.distance(ellipsoid.transform(t.getVertices()[2]),
210                                              ellipsoid.transform(t.getVertices()[3])),
211                            10.0);
212         Assert.assertEquals(4.0 / 16.0 * 50000.0,
213                             Vector3D.distance(ellipsoid.transform(t.getVertices()[3]),
214                                              ellipsoid.transform(t.getVertices()[0])),

218     @Test
219     public void testStairedTruncatedTiles() throws OrekitException {
220 
221         TileAiming aiming = new ConstantAzimuthAiming(ellipsoid, FastMath.toRadians(170.0));
222         EllipsoidTessellator tessellator =
223                 new EllipsoidTessellator(ellipsoid, aiming, 16);
224 
225         SphericalPolygonsSet small = buildSimpleZone(1.0e-10, new double[][] {
226             { 45.335,  0.457 },
227             { 45.342,  0.469 },
228             { 45.371,  0.424 }
229         });
230 
231         final double maxWidth  = 800.0;
232         final double maxLength = 10000.0;
233         final List<List<Tile>> tiles = tessellator.tessellate(small, maxWidth, maxLength, 0, 0,
234                                                               false, true);
235         Assert.assertEquals(1, tiles.size());
236         Assert.assertEquals(4, tiles.get(0).size());
237         for (final Tile tile : tiles.get(0)) {
238             Vector3D v0 = ellipsoid.transform(tile.getVertices()[0]);
239             Vector3D v1 = ellipsoid.transform(tile.getVertices()[1]);
240             Vector3D v2 = ellipsoid.transform(tile.getVertices()[2]);
241             Vector3D v3 = ellipsoid.transform(tile.getVertices()[3]);
242             Assert.assertTrue(Vector3D.distance(v0, v1) < (6.0002 / 16.0) * maxLength);
243             Assert.assertTrue(Vector3D.distance(v2, v3) < (6.0002 / 16.0) * maxLength);
244             Assert.assertEquals(maxWidth, Vector3D.distance(v1, v2), 1.0e-3);
245             Assert.assertEquals(maxWidth, Vector3D.distance(v3, v0), 1.0e-3);
246         }
247 
248     }

326     private void checkTilesDontOverlap(final List<List<Tile>> tiles) {
327         for (final List<Tile> list : tiles) {
328             for (final Tile tile : list) {
329                 final SphericalPolygonsSet quadrilateral =
330                         new SphericalPolygonsSet(1.0e-10,
331                                                  toS2Point(tile.getVertices()[0]),
332                                                  toS2Point(tile.getVertices()[1]),
333                                                  toS2Point(tile.getVertices()[2]),
334                                                  toS2Point(tile.getVertices()[3]));
335                 for (final List<Tile> otherList : tiles) {
336                     for (final Tile otherTile : otherList) {
337                         if (otherTile != tile) {
338                             for (final GeodeticPoint vertex : otherTile.getVertices()) {
339                                 Assert.assertEquals("tiles overlap at: " + vertex,
340                                                     Location.OUTSIDE, quadrilateral.checkPoint(toS2Point(vertex)));
341                             }
342                         }
343                     }
344                 }
345             }
346         }
347     }

349     private S2Point toS2Point(final GeodeticPoint point) {
350         return new S2Point(point.getLongitude(), 0.5 * FastMath.PI - point.getLatitude());
351     }

419     private SphericalPolygonsSet buildSimpleZone(double tolerance, double[][] points) {
420         for (int i = 0; i < points.length; ++i) {
421             points[i][0] = FastMath.toRadians(points[i][0]);
422             points[i][1] = FastMath.toRadians(points[i][1]);
423         }
424         return EllipsoidTessellator.buildSimpleZone(tolerance, points);
425     }
