109     public void setSessionContext(SessionContext arg0) throws EJBException,
110             RemoteException {
111         Context jndiCtx = null;
112         try {
113             jndiCtx = new InitialContext();
114             patHome = (PatientLocalHome) jndiCtx
115                     .lookup("java:comp/env/ejb/Patient");
116             studyHome = (StudyLocalHome) jndiCtx
117                     .lookup("java:comp/env/ejb/Study");
118             seriesHome = (SeriesLocalHome) jndiCtx
119                     .lookup("java:comp/env/ejb/Series");
120             instHome = (InstanceLocalHome) jndiCtx
121                     .lookup("java:comp/env/ejb/Instance");
122 
123         } catch (NamingException e) {
124             throw new EJBException(e);
125         } finally {
126             if (jndiCtx != null) {
127                 try {
128                     jndiCtx.close();
129                 } catch (NamingException ignore) {
130                 }
131             }
132         }
133     }
134 
135     public void unsetSessionContext() {
136         patHome = null;
137         studyHome = null;
138         seriesHome = null;
139         instHome = null;
140     }

142     /**
143      * @param matching 
144      * @ejb.interface-method
145      */
146     public void createStudy(Dataset ds, PatientMatching matching)
147             throws DcmServiceException, CreateException, FinderException {
148         checkDuplicateStudy(ds.getString(Tags.StudyInstanceUID));
149         studyHome.create(ds, findOrCreatePatient(ds, matching));
150     }
151 
152     private PatientLocal findOrCreatePatient(Dataset ds,
153             PatientMatching matching) throws FinderException, CreateException {
154         try {
155             return patHome.selectPatient(ds, matching, true);
156         } catch (ObjectNotFoundException onfe) {
157             return patHome.create(ds);
158         }
159     }
160 
161     private void checkDuplicateStudy(String suid) throws FinderException,
162             DcmServiceException {
163         try {
164             studyHome.findByStudyIuid(suid);
165             throw new DcmServiceException(Status.DuplicateSOPInstance, suid);
166         } catch (ObjectNotFoundException e) {
167         }
168     }
169 
170     private StudyLocal getStudy(String suid) throws FinderException,
171             DcmServiceException {
172         try {
173             return studyHome.findByStudyIuid(suid);
174         } catch (ObjectNotFoundException e) {
175             throw new DcmServiceException(Status.NoSuchObjectInstance, suid);
176         }
177     }

179     /**
180      * @ejb.interface-method
181      */
182     public void updateStudy(String iuid, Dataset ds, PatientMatching matching) throws DcmServiceException {
183         try {
184             StudyLocal study = getStudy(iuid);
185             if (study == null) {
186                 // Study may be deleted already
187                 log
188                         .warn("Unable to update the study that does not exist. StudyIuid: "
189                                 + iuid);
190                 return;
191             }
192 
193             Dataset attrs = study.getAttributes(false);
194             attrs.putAll(ds);
195             study.setAttributes(attrs);
196             DcmElement seriesSq = ds.get(Tags.RefSeriesSeq);
197             if (seriesSq != null) {
198                 Set dirtyStudies = new HashSet();
199                 Set dirtySeries = new HashSet();
200                 for (int i = 0, n = seriesSq.countItems(); i < n; ++i) {
201                     updateSeries(seriesSq.getItem(i), study, dirtyStudies,
202                             dirtySeries);
203                 }
204                 updateDerivedSeriesFields(dirtySeries);
205                 updateDerivedStudyFields(dirtyStudies);
206             }
207             if ( ds.contains(Tags.PatientID) || ds.contains(Tags.PatientName)) {
208                 PatientLocal pat = findOrCreatePatient(ds, matching);
209                 if ( pat.getPk() != study.getPatient().getPk()) {
210                     study.setPatient(pat);
211                 }
212             }
213         } catch (FinderException e) {
214             throw new EJBException(e);
215         } catch (CreateException e) {
216             throw new EJBException(e);
217         }
218     }

223     public void updatePatientOnly(String iuid, Dataset ds, Dataset modAttrs)
224             throws DcmServiceException {
225         try {
226             PatientLocal patient = getStudy(iuid).getPatient();
227             Dataset origModAttrs = (modAttrs == null)?null: DcmObjectFactory.getInstance().newDataset(); 	
228             if(patient.updateAttributes(ds, origModAttrs)) {
229                 AttrUtils.fetchModifiedAttributes(ds, origModAttrs, modAttrs, AttributeFilter.getPatientAttributeFilter());
230             }   
231         } catch (Exception e) {
232             throw new EJBException(e);
233         }
234     }

252     private void updateDerivedStudyFields(Set dirtyStudies)
253             throws FinderException {
254         for (Iterator it = dirtyStudies.iterator(); it.hasNext();) {
255             String iuid = (String) it.next();
256             StudyLocal study = studyHome.findByStudyIuid(iuid);
257             UpdateDerivedFieldsUtils.updateDerivedFieldsOf(study);
258         }
259     }
260 
261     private void updateDerivedSeriesFields(Set dirtySeries)
262             throws FinderException {
263         for (Iterator it = dirtySeries.iterator(); it.hasNext();) {
264             String iuid = (String) it.next();
265             SeriesLocal series = seriesHome.findBySeriesIuid(iuid);
266             UpdateDerivedFieldsUtils.updateDerivedFieldsOf(series);
267         }
268     }
269 
270     private void updateSeries(Dataset ds, StudyLocal study, Set dirtyStudies,
271             Set dirtySeries) throws FinderException, CreateException {
272         try {
273             SeriesLocal series = seriesHome.findBySeriesIuid(ds
274                     .getString(Tags.SeriesInstanceUID));
275             StudyLocal prevStudy = series.getStudy();
276             if (!study.isIdentical(prevStudy)) {
277                 log.info("Move " + series.asString() + " from "
278                         + prevStudy.asString() + " to " + study.asString());
279                 series.setStudy(study);
280                 dirtyStudies.add(study.getStudyIuid());
281                 dirtyStudies.add(prevStudy.getStudyIuid());
282             }
283             Dataset attrs = series.getAttributes(false);
284             String newModality = ds.getString(Tags.Modality);
285             if (newModality != null
286                     && !newModality.equals(attrs.getString(Tags.Modality))) {
287                 dirtyStudies.add(study.getStudyIuid());
288             }
289             attrs.putAll(ds);
290             series.setAttributes(attrs);
291             DcmElement sopSq = ds.get(Tags.RefSOPSeq);
292             if (sopSq != null) {
293                 for (int i = 0, n = sopSq.countItems(); i < n; ++i) {
294                     updateInstance(sopSq.getItem(i), series, dirtyStudies,
295                             dirtySeries);
296                 }
297             }
298         } catch (ObjectNotFoundException e) {
299             seriesHome.create(ds, study);
300             dirtyStudies.add(study.getStudyIuid());
301         }
302     }
303 
304     private void updateInstance(Dataset ds, SeriesLocal series,
305             Set dirtyStudies, Set dirtySeries) throws FinderException,
306             CreateException {
307         try {
308             InstanceLocal inst = instHome.findBySopIuid(ds
309                     .getString(Tags.RefSOPInstanceUID));
310             SeriesLocal prevSeries = inst.getSeries();
311             if (!series.isIdentical(prevSeries)) {
312                 log.info("Move " + inst.asString() + " from "
313                         + prevSeries.asString() + " to " + series.asString());
314                 inst.setSeries(series);
315                 dirtySeries.add(series.getSeriesIuid());
316                 dirtyStudies.add(series.getStudy().getStudyIuid());
317                 dirtySeries.add(prevSeries.getSeriesIuid());
318                 dirtyStudies.add(prevSeries.getStudy().getStudyIuid());
319             }
320             Dataset attrs = inst.getAttributes(false);
321             attrs.putAll(ds);
322             inst.setAttributes(attrs);
323         } catch (ObjectNotFoundException e) {
324             instHome.create(ds, series);
325             dirtySeries.add(series.getSeriesIuid());
326             dirtyStudies.add(series.getStudy().getStudyIuid());
327         }
328     }

332      */
333     public void updateStudyStatusId(String iuid, String statusId)
334             throws FinderException, DcmServiceException {
335         StudyLocal study = getStudy(iuid);
336         Dataset attrs = study.getAttributes(false);
337         if (!statusId.equals(attrs.getString(Tags.StudyStatusID))) {
338             attrs.putCS(Tags.StudyStatusID, statusId);
339             study.setAttributes(attrs);
340         }
341     }
