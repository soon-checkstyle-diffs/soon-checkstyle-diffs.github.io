20  package edu.umd.cs.findbugs.annotations;
21  
22  import java.lang.annotation.Annotation;
23  import java.lang.annotation.Documented;
24  import java.lang.annotation.ElementType;
25  import java.lang.annotation.Retention;
26  import java.lang.annotation.RetentionPolicy;
27  import java.lang.annotation.Target;
28  
29  import javax.annotation.Nonnegative;
30  import javax.annotation.meta.TypeQualifierDefault;
31  
32  
33  /**
34   * Indicates that all members of the class or package should be annotated with
35   * the default value of the supplied annotation class.
36   *
37   * This would be used for behavior annotations such as @NonNull, @CheckForNull,
38   * or @CheckReturnValue.
39   *
40   * In particular, you can use @DefaultAnnotation(NonNull.class) on a class or
41   * package, and then use @Nullable only on those parameters, methods or fields
42   * that you want to allow to be null.
43   *
44   * @deprecated -  Use the JSR305 annotations instead.
45   * For example, you can use {@link javax.annotation.ParametersAreNonnullByDefault} instead
46   * of @DefaultAnnotation(NonNull.class) so that method parameters are nonnull by default in the annotated
47   * element. You can also use {@link javax.annotation.meta.TypeQualifierDefault}
48   * in general to define your own annotation that specifies a default type qualifier. For example,
49   * <p><pre><code>
50   * {@link Nonnegative}
51   * {@link TypeQualifierDefault}({@link ElementType#PARAMETER})
52   * public @interface ParametersAreNonnegativeByDefault {}
53   * </code></pre>
54   *
55   * <p>The JSR305 {@link javax.annotation.CheckReturnValue}
56   * annotation can be applied to a type or package, and it will act as a default for all methods
57   * in that class or package unless otherwise overridden.
58   *
59   * @author William Pugh
60   */
61  
62  @Documented
63  @Target({ ElementType.TYPE, ElementType.PACKAGE })
64  @Retention(RetentionPolicy.CLASS)
65  @Deprecated
66  public @interface DefaultAnnotation {
67      Class<? extends Annotation>[] value();
68  
69      @Deprecated
70      Priority priority() default Priority.MEDIUM;
71  
72      Confidence confidence() default Confidence.MEDIUM;
73  }
