6   package edu.umd.cs.findbugs.annotations;
7   
8   import java.lang.annotation.Retention;
9   import java.lang.annotation.RetentionPolicy;
10  
11  /**
12   * Annotation indicating that <em>no</em> FindBugs warnings of the specified type
13   * is desired.
14   *
15   * See http://code.google.com/p/findbugs/wiki/FindbugsTestCases
16   *
17   * @author David Hovemeyer
18   */
19  @Retention(RetentionPolicy.CLASS)
20  public @interface DesireNoWarning {
21      /**
22       * The value indicates the bug code (e.g., NP) or bug pattern (e.g.,
23       * IL_INFINITE_LOOP) that is desired to not be reported
24       */
25      public String value();
26  
27      /** @deprecated - use confidence instead */
28      @Deprecated
29      Priority priority() default Priority.LOW;
30  
31      /** Want no warning at this priority or higher */
32      Confidence confidence() default Confidence.LOW;
33  
34      /** Tolerate up to this many warnings */
35      public int num() default 0;
36  }
