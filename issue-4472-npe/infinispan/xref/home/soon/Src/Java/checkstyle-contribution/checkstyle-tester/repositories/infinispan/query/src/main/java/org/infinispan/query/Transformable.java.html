1   package org.infinispan.query;
2   
3   import java.lang.annotation.Target;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Retention;
6   import java.lang.annotation.RetentionPolicy;
7   
8   import org.infinispan.query.impl.DefaultTransformer;
9   
10  /**
11   * If you annotate your object with this, it can be used as a valid key for Infinispan to index.
12   *
13   * @author Manik Surtani
14   * @since 4.0
15   */
16  @Target(ElementType.TYPE)
17  @Retention(RetentionPolicy.RUNTIME)
18  public @interface Transformable {
19  
20     // This should really be Class<? extends Transformer> however, since
21     // migrating to JBoss Logging and using annotation processor for generating
22     // log objects, if the previous definition is used, a compiler bug is hit.
23     // You can find more info in https://issues.jboss.org/browse/ISPN-380
24     Class transformer() default DefaultTransformer.class;
25  }
