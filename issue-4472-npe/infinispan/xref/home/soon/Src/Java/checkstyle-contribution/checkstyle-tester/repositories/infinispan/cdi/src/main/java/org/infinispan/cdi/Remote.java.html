1   package org.infinispan.cdi;
2   
3   import javax.enterprise.util.Nonbinding;
4   import javax.inject.Qualifier;
5   import java.lang.annotation.Documented;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.Target;
8   
9   import static java.lang.annotation.ElementType.FIELD;
10  import static java.lang.annotation.ElementType.METHOD;
11  import static java.lang.annotation.ElementType.PARAMETER;
12  import static java.lang.annotation.ElementType.TYPE;
13  import static java.lang.annotation.RetentionPolicy.RUNTIME;
14  
15  /**
16   * Qualifier used to specify which remote cache will be injected.
17   *
18   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
19   */
20  @Target({METHOD, FIELD, PARAMETER, TYPE})
21  @Retention(RUNTIME)
22  @Documented
23  @Qualifier
24  public @interface Remote {
25     /**
26      * The remote cache name. If no value is provided the default cache is assumed.
27      */
28     @Nonbinding String value() default "";
29  }
