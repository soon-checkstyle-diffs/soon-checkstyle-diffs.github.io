1   package org.infinispan.client.hotrod.annotation;
2   
3   import java.lang.annotation.ElementType;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   /**
9    * Annotation that marks a class to receive remote events from Hot Rod caches.
10   * Classes with this annotation are expected to have at least one callback
11   * annotated with one of the events it can receive:
12   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryCreated},
13   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryModified},
14   * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryRemoved},
15   * {@link org.infinispan.client.hotrod.annotation.ClientCacheFailover}
16   *
17   * @author Galder Zamarre√±o
18   */
19  @Retention(RetentionPolicy.RUNTIME)
20  @Target(ElementType.TYPE)
21  public @interface ClientListener {
22  
23     /**
24      * Defines the key/value filter factory for this client listener. Filter
25      * factories create filters that help decide which events should be sent
26      * to this client listener. This helps with reducing traffic from server
27      * to client. By default, no filtering is applied.
28      */
29     String filterFactoryName() default "";
30  
31     /**
32      * Defines the converter factory for this client listener. Converter
33      * factories create event converters that customize the contents of the
34      * events sent to this listener. When event customization is enabled,
35      * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryCreated},
36      * {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryModified},
37      * and {@link org.infinispan.client.hotrod.annotation.ClientCacheEntryRemoved}
38      * callbacks receive {@link org.infinispan.client.hotrod.event.ClientCacheEntryCustomEvent}
39      * instances as parameters instead of their corresponding create/modified/removed
40      * event. Event customization helps reduce the payload of events, or
41      * increase to send even more information back to the client listener.
42      * By default, no event customization is applied.
43      */
44     String converterFactoryName() default "";
45  
46     /**
47      * This option affects the type of the parameters received by a configured
48      * filter and/or converter. If using raw data, filter and/or converter
49      * classes receive raw binary arrays as parameters instead of unmarshalled
50      * instances, which is the default. On top of that, when raw data is
51      * enabled, custom events produced by the converters are expected to be
52      * byte arrays. This option is useful when trying to avoid marshalling
53      * costs involved in unmarshalling data to pass to filter/converter
54      * callbacks or costs involved in marshalling custom event POJOs.
55      * Using raw data also helps with potential classloading issues related to
56      * loading callback parameter classes or custom event POJOs. By using raw
57      * data, there's no need for class sharing between the server and client.
58      * By default, using raw binary data for filter/converter callbacks is
59      * disabled.
60      */
61     boolean useRawData() default false;
62  
63     /**
64      * This flag enables cached state to be sent back to remote clients when
65      * either adding a cache listener for the first time, or when the node where
66      * a remote listener is registered changes. When enabled, state is sent
67      * back as cache entry created events to the clients. In the special case
68      * that the node where the remote listener is registered changes, before
69      * sending any cache entry created events, the client receives a failover
70      * event so that it's aware of the change of node. This is useful in order
71      * to do local clean up before receiving the state again. For example, a
72      * client building a local near cache and keeping it up to date with remote
73      * events might decide to clear in when the failover event is received and
74      * before the state is received.
75      *
76      * If disabled, no state is sent back to the client when adding a listener,
77      * nor it gets state when the node where the listener is registered changes.
78      *
79      * By default, including state is disabled in order to provide best
80      * performance. If clients must receive all events, enable including state.
81      */
82     boolean includeCurrentState() default false;
83  
84  }
